[1547. Minimum Cost to Cut a Stick](https://leetcode.com/problems/minimum-cost-to-cut-a-stick/)

Given a wooden stick of length `n` units. The stick is labelled from `0` to `n`. For example, a stick of length **6** is labelled as follows:

![](https://assets.leetcode.com/uploads/2020/07/21/statement.jpg)

Given an integer array `cuts` where `cuts[i]` denotes a position you should perform a cut at.

You should perform the cuts in order, you can change the order of the cuts as you wish.

The cost of one cut is the length of the stick to be cut, the total cost is the sum of costs of all cuts. When you cut a stick, it will be split into two smaller sticks (i.e. the sum of their lengths is the length of the stick before the cut). Please refer to the first example for a better explanation.

Return _the minimum total cost_ of the cuts.

**Example 1:**

![](https://assets.leetcode.com/uploads/2020/07/23/e1.jpg)

**Input:** n = 7, cuts = [1,3,4,5]
**Output:** 16
**Explanation:** Using cuts order = [1, 3, 4, 5] as in the input leads to the following scenario:
![](https://assets.leetcode.com/uploads/2020/07/21/e11.jpg)
The first cut is done to a rod of length 7 so the cost is 7. The second cut is done to a rod of length 6 (i.e. the second part of the first cut), the third is done to a rod of length 4 and the last cut is to a rod of length 3. The total cost is 7 + 6 + 4 + 3 = 20.
Rearranging the cuts to be [3, 5, 1, 4] for example will lead to a scenario with total cost = 16 (as shown in the example photo 7 + 4 + 3 + 2 = 16).

**Example 2:**

**Input:** n = 9, cuts = [5,6,1,4,2]
**Output:** 22
**Explanation:** If you try the given cuts ordering the cost will be 25.
There are much ordering with total cost <= 25, for example, the order [4, 6, 5, 2, 1] has total cost = 22 which is the minimum possible.

**Constraints:**

- `2 <= n <= 106`
- `1 <= cuts.length <= min(n - 1, 100)`
- `1 <= cuts[i] <= n - 1`
- All the integers in `cuts` array are **distinct**.

# Using Recursion
TLE
```java
class Solution {
    private int helper(int cuts[], int i, int j, int st, int end) {
        if (i > j)return 0;
        int res = Integer.MAX_VALUE;
        for (int k = i; k <= j; k++){
            int cost = end - st;
            res = Math.min(res, cost + helper(cuts, i, k-1, st, cuts[k]) + helper(cuts, k+1, j, cuts[k], end));
        }
        return res;
    }
    public int minCost(int n, int[] cuts) {
        int len = cuts.length;
        Arrays.sort(cuts);
        return helper(cuts, 0, len-1, 0, n);
    }
}
```

# Memoization

```java
class Solution {
    int memo[][];
    private int helper(int cuts[], int i, int j, int st, int end) {
        if (i > j)return 0;
        if (memo[st][end] != -1)return memo[st][end];
        int res = Integer.MAX_VALUE;
        for (int k = i; k <= j; k++){
            int cost = end - st;
            res = Math.min(res, cost + helper(cuts, i, k-1, st, cuts[k]) + helper(cuts, k+1, j, cuts[k], end));
        }
        return memo[st][end] = res;
    }
    public int minCost(int n, int[] cuts) {
        int len = cuts.length;
        Arrays.sort(cuts);
        memo = new int[n+1][n+1];
        for (int i = 0; i <= n; i++) {
            Arrays.fill(memo[i],-1);
        }
        return helper(cuts, 0, len-1, 0, n);
    }
}
```

```java
class Solution {
    private int[] extendCuts(int[] cuts, int len) {
        int n = cuts.length;
        int[] newCuts = new int[n + 2];

        newCuts[0] = 0;
        for (int i = 0; i < n; i++) {
            newCuts[i + 1] = cuts[i];
        }
        newCuts[n + 1] = len;

        return newCuts;
    }  
    int[][] memo;
    private int helper(int[] cuts,int i, int j) { 
        if (i+1 == j) return 0;
        if (memo[i][j] != -1) return memo[i][j];
        int min_cost = (int) 1e9;
        for (int k = i+1; k < j; k++) {
            int cost = cuts[j] - cuts[i] + helper(cuts, i, k) + helper(cuts, k, j);
            min_cost = Math.min(min_cost, cost);
        }
        return memo[i][j] = min_cost;
    }
    public int minCost(int n, int[] cuts) {
        cuts = extendCuts(cuts, n);
        Arrays.sort(cuts);
        int c = cuts.length;
        memo = new int[c][c];
        for (int i = 0; i < c; i++) {
            Arrays.fill(memo[i], -1);
        }
        return helper(cuts, 0, c-1);
    }
}
```

```java
class Solution {
    int[][] memo;
    private int helper(int[] cuts, int i, int j, int x, int y) {
        if (x > y) return 0;
        if(memo[x][y] != -1) return memo[x][y];
        int minCost = (int) 1e9;
        for (int k = x; k <= y; k++) {
            int cut = cuts[k];
            int cost = j-i + helper(cuts, i, cut, x, k-1) + helper(cuts, cut, j, k+1, y);
            minCost = Math.min(minCost, cost);
        }
        return memo[x][y] = minCost;
    }
    public int minCost(int n, int[] cuts) {
        memo = new int[cuts.length][cuts.length];
        for (int i = 0; i < cuts.length; i++) {
            Arrays.fill(memo[i], -1);
        }
        Arrays.sort(cuts);
        return helper(cuts, 0, n, 0, cuts.length-1);
    }
}
```

# Tabulation

```java
class Solution {
    public int minCost(int n, int[] cuts) {
        int len = cuts.length;
        int allCuts[] = new int[len + 2];
        allCuts[len+1] = n;
        for (int i = 0; i < len; i++){
            allCuts[i+1] = cuts[i];
        }
        Arrays.sort(allCuts);
        int dp[][]  = new int[len+2][len+2];
        for (int i = len; i >= 1; i--){
            for (int j = 1; j <= len; j++) {
                int res = Integer.MAX_VALUE;
                for (int k = i; k <= j; k++){
                    int cost = allCuts[j+1] - allCuts[i-1];
                    res = Math.min(res, cost + dp[i][k-1] + dp[k+1][j]);
                }
                dp[i][j] = (res == Integer.MAX_VALUE) ? 0 : res;
            }
        }
        return dp[1][len];
    }
}
```

```java
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int c = cuts.length;
        int[] allCuts = new int[c + 2];
        for (int i = 0; i < c; i++) {
            allCuts[i+1] = cuts[i];
        }
        c = c + 2;
        allCuts[c-1] = n;
        int[][] dp = new int[c][c];

        for (int i = 0; i < c; i++) {
            Arrays.fill(dp[i], (int) 1e9);
            dp[i][i] = 0;
            if (i+1 < c) dp[i][i+1] = 0;
        }
        for (int i = c-2; i >= 0; i--) {
            for (int j = i+1; j < c; j++) {
                for (int k = i; k < j; k++) {
                    dp[i][j] = Math.min(dp[i][j], dp[i][k] + dp[k][j] + (allCuts[j] - allCuts[i]));
                }
            }
        }
        return dp[0][c-1];
    }
}
```

```java
class Solution {
    public int minCost(int n, int[] cuts) {
        Arrays.sort(cuts);
        int c = cuts.length;
        int[] allcuts = new int[c+2];
        for (int i = 0; i < c; i++) {
            allcuts[i+1] = cuts[i];
        }
        c+=2;
        allcuts[c-1] = n;
        int[][] dp = new int[c][c];
        for (int i = 0; i < c; i++) {
            Arrays.fill(dp[i], (int) 1e9);
            dp[i][i] = 0;
            if (i+1 < c) dp[i][i+1] = 0;
        }
        
        for (int j = 2; j < c; j++) {
            for (int i = j-2; i >= 0; i--) {
                for (int k = i+1; k < j; k++) {
                    int cost = allcuts[j]-allcuts[i] + dp[i][k] + dp[k][j];
                    dp[i][j] = Math.min(dp[i][j], cost);
                }
            }
        }

        return dp[0][c-1];
    }
}
```

[2327. Number of People Aware of a Secret](https://leetcode.com/problems/number-of-people-aware-of-a-secret/)

On day `1`, one person discovers a secret.

You are given an integer `delay`, which means that each person will **share** the secret with a new person **every day**, starting from `delay` days after discovering the secret. You are also given an integer `forget`, which means that each person will **forget** the secret `forget` days after discovering it. A person **cannot** share the secret on the same day they forgot it, or on any day afterwards.

Given an integer `n`, return _the number of people who know the secret at the end of day_ `n`. Since the answer may be very large, return it **modulo** `109 + 7`.

**Example 1:**

**Input:** n = 6, delay = 2, forget = 4
**Output:** 5
**Explanation:**
Day 1: Suppose the first person is named A. (1 person)
Day 2: A is the only person who knows the secret. (1 person)
Day 3: A shares the secret with a new person, B. (2 people)
Day 4: A shares the secret with a new person, C. (3 people)
Day 5: A forgets the secret, and B shares the secret with a new person, D. (3 people)
Day 6: B shares the secret with E, and C shares the secret with F. (5 people)

**Example 2:**

**Input:** n = 4, delay = 1, forget = 3
**Output:** 6
**Explanation:**
Day 1: The first person is named A. (1 person)
Day 2: A shares the secret with B. (2 people)
Day 3: A and B share the secret with 2 new people, C and D. (4 people)
Day 4: A forgets the secret. B, C, and D share the secret with 3 new people. (6 people)

**Constraints:**

- `2 <= n <= 1000`
- `1 <= delay < forget <= n`


# Using DP

```java
class Solution {
    public int peopleAwareOfSecret(int n, int delay, int forget) {
        int[] keeper = new int[n+1];
        int[] blabbermouth = new int[n+1];
        blabbermouth[1] = 1;
        int mod = (int) 1e9 + 7;
        for (int day = 1; day <= n; day++) {
            int newKeeper = blabbermouth[day];
            int rememberUpto = Math.min(day + forget - 1, n);
            for (int remDay = day; remDay <= rememberUpto; remDay++) {
                keeper[remDay] = (keeper[remDay] + newKeeper) % mod;
            }
            for (int shareDay = day + delay; shareDay <= rememberUpto; shareDay++) {
                blabbermouth[shareDay] = (blabbermouth[shareDay] + newKeeper) % mod;
            }
        }
        return keeper[n];
    }
}
```

```java
class Solution {
    public int peopleAwareOfSecret(int n, int delay, int forget) {
        int MOD = (int)1e9 + 7;
        long keeper[] = new long[n+1];
        long sharer[] = new long[n+1];

        sharer[1] = 1;
        sharer[2] = -1;

        for (int day = 1; day <= n; day++) {
            keeper[day] = (keeper[day] + keeper[day - 1]) % MOD;
            long curSharer = (sharer[day] + sharer[day - 1]) % MOD;
            sharer[day] = curSharer;

            keeper[day] = (keeper[day] + curSharer) % MOD;
            if (day + forget <= n) {
                keeper[day + forget] = (keeper[day + forget] - curSharer + MOD) % MOD;
                sharer[day + forget] = (sharer[day + forget] - curSharer + MOD)  % MOD;
            }

            if (day + delay <= n) sharer[day + delay] = (sharer[day + delay] + curSharer) % MOD;
        }
        return (int)keeper[n];
    }
}
```

# Optimal

```java
class Solution {
    public int peopleAwareOfSecret(int n, int delay, int forget) {
        int dp[] = new int[n+1];
        dp[1] = 1;
        int MOD = (int) 1e9 + 7;
        int sharers = 0;
        for (int day = 2; day <= n; day++) {
            if (day - delay >= 1) sharers = (sharers + dp[day - delay]) % MOD;
            if (day - forget >= 1) sharers = (sharers - dp[day - forget] + MOD) % MOD;
            dp[day] = sharers;
        }

        long res = 0;
        for (int day = n - forget + 1; day <= n; day++) {
            res = (res + dp[day]) % MOD;
        }
        return (int) res;
    }
}
```

# Using Priority Queue


### Brute
```java
class Solution {
    class BlabberMouth{
        int day;
        int count;
        BlabberMouth(int day, int count) {
            this.day = day;
            this.count = count;
        }
    }
    public int peopleAwareOfSecret(int n, int delay, int forget) {
        int keeper = 0;
        PriorityQueue<BlabberMouth> pq = new PriorityQueue<>((a, b) -> a.day - b.day);
        HashMap<Integer, BlabberMouth> hm = new HashMap<>();
        BlabberMouth src = new BlabberMouth(1, 1);
        pq.offer(src);
        hm.put(1, src);
        int mod = (int)1e9 + 7;
        while (!pq.isEmpty() && pq.peek().day <= n) {
            BlabberMouth cur = pq.poll();
            if (cur.day + forget - 1 >= n) {
                keeper = (keeper + cur.count) % mod;
            }
            int lastDay = Math.min(cur.day + forget - 1, n);
            for (int day = cur.day + delay; day <= lastDay; day++) {
                if (!hm.containsKey(day)) {
                    BlabberMouth ob = new BlabberMouth(day, cur.count);
                    pq.offer(ob);
                    hm.put(day, ob);
                } else {
                    BlabberMouth ob = hm.get(day);
                    ob.count = (ob.count + cur.count) % mod;
                }
            }
        }
        return keeper;
    }
}
```

```java
class Solution {
    class Event{
        int day;
        long count;
        Event(int day, long count) {
            this.day = day;
            this.count = count;
        }
    }
    public int peopleAwareOfSecret(int n, int delay, int forget) {  
        int mod = (int)1e9 + 7;
        PriorityQueue<Event> pq = new PriorityQueue<>((a, b) -> a.day - b.day);
        pq.offer(new Event(1 + delay, 1));
        pq.offer(new Event(1 + forget, -1));
        long sharersOnIthDay[] = new long[n+1];
        sharersOnIthDay[1] = 1;
        long sharers = 0;
        for (int day = 2; day <= n; day++) {
            while (!pq.isEmpty() && pq.peek().day == day) {
                sharers = (sharers + pq.poll().count + mod) % mod;
            }
            sharersOnIthDay[day] = sharers;
            if (sharers > 0) {
                if (day + delay <= n) pq.offer(new Event(day + delay, sharers));
                if (day + forget <= n) pq.offer(new Event(day + forget, -sharers));
            }
        }
        long totalSharersAtN = 0;
        for (int day = n - forget + 1; day <= n; day++) {
            if(day >= 1) totalSharersAtN = (totalSharersAtN + sharersOnIthDay[day]) % mod;
        }
        return (int)totalSharersAtN;
    }
}
```

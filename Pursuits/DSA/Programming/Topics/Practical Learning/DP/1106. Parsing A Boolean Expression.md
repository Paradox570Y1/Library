[1106. Parsing A Boolean Expression](https://leetcode.com/problems/parsing-a-boolean-expression/)


A **boolean expression** is an expression that evaluates to either `true` or `false`. It can be in one of the following shapes:

- `'t'` that evaluates to `true`.
- `'f'` that evaluates to `false`.
- `'!(subExpr)'` that evaluates to **the logical NOT** of the inner expression `subExpr`.
- `'&(subExpr1, subExpr2, ..., subExprn)'` that evaluates to **the logical AND** of the inner expressions `subExpr1, subExpr2, ..., subExprn` where `n >= 1`.
- `'|(subExpr1, subExpr2, ..., subExprn)'` that evaluates to **the logical OR** of the inner expressions `subExpr1, subExpr2, ..., subExprn` where `n >= 1`.

Given a string `expression` that represents a **boolean expression**, return _the evaluation of that expression_.

It is **guaranteed** that the given expression is valid and follows the given rules.

**Example 1:**

**Input:** expression = "&(|(f))"
**Output:** false
**Explanation:** 
First, evaluate |(f) --> f. The expression is now "&(f)".
Then, evaluate &(f) --> f. The expression is now "f".
Finally, return false.

**Example 2:**

**Input:** expression = "|(f,f,f,t)"
**Output:** true
**Explanation:** The evaluation of (false OR false OR false OR true) is true.

**Example 3:**

**Input:** expression = "!(&(f,t))"
**Output:** true
**Explanation:** 
First, evaluate &(f,t) --> (false AND true) --> false --> f. The expression is now "!(f)".
Then, evaluate !(f) --> NOT false --> true. We return true.

**Constraints:**

- `1 <= expression.length <= 2 * 104`
- expression[i] is one following characters: `'('`, `')'`, `'&'`, `'|'`, `'!'`, `'t'`, `'f'`, and `','`.



# Using Stack

#### Easy implementation
```java
class Solution {
    private Character performOperation(List<Character> values, char op) {
        if (op == '!') {
            return (values.get(0) == 't')? 'f' : 't';
        }
        else if(op == '&') {
            for(char val:values) {
                if (val == 'f')return 'f';
            }
            return 't';
        }
        else{
            for(char val:values) {
                if (val == 't')return 't';
            }
            return 'f';
        }
    }
    public boolean parseBoolExpr(String expression) {
        Stack<Character> operator = new Stack<>();
        Stack<Character> data = new Stack<>();
        for (char ele : expression.toCharArray()) {
            if (ele == '&' || ele == '|' || ele == '!') {
                operator.push(ele);
            }
            else if(ele == '(' || ele == 't' || ele == 'f') {
                data.push(ele);
            }
            else if(ele == ')') {
                char op = operator.pop();
                List<Character> values = new ArrayList<>();
                while(data.peek() != '(') {
                    values.add(data.pop());
                }
                data.pop();
                data.push(performOperation(values,op));
            }
        }
        return (data.peek() == 't')? true : false;
    }
}
```

```java
import java.util.Stack;

class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Character> stack = new Stack<>();
        
        for (char c : expression.toCharArray()) {
            if (c == ')') {
                // Process the expression until we find '('
                boolean hasTrue = false;
                boolean hasFalse = false;
                
                // Collect all values until '('
                while (stack.peek() != '(') {
                    char value = stack.pop();
                    if (value == 't') hasTrue = true;
                    if (value == 'f') hasFalse = true;
                }
                
                stack.pop(); // remove '('
                char operator = stack.pop(); // get the operator
                
                char result;
                if (operator == '&') {
                    result = hasFalse ? 'f' : 't';
                } else if (operator == '|') {
                    result = hasTrue ? 't' : 'f';
                } else { // '!'
                    result = hasTrue ? 'f' : 't';
                }
                
                stack.push(result);
            } else if (c != ',') {
                stack.push(c);
            }
        }
        
        return stack.pop() == 't';
    }
}
```

```java
class Solution {
    public boolean parseBoolExpr(String expression) {
        int n = expression.length();
        char buffer[] = new char[n+1];
        int top = 0;
        for (char c : expression.toCharArray()) {
            if (c == ')') {
                boolean hasTrue = false;
                boolean hasFalse = false;

                while(buffer[top-1] != '(') {
                    if (buffer[top-1] == 't') hasTrue = true;
                    if (buffer[top-1] == 'f') hasFalse = true;
                    top--;
                }
                top--;
                
                char op = buffer[--top];
                char res;
                if (op == '&') {
                    res = hasFalse ? 'f':'t';
                }
                else if(op == '|') {
                    res = hasTrue ? 't':'f';
                } else {
                    res = hasTrue ? 'f':'t';
                }
                buffer[top++] = res;
            }
            else if (c != ',') {
                buffer[top++] = c;
            }
        }
        return (buffer[top-1] == 't');
    }
}
```

```java
class Solution {
    public boolean parseBoolExpr(String expression) {
        Stack<Character> st = new Stack();
        for (int i = expression.length()-1; i >= 0; i--) {
            char ch = expression.charAt(i);
            if (ch == '(' || ch == ',') continue;
            if (ch != '!' && ch != '|' && ch != '&') {
                st.push(ch);
                continue;
            }
            boolean res;
            char val = st.pop();
            res = eval(val);
            if (ch == '&') {
                while (!st.isEmpty() && st.peek() != ')') {
                    res &= eval(st.pop());
                }
            } else if(ch == '|') {
                while (!st.isEmpty() && st.peek() != ')') {
                    res |= eval(st.pop());
                }
            } else res = !res;
            st.pop();
            st.push((res)?'t':'f');
        }
        return eval(st.peek());
    }
    private boolean eval(char ch) {
        if (ch == 't') return true;
        return false;
    }
}
```

# Optimal
Using Recursion

```java
class Solution {
    int idx;
    private boolean parse(String s) {
        char ch = s.charAt(idx);
        if (ch == 't') {
            idx++;
            return true;
        }
        if (ch == 'f') {
            idx++;
            return false;
        }
        idx+=2;
        if (ch == '!') {
            boolean val = parse(s);
            idx++;
            return !val;
        }
        boolean res = (ch == '&');
        while (true) {
            boolean val = parse(s);
            if(ch == '&') res &= val;
            else res |= val;

            if (s.charAt(idx) == ')') {
                idx++;
                return res;
            }
            idx++;
        }
    }
    public boolean parseBoolExpr(String expression) {
        idx = 0;
        return parse(expression);
    }
}
```

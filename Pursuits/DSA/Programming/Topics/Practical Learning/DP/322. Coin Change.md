[322. Coin Change](https://leetcode.com/problems/coin-change/)

You are given an integer array `coins` representing coins of different denominations and an integer `amount` representing a total amount of money.

Return _the fewest number of coins that you need to make up that amount_. If that amount of money cannot be made up by any combination of the coins, return `-1`.

You may assume that you have an infinite number of each kind of coin.

**Example 1:**

**Input:** coins = [1,2,5], amount = 11
**Output:** 3
**Explanation:** 11 = 5 + 5 + 1

**Example 2:**

**Input:** coins = [2], amount = 3
**Output:** -1

**Example 3:**

**Input:** coins = [1], amount = 0
**Output:** 0

**Constraints:**

- `1 <= coins.length <= 12`
- `1 <= coins[i] <= 231 - 1`
- `0 <= amount <= 104`

# Using memoization


```java
class Solution {
    private int helper(int[] coins,int amt,int i,int[][] memo){
        if(i==0){
            if(amt%coins[0]==0){
                return amt/coins[0];
            }
            return (int)1e9;
        }
        if(memo[i][amt]!=-1)return memo[i][amt];
        int notpick = 0 + helper(coins,amt,i-1,memo);
        int pick = Integer.MAX_VALUE;
        if(amt>=coins[i])pick = 1 + helper(coins,amt-coins[i],i,memo);
        return memo[i][amt] = Math.min(pick,notpick);
    }
    public int coinChange(int[] coins, int amount) {
        if(amount == 0)return 0;
        int n = coins.length;
        int memo[][] = new int[n][amount+1];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-1);
        }
        int c =  helper(coins,amount,n-1,memo);
        if(c>=(int)1e9)return -1;
        return c;
    }
}
```

```java
class Solution {
    private int helper(int[] coins,int amt,int i,int[][] memo){
        if(amt<0)return -1;
        if(i==0){
            if(amt%coins[i] == 0 )return amt/coins[i];
            return -1;

        int notPick = helper(coins,amt,i-1,memo);
        int res = -1;
        if(notPick!=-1 && pick!=-1)res = Math.min(1+pick,notPick);
        else if(pick!=-1)res = pick+1;
        else if(notPick!=-1)res = notPick;
        return memo[i][amt] = res;
    }
    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int memo[][] = new int[n][amount+1];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-2);
        }
        return helper(coins,amount,n-1,memo);
    }
}
```

```java
class Solution {
    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int memo[][] = new int[n][amount+1];
        for(int i = 0; i < n; i++) {
            Arrays.fill(memo[i], -1);
        }
        int res = helper(coins, amount, n-1, memo);
        if (res == (int) 1e9) return -1;
        return res;
    }
    private int helper(int[] coins, int T, int i, int[][] memo) {
        if(T == 0) return 0;
        if(T < 0 || i < 0) return (int) 1e9;
        if (memo[i][T] != -1) return memo[i][T];
        int nt = helper(coins, T, i-1, memo);
        int t = 1 + helper(coins, T-coins[i], i, memo);
        return memo[i][T] = Math.min(t, nt);
    }
}
```

# Using tabulation

```java
class Solution {
    public int coinChange(int[] coins, int amount) {
        int n = coins.length;
        int dp[][] = new int[n][amount+1];
        
        for(int T=0;T<=amount;T++){
            if(T % coins[0] == 0)dp[0][T] = T/coins[0];
            else dp[0][T] = (int)1e9;
        }
        
        for(int i=1;i<n;i++){
            for(int T=0;T<=amount;T++){
                int notpick = 0 + dp[i-1][T];
                int pick = Integer.MAX_VALUE;
                if(coins[i]<=T)pick = 1 + dp[i][T-coins[i]];
                dp[i][T] = Math.min(pick,notpick);
            }
        }
        if(dp[n-1][amount]>=(int)1e9)return -1;
        return dp[n-1][amount];
    }
}
```

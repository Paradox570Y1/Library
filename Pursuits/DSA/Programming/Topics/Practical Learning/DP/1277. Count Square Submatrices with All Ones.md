[1277. Count Square Submatrices with All Ones](https://leetcode.com/problems/count-square-submatrices-with-all-ones/)

Given a `m * n` matrix of ones and zeros, return how many **square** submatrices have all ones.

**Example 1:**

**Input:** matrix =
[
  [0,1,1,1],
  [1,1,1,1],
  [0,1,1,1]
]
**Output:** 15
**Explanation:** 
There are **10** squares of side 1.
There are **4** squares of side 2.
There is  **1** square of side 3.
Total number of squares = 10 + 4 + 1 = **15**.

**Example 2:**

**Input:** matrix = 
[
  [1,0,1],
  [1,1,0],
  [1,1,0]
]
**Output:** 7
**Explanation:** 
There are **6** squares of side 1.  
There is **1** square of side 2. 
Total number of squares = 6 + 1 = **7**.

**Constraints:**

- `1 <= arr.length <= 300`
- `1 <= arr[0].length <= 300`
- `0 <= arr[i][j] <= 1`

```java
class Solution {
    private int cntSquares(int size) {
        int res = 0;
        for (int i = 1; i <= size; i++) {
            res += i * i;
        }
        return res;
    }
    private int helper(int[][] matrix, int i, int j, int x, int y) {
        int row = matrix.length, col = matrix[0].length;
        if (i >= row || j >= col) return 0;
        for (int k = y; k <= j; k++) {
            if (matrix[i][k] == 0) return 0;
        }
        for (int k = x; k <= i; k++ ) {
            if (matrix[k][j] == 0) return 0;
        }
        for (int k = x; k <= i; k++ ) {
            matrix[k][j] = -1;
        }
        for (int k = y; k <= j; k++) {
            matrix[i][k] = -1;
        }
        return 1 + helper(matrix, i+1, j+1, x, y);
    }
    public int countSquares(int[][] matrix) {
        int row = matrix.length;
        int col = matrix[0].length;
        int res = 0;
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                if (matrix[i][j] == 1) {
                    int squareSize = helper(matrix, i, j, i, j);
                    res += cntSquares(squareSize);
                }
            }
        }
        return res;
    }
}
```

# Brute

```java
class Solution {
    public int countSquares(int[][] matrix) {
        int rows = matrix.length;
        int cols = matrix[0].length;
        int dp[][] = new int[rows][cols];
        int res = 0;
        for (int i = 0; i < rows; i++) {
            dp[i][0] = matrix[i][0];
            res += dp[i][0];
        }
        for (int j = 1; j < cols; j++) {
            dp[0][j] = matrix[0][j];
            res += dp[0][j];
        }
        
        for (int i = 1; i < rows; i++) {
            for (int j = 1; j < cols; j++) {
                if (matrix[i][j] == 1) {
                    int mini = Math.min(dp[i-1][j], dp[i][j-1]);
                    mini = Math.min(mini, dp[i-1][j-1]);
                    if (mini != 0) {
                        dp[i][j] = mini + 1;
                    } else {
                        dp[i][j] = 1;
                    }
                }
                res += dp[i][j];
                System.out.print(dp[i][j]+ " ");
            }
            System.out.println();
        }
        return res;
    }
}
```

[309. Best Time to Buy and Sell Stock with Cooldown](https://leetcode.com/problems/best-time-to-buy-and-sell-stock-with-cooldown/)

You are given an array `prices` where `prices[i]` is the price of a given stock on the `ith` day.

Find the maximum profit you can achieve. You may complete as many transactions as you like (i.e., buy one and sell one share of the stock multiple times) with the following restrictions:

- After you sell your stock, you cannot buy stock on the next day (i.e., cooldown one day).

**Note:** You may not engage in multiple transactions simultaneously (i.e., you must sell the stock before you buy again).

**Example 1:**

**Input:** prices = [1,2,3,0,2]
**Output:** 3
**Explanation:** transactions = [buy, sell, cooldown, buy, sell]

**Example 2:**

**Input:** prices = [1]
**Output:** 0

**Constraints:**

- `1 <= prices.length <= 5000`
- `0 <= prices[i] <= 1000`


# Using Recursion

```java
class Solution {
    private int profit(int prices[],int i,int buy){
        if(i>=prices.length)return 0;
        int profit=0;
        if(buy==1){
            int b = -prices[i] + profit(prices,i+1,0);
            int nb = profit(prices,i+1,1);
            profit = Math.max(b,nb);
        } else {
            int s = prices[i] + profit(prices,i+2,1);
            int ns = profit(prices,i+1,0);
            profit = Math.max(s,ns);
        }
        return profit;
    }
    public int maxProfit(int[] prices) {
        return profit(prices,0,1);
    }
}
```

# Memoization

```java
class Solution {
    private int helper(int[] prices, int i, int c, int b, int[][][] memo) {
        if (i == prices.length) return 0;
        if (memo[i][c][b] != -1) return memo[i][c][b];
        int res = 0;
        if (b == 1) {
            if (c==0) {
                int w1 = helper(prices, i+1, c, 1, memo);
                int w2 = -prices[i] + helper(prices, i+1, c, 0, memo);
                res = Math.max(w1, w2);
            } else {
                return memo[i][c][b] = helper(prices, i+1, 0, 1, memo);
            }
        } else {
            int w1 = prices[i] + helper(prices, i+1, 1, 1, memo);
            int w2 = helper(prices, i+1, 0, 0, memo);
            res = Math.max(w1, w2);
        }
        return memo[i][c][b] = res;
    }
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int memo[][][] = new int[n][2][2];
        for (int i = 0; i < n ;i++) {
            for (int j = 0; j < 2; j++) {
                memo[i][j][0] = -1;
                memo[i][j][1] = -1;
            }
        }
        return helper(prices, 0, 0, 1, memo);
    }
}
```

# DP

```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int dp[][] = new int[n+2][2];
        for(int i=n-1;i>=0;i--){
            int sell = prices[i] + dp[i+2][0];
            int notSell = dp[i+1][1];
            dp[i][1] = Math.max(sell,notSell);
            int buy = -prices[i] + dp[i+1][1];
            int notBuy = dp[i+1][0];
            dp[i][0] = Math.max(buy,notBuy);
        }
        return dp[0][0];
    }
}
```

```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int prev[] = new int[2];
        int prev2[] = new int[2];
        for(int i=n-1;i>=0;i--){
            int cur[] = new int[2];
            int sell = prices[i] + prev2[0];
            int notSell = prev[1];
            cur[1] = Math.max(sell,notSell);
            int buy = -prices[i] + prev[1];
            int notBuy = prev[0];
            cur[0] = Math.max(buy,notBuy);
            prev2 = prev;
            prev = cur;
        }
        return prev[0];
    }
}
```

```java
class Solution {

        int n = prices.length;
        int[] t1 = new int[2];
        int[] t2 = new int[2];
        for (int i = n-1; i >= 0; i--){
            int[] t0 = new int[2];
            t0[0] = Math.max(t1[0], t1[1] - prices[i]);
            t0[1] = Math.max(t1[1], t2[0] + prices[i]);
            t2 = t1;
            t1 = t0;
        }
        return t1[0];
    }
}
```

```java
class Solution {
    public int maxProfit(int[] prices) {
        int pb = Integer.MIN_VALUE, ps = 0, p2s = 0;
        for (int price: prices) {
            int b = Math.max(p2s - price, pb);
            int s = Math.max(pb + price, ps);
            p2s = ps;
            pb = b;
            ps = s;
        }
        return ps;
    }
}
```

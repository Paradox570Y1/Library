[516. Longest Palindromic Subsequence](https://leetcode.com/problems/longest-palindromic-subsequence/)

Given a string `s`, find _the longest palindromic **subsequence**'s length in_ `s`.

A **subsequence** is a sequence that can be derived from another sequence by deleting some or no elements without changing the order of the remaining elements.

**Example 1:**

**Input:** s = "bbbab"
**Output:** 4
**Explanation:** One possible longest palindromic subsequence is "bbbb".

**Example 2:**

**Input:** s = "cbbd"
**Output:** 2
**Explanation:** One possible longest palindromic subsequence is "bb".

**Constraints:**

- `1 <= s.length <= 1000`
- `s` consists only of lowercase English letters.

```java
class Solution {
    private boolean isPal(StringBuilder s){
        String cur = s.toString();
        StringBuilder rev = new StringBuilder();
        for(int i=cur.length()-1;i>=0;i--){
            rev.append(s.charAt(i));
        }
        return rev.toString().equals(cur);
    }
    private int helper(String s,int i,StringBuilder sb){
        if(i<0){
            if(isPal(sb)){
                return sb.length();
            }
            return 0;
        }
        int notpick = helper(s,i-1,sb);
        sb.append(s.charAt(i));
        int pick = helper(s,i-1,sb);
        sb.setLength(sb.length()-1);
        return Math.max(pick,notpick);
    }
    public int longestPalindromeSubseq(String s) {
        int n = s.length();
        return helper(s,n-1,new StringBuilder());
    }
}
```

```java
class Solution {
    private String reverseOf(String s){
        StringBuilder sb = new StringBuilder();
        for(int i=s.length()-1;i>=0;i--){
            sb.append(s.charAt(i));
        }
        return sb.toString();
    }
    private int helper(String a,String b,int i,int j){
        if(i==0 || j==0){
            return 0;
        }
        int cnt=0;
        if(a.charAt(i-1) == b.charAt(j-1)){
            cnt = 1 + helper(a,b,i-1,j-1);
        }
        else{
            cnt = Math.max(helper(a,b,i-1,j),helper(a,b,i,j-1));
        }
        return cnt;
    }
    public int longestPalindromeSubseq(String s) {
        int n= s.length();
        return helper(s,reverseOf(s),n,n);
    }
}
```

# Using Memoization

This worsen the time complexity bcz the number of unique combinations to generate a key is exponential and `toString()` method even adds to the time complexity as it takes O(N).

```java
class Solution {
    Map<String,Integer> memo;
    private boolean isPal(String s){
        int i=0;
        int j=s.length()-1;
        while(i<j){
            if(s.charAt(i)!=s.charAt(j))return false;
            i++;j--;
        }
        return true;
    }
    private int helper(String s,StringBuilder sb,int i){
        if(i==0){
            if(isPal(sb.toString()))return sb.length();
            return 0;
        }
        String key = String.valueOf(i)+"|"+sb.toString();
        if(memo.containsKey(key))return memo.get(key);
        sb.append(s.charAt(i-1));
        int pick = helper(s,sb,i-1);
        sb.setLength(sb.length()-1);
        int notPick = helper(s,sb,i-1);
        int res = Math.max(pick,notPick);
        memo.put(key,res);
        return res;
    }
    public int longestPalindromeSubseq(String s) {
        memo = new HashMap<>();
        return helper(s,new StringBuilder(),s.length());
    }
}
```

```java
class Solution {
    private String reverseOf(String s){
        StringBuilder sb = new StringBuilder();
        for(int i=s.length()-1;i>=0;i--){
            sb.append(s.charAt(i));
        }
        return sb.toString();
    }
    private int helper(String a,String b,int i,int j,int[][] memo){
        if(i==0 || j==0){
            return 0;
        }
        if(memo[i][j] != -1)return memo[i][j];
        int cnt=0;
        if(a.charAt(i-1) == b.charAt(j-1)){
            cnt = 1 + helper(a,b,i-1,j-1,memo);
        }
        else{
            cnt = Math.max(helper(a,b,i-1,j,memo),helper(a,b,i,j-1,memo));
        }
        return memo[i][j] = cnt;
    }
    public int longestPalindromeSubseq(String s) {
        int n= s.length();
        int memo[][] = new int[n+1][n+1];
        for(int i=0;i<=n;i++){
            Arrays.fill(memo[i],-1);
        }
        return helper(s,reverseOf(s),n,n,memo);
    }
}
```

# Using DP

```java
class Solution {
    public int longestPalindromeSubseq(String s) {
        int n = s.length();
        int dp[] = new int[n+2];
        for(int i=1;i<=n;i++){
            int cur[] = new int[n+2];
            for(int j=n;j>=1;j--){
                if(s.charAt(i-1) == s.charAt(j-1)){
                    cur[j] = 1 + dp[j+1];
                }
                else cur[j] = Math.max(cur[j+1],dp[j]);
            }
            dp = cur;
        }
        return dp[1];
    }
}
```

```java
class Solution {
    public int longestPalindromeSubseq(String s) {
        int n = s.length();
        int[] dp = new int[n+1];

        for (int i = 1; i <= n; i++) {
            int prev = 0;
            for (int j = 1; j <= n; j++) {
                int b = dp[j];
                if (s.charAt(i-1) == s.charAt(n-j)) {
                    dp[j] = prev + 1;
                } else {
                    dp[j] = Math.max(dp[j-1], dp[j]);
                }
                prev = b;
            }
        }
        return dp[n];
    }
}
```

[72. Edit Distance](https://leetcode.com/problems/edit-distance/)

Given two strings `word1` and `word2`, return _the minimum number of operations required to convert `word1` to `word2`_.

You have the following three operations permitted on a word:

- Insert a character
- Delete a character
- Replace a character

**Example 1:**

**Input:** word1 = "horse", word2 = "ros"
**Output:** 3
**Explanation:** 
horse -> rorse (replace 'h' with 'r')
rorse -> rose (remove 'r')
rose -> ros (remove 'e')

**Example 2:**

**Input:** word1 = "intention", word2 = "execution"
**Output:** 5
**Explanation:** 
intention -> inention (remove 't')
inention -> enention (replace 'i' with 'e')
enention -> exention (replace 'n' with 'x')
exention -> exection (replace 'n' with 'c')
exection -> execution (insert 'u')

**Constraints:**

- `0 <= word1.length, word2.length <= 500`
- `word1` and `word2` consist of lowercase English letters.


# Recursion

```java
class Solution {
    private int helper(String s1,String s2,int i,int j){
        if(j==0)return i;
        if(i==0)return j;
        int res=Integer.MAX_VALUE;
        if(s1.charAt(i-1) != s2.charAt(j-1)){
            int insert = 1 + helper(s1,s2,i,j-1);
            int delete = 1 + helper(s1,s2,i-1,j);
            int replace = 1 + helper(s1,s2,i-1,j-1);
            res = Math.min(res,Math.min(insert,delete));
            res = Math.min(res,replace);
        }
        else{
            res = helper(s1,s2,i-1,j-1);
        }
        return res;
    }
    public int minDistance(String word1, String word2) {
        int n = word1.length();
        int m = word2.length();
        return helper(word1,word2,n,m);
    }
}
```

```java
class Solution {
    int memo[][];
    private int helper(String s1,String s2,int i,int j){
        if(j==0)return i;
        if(i==0)return j;
        if(memo[i][j] != -1)return memo[i][j];
        int res=Integer.MAX_VALUE;
        if(s1.charAt(i-1) != s2.charAt(j-1)){
            int insert = 1 + helper(s1,s2,i,j-1);
            int delete = 1 + helper(s1,s2,i-1,j);
            int replace = 1 + helper(s1,s2,i-1,j-1);
            res = Math.min(res,Math.min(insert,delete));
            res = Math.min(res,replace);
        }
        else{
            res = helper(s1,s2,i-1,j-1);
        }
        return memo[i][j] = res;
    }
    public int minDistance(String word1, String word2) {
        int n = word1.length();
        int m = word2.length();
        memo = new int[n+1][m+1];
        for(int i=0;i<=n;i++){
            Arrays.fill(memo[i],-1);
        }
        return helper(word1,word2,n,m);
    }
}
```

```java
class Solution {
    private int helper(String w1, String w2, int l1, int l2, int[][] memo) {
        if (l2 == 0) return l1;
        if (l1 == 0) return l2;
        if (memo[l1][l2] != -1) return memo[l1][l2];
        int count = (int) 1e9;
        if (w1.charAt(l1-1) == w2.charAt(l2-1)) {
            count = helper(w1, w2, l1-1, l2-1, memo);
        } else {
            int a = helper(w1, w2, l1-1, l2, memo);
            int b = helper(w1, w2, l1, l2-1, memo);
            int c = helper(w1, w2, l1-1, l2-1, memo);
            int min = Math.min(a, Math.min(b,c));
            count = Math.min(count, min+1);
        }
        return memo[l1][l2] = count;
    }
    public int minDistance(String word1, String word2) {
        int memo[][] = new int[word1.length()+1][word2.length()+1];
        for (int i = 0; i <= word1.length(); i++) {
            Arrays.fill(memo[i], -1);
        }
        return helper(word1, word2, word1.length(), word2.length(), memo);
    }
}
```

```java
class Solution {
    private int[][] memo;
    public int minDistance(String word1, String word2) {
        int n = word1.length();
        int m = word2.length();
        memo = new int[n+1][m+1];
        for (int i = 0; i <= n; i++) {
            Arrays.fill(memo[i], -1);
        }
        return helper(word1, word2, n, m);
    }

    private int helper(String word1, String word2, int i, int j) {
        if (i == 0) return j;
        if (j == 0) return i;
        if (memo[i][j] != -1) return memo[i][j];
        int cnt = 0;
        if (word1.charAt(i-1) == word2.charAt(j-1)) {
            cnt = helper(word1, word2, i-1, j-1);
        } else {
            cnt = 1 + Math.min(
                helper(word1, word2, i-1, j-1),
                Math.min(
                    helper(word1, word2, i-1, j),
                    helper(word1, word2, i, j-1)
                )
            );
        }
        return memo[i][j] = cnt;
    }
}
```

# DP

```java
class Solution {
    public int minDistance(String word1, String word2) {
        int n = word1.length();
        int m = word2.length();
        int dp[][] = new int[n+1][m+1];
        for(int i=0;i<=n;i++)dp[i][0] = i;
        for(int j=0;j<=m;j++)dp[0][j] = j;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=m;j++){
                if(word1.charAt(i-1) == word2.charAt(j-1)){
                    dp[i][j] = dp[i-1][j-1]; 
                }
                else dp[i][j] = 1 + Math.min(dp[i-1][j-1],Math.min(dp[i-1][j],dp[i][j-1]));
            }
        }
        return dp[n][m];
    }
}
```

```java
                }
            }
            dp = cur;
        }

    }
```

```java
    public int minDistance(String word1, String word2) {
        int n = word1.length();
        int m = word2.length();

        int[] dp = new int[n+1];
        for (int i = 1; i <= n; i++) {
            dp[i] = i;
        }
        for (int i = 0; i < m; i++) {
            int[] cur = new int[n+1];
            cur[0] = dp[0] + 1;
            for (int j = 0; j < n; j++) {
                if(word1.charAt(j) == word2.charAt(i)) {
                    cur[j+1] = dp[j];
                } else {
                    cur[j+1] = 1 + Math.min(cur[j], Math.min(dp[j], dp[j+1]));
                }
            }
            dp = cur;
        }
        return dp[n];
    }
}
```

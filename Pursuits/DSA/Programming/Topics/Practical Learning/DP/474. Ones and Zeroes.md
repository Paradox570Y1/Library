[474. Ones and Zeroes](https://leetcode.com/problems/ones-and-zeroes/)

You are given an array of binary strings `strs` and two integers `m` and `n`.

Return _the size of the largest subset of `strs` such that there are **at most**_ `m` `0`_'s and_ `n` `1`_'s in the subset_.

A set `x` is a **subset** of a set `y` if all elements of `x` are also elements of `y`.

**Example 1:**

**Input:** strs = ["10","0001","111001","1","0"], m = 5, n = 3
**Output:** 4
**Explanation:** The largest subset with at most 5 0's and 3 1's is {"10", "0001", "1", "0"}, so the answer is 4.
Other valid but smaller subsets include {"0001", "1"} and {"10", "1", "0"}.
{"111001"} is an invalid subset because it contains 4 1's, greater than the maximum of 3.

**Example 2:**

**Input:** strs = ["10","0","1"], m = 1, n = 1
**Output:** 2
**Explanation:** The largest subset is {"0", "1"}, so the answer is 2.

**Constraints:**

- `1 <= strs.length <= 600`
- `1 <= strs[i].length <= 100`
- `strs[i]` consists only of digits `'0'` and `'1'`.
- `1 <= m, n <= 100`


# Ultra Brute
TLE

```java
class Solution {
    private int m,n;
    int res = 0;
    public int findMaxForm(String[] strs, int m, int n) {
        this.m = m;
        this.n = n;
        helper(strs, 0, 0, 0, strs.length-1);
        return res;
    }
    private void helper(String[] strs, int zero, int one, int len, int i) {
        if (i < 0) {
            res = Math.max(res, len);
            return;
        }
        int cur[] = findDigits(strs[i]);
        if (cur[0] > m || cur[1] > n) {
            helper(strs, zero, one, len, i-1);
            return;
        }
        if(cur[0] + zero <= m && cur[1] + one <= n) {
            helper(strs, cur[0] + zero, cur[1] + one, len + 1, i-1);
        }
        helper(strs, zero, one, len, i-1);
    }
    private int[] findDigits(String s) {
        int count = 0;
        for (char c : s.toCharArray()) {
            if (c == '0') count++;
        }
        return new int[]{count, s.length() - count};
    }
}
```

# Brute with memoization

```java
class Solution {
    private int m,n;
    private int memo[][][];
    public int findMaxForm(String[] strs, int m, int n) {
        this.m = m;
        this.n = n;
        memo = new int[m+1][n+1][strs.length];
        for (int i = 0; i <= m; i++) {
            for (int j = 0; j <= n; j++) {
                Arrays.fill(memo[i][j], -1);
            }
        }
        return helper(strs, 0, 0, 0, strs.length-1);
    }
    private int helper(String[] strs, int zero, int one, int len, int i) {
        if (i < 0) {
            return len;
        }
        if (memo[zero][one][i] != -1) return len + memo[zero][one][i];
        int cur[] = findDigits(strs[i]);
        if (cur[0] > m || cur[1] > n) {
            return  helper(strs, zero, one, len, i-1);
        }
        int pick = 0;
        if(cur[0] + zero <= m && cur[1] + one <= n) {
            pick = helper(strs, cur[0] + zero, cur[1] + one, len + 1, i-1);
        }
        int not_pick = helper(strs, zero, one, len, i-1);
        memo[zero][one][i] = Math.max(pick, not_pick) - len;
        return Math.max(pick, not_pick);
    }
    private int[] findDigits(String s) {
        int count = 0;
        for (char c : s.toCharArray()) {
            if (c == '0') count++;
        }
        return new int[]{count, s.length() - count};
    }
}
```

# Do with bottom up DP


```java
class Solution {
    private int[] getCount(String s) {
        int zero = 0;
        for (char ch : s.toCharArray()) {
            if (ch == '0') zero++;
        }
        return new int[]{zero, s.length() - zero};
    }
    public int findMaxForm(String[] strs, int m, int n) {
        int len = strs.length;
        int dp[][] = new int[m+1][n+1];
        for (int i = 0; i <= m; i++) {
            Arrays.fill(dp[i], -1);
        }
        dp[0][0] = 0;
        int res = 0;
        for (int i = 0; i < len; i++) {
            int[] count = getCount(strs[i]);
            int zero = count[0], one = count[1];
            int cur[][] = new int[m+1][n+1];
            for (int t = 0; t <= m; t++) {
                for (int p = 0; p <= n; p++) {
                    cur[t][p] = dp[t][p];
                }
            }
            for (int j = zero; j <= m; j++) {
                for (int k = one; k <= n; k++) {
                    if (dp[j-zero][k-one] != -1)
                    cur[j][k] = Math.max(cur[j][k], dp[j-zero][k-one] + 1);
                    res = Math.max(res, cur[j][k]);
                }
            }
            dp = cur;
        }

        return res;
    }
}
```

```java
class Solution {
    private int[] getCount(String s) {
        int zero = 0;
        for (char ch : s.toCharArray()) {
            if (ch == '0') zero++;
        }
        return new int[]{zero, s.length() - zero};
    }
    public int findMaxForm(String[] strs, int m, int n) {
        int len = strs.length;
        int[][] dp = new int[m+1][n+1];
        for (int k = 0; k < len; k++) {
            int[] count = getCount(strs[k]);
            int zero = count[0], one = count[1];
            for (int i = m; i >= zero; i--) {
                for (int j = n; j >= one; j--) {
                    dp[i][j] = Math.max(dp[i][j], dp[i-zero][j-one] + 1);
                }
            }
        }
        return dp[m][n];
    }
}
```

[1143. Longest Common Subsequence](https://leetcode.com/problems/longest-common-subsequence/)

Given two strings `text1` and `text2`, return _the length of their longest **common subsequence**._ If there is no **common subsequence**, return `0`.

A **subsequence** of a string is a new string generated from the original string with some characters (can be none) deleted without changing the relative order of the remaining characters.

- For example, `"ace"` is a subsequence of `"abcde"`.

A **common subsequence** of two strings is a subsequence that is common to both strings.

**Example 1:**

**Input:** text1 = "abcde", text2 = "ace" 
**Output:** 3  
**Explanation:** The longest common subsequence is "ace" and its length is 3.

**Example 2:**

**Input:** text1 = "abc", text2 = "abc"
**Output:** 3
**Explanation:** The longest common subsequence is "abc" and its length is 3.

**Example 3:**

**Input:** text1 = "abc", text2 = "def"
**Output:** 0
**Explanation:** There is no such common subsequence, so the result is 0.

**Constraints:**

- `1 <= text1.length, text2.length <= 1000`
- `text1` and `text2` consist of only lowercase English characters.


# Using Recursion

```java
class Solution {
    HashSet<String> hs;
    int ans=0;
    private void helper(String s,int i,StringBuilder sb,boolean check){
        if(i<0){
            if(check){
                if(hs.contains(sb.toString())){
                    ans = Math.max(ans,sb.length());
                }
            }
            else hs.add(sb.toString());
            return;
        }
        //not take
        helper(s,i-1,sb,check);
        //take
        sb.append(s.charAt(i));
        helper(s,i-1,sb,check);
        sb.setLength(sb.length()-1);
    }
    public int longestCommonSubsequence(String text1, String text2) {
        hs = new HashSet<>();
        helper(text1,text1.length()-1,new StringBuilder(),false);
        helper(text2,text2.length()-1,new StringBuilder(),true);
        return ans;
    }
}
```

# Using memoization

```java
class Solution {
    private int helper(String a,String b,int i,int j,int memo[][]){
        if(i==0){
            while(j>=0){
                if(a.charAt(0) == b.charAt(j))return 1;
                j--;
            }
            return 0;
        }
        if(j==0){
            while(i>=0){
                if(b.charAt(0) == a.charAt(i))return 1;
                i--;
            }
            return 0;
        }
        if(memo[i][j] != -1)return memo[i][j];
        int cnt=0;
        if(a.charAt(i) ==  b.charAt(j)) cnt = 1 + helper(a,b,i-1,j-1,memo);
        else{
            int c1 = helper(a,b,i-1,j,memo);
            int c2 = helper(a,b,i,j-1,memo);
            cnt = (c1>c2)?c1:c2;
        }
        memo[i][j] = cnt;
        return cnt;
    }
    public int longestCommonSubsequence(String text1, String text2) {
        int n = text1.length();
        int m = text2.length();
        int memo[][] = new int[n+1][m+1];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                memo[i][j] = -1;
            }
        }
        return helper(text1,text2,n-1,m-1,memo);
    }
}
```

```java
        int n = text1.length();
        int m = text2.length();
        int[][] memo = new int[n][m];
        for (int i = 0; i <= n; i++) {

        if (memo[i][j] != -1) return memo[i][j];

            return memo[i][j] = 1 + helper(text1, text2, i-1, j-1, memo);
        }
        return memo[i][j] = Math.max(helper(text1, text2, i-1, j, memo), helper(text1, text2, i, j-1, memo));
    }
```

Without shifting

```java
class Solution {
    public int longestCommonSubsequence(String text1, String text2) {
        int n = text1.length();
        int m = text2.length();
        int dp[][] = new int[n][m];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                if(text1.charAt(i) == text2.charAt(j)){
                    dp[i][j] = 1 + ((i==0 || j==0)?0:dp[i-1][j-1]);
                }
                else{
                    dp[i][j] = Math.max((j==0)?0:dp[i][j-1],(i==0)?0:dp[i-1][j]);
                }
            }
        }
        return dp[n-1][m-1];
    }
}
```

```java
class Solution {
    public int longestCommonSubsequence(String text1, String text2) {
        int n = text1.length();
        int m = text2.length();
        int dp[] = new int[n];
        for (int i = 0; i < m; i++) {
            int[] cur = new int[n];
            cur[0] = (dp[0] == 1)? 1: (text1.charAt(0) == text2.charAt(i))? 1: 0;
            for (int j = 1; j < n; j++) {
                if (text1.charAt(j) == text2.charAt(i)) {
                    cur[j]  = 1 + dp[j-1];
                } else {
                    cur[j] = Math.max(cur[j-1], dp[j]);
                }
            }
            dp = cur;
        }
        return dp[n-1];
    }
}
```

```java
class Solution {
    public int longestCommonSubsequence(String text1, String text2) {
        int n = text1.length();
        int m = text2.length();
        int[] dp = new int[m+1];
        for (int i = 0; i < n; i++) {
            int prev = 0;
            for (int j = 1; j <= m; j++) {
                int buffer = dp[j];
                if (text1.charAt(i) == text2.charAt(j-1)) {
                    dp[j] = 1 + prev;
                } else {
                    dp[j] = Math.max(dp[j-1], dp[j]);
                }
                prev = buffer;
            }
        }
        return dp[m];
    }
}
```

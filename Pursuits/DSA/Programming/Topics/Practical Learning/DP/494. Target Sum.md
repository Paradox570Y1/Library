[494. Target Sum](https://leetcode.com/problems/target-sum/)

You are given an integer array `nums` and an integer `target`.

You want to build an **expression** out of nums by adding one of the symbols `'+'` and `'-'` before each integer in nums and then concatenate all the integers.

- For example, if `nums = [2, 1]`, you can add a `'+'` before `2` and a `'-'` before `1` and concatenate them to build the expression `"+2-1"`.

Return the number of different **expressions** that you can build, which evaluates to `target`.

**Example 1:**

**Input:** nums = [1,1,1,1,1], target = 3
**Output:** 5
**Explanation:** There are 5 ways to assign symbols to make the sum of nums be target 3.
-1 + 1 + 1 + 1 + 1 = 3
+1 - 1 + 1 + 1 + 1 = 3
+1 + 1 - 1 + 1 + 1 = 3
+1 + 1 + 1 - 1 + 1 = 3
+1 + 1 + 1 + 1 - 1 = 3

**Example 2:**

**Input:** nums = [1], target = 1
**Output:** 1

**Constraints:**

- `1 <= nums.length <= 20`
- `0 <= nums[i] <= 1000`
- `0 <= sum(nums[i]) <= 1000`
- `-1000 <= target <= 1000`



# Using Recursion

```java
class Solution {
    private int helper(int[] nums,int i,int target){
        if(i==0){
            if(target==nums[0] && target == -nums[0])return 2;
            if(target==nums[0] || target == -nums[0])return 1;
            return 0;
        }
	        int neg = helper(nums,i-1,target-nums[i]);
        int pos = helper(nums,i-1,target+nums[i]);
        return neg+pos;
    }
    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length;
        return helper(nums,n-1,target);
    }
}
```

```java
class Solution {
    private int helper(int[] nums,int i,int target,int[][] memo){
        if(i==0){
            if(target==nums[0] && target == -nums[0])return 2;
            if(target==nums[0] || target == -nums[0])return 1;
            return 0;
        }
        if(memo[i][target+2000]!=-1)return memo[i][target+2000];
        int neg = helper(nums,i-1,target-nums[i],memo);
        int pos = helper(nums,i-1,target+nums[i],memo);
        return memo[i][target+2000] = neg+pos;
    }
    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length;
        int memo[][] = new int[n][4001];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-1);
        }
        return helper(nums,n-1,target,memo);
    }
}
```

```java
        if(memo[i][t+2000] != -1) return memo[i][t+2000];
        return memo[i][t+2000] = helper(nums, t-nums[i], i-1, memo) + helper(nums, t+nums[i], i-1, memo);
    }
}
```

# Using Tabulation

```java
class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length;
        int total = Arrays.stream(nums).sum();
        if((total+target)%2==1 || total+target<0)return 0;
        int set1 = (total+target)/2;

        int dp[] = new int[set1+1];
        dp[0]=1;
        if(nums[0]<=set1)dp[nums[0]]++;
        for(int i=1;i<n;i++){
            int cur[] = new int[set1+1];
            for(int T=0;T<=set1;T++){
                int notpick = dp[T];
                int pick = 0;
                if(nums[i]<=T)pick = dp[T-nums[i]];
                cur[T] = pick+notpick;
            }
            dp=cur;
        }
        return dp[set1];
    }
}
```

```java
class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int n = nums.length;
        int total_sum = 0;
        for(int ele:nums)total_sum+=ele;
        if(total_sum < Math.abs(target) || (total_sum + target)%2==1)return 0;
        int new_target = (total_sum + target)/2;
        int dp[] = new int[new_target+1];
        dp[0] = 1;
        for(int i=0;i<n;i++){
            int prev[] = new int[new_target+1];
            for(int t=0;t<=new_target;t++){
                int notpick = dp[t];
                int pick = 0;
                if(t>=nums[i])pick = dp[t-nums[i]];
                prev[t] = pick+notpick;
            }
            dp = prev;
        }
        return dp[new_target];
    }
}
```

```java
class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int T = 0;
        for (int num : nums) T += num;
        if (T + target < 0 || (T + target) % 2 == 1) return 0;
        int sum = (T + target) / 2;
        return subsetWithSumK(nums, sum);
    }
    private int subsetWithSumK(int[] nums, int sum) {
        int dp[] = new int[sum+1];
        dp[0] = 1;
        for (int num : nums) {
            int[] cur = new int[sum+1];
            for (int i = 0; i <= sum; i++) {
                cur[i] = dp[i]; // not take
                if (num <= i) cur[i] += dp[i-num];
            }
            dp = cur;
        }
        return dp[sum];
    }
}
```

```java
class Solution {
    public int findTargetSumWays(int[] nums, int target) {
        int T = 0;
        for (int num : nums) T += num;
        if (T + target < 0) return 0;
        else if (target > 0 && T - target < 0) return 0; // not needed as per constraint that target can never exceed Total
        int[] dp = new int[2*T + 1];
        dp[nums[0]+T] = 1;
        dp[-nums[0]+T] += 1;
        int n = nums.length;
        for (int i = 1; i < n; i++) {
            int[] cur = new int[2*T+1];
            for (int j = -T; j <= T; j++) {
                if(dp[j+T] == 0) continue;
                if (j - nums[i] >= -T) {
                    cur[j - nums[i] + T] += dp[j+T]; // Take 1
                }
                if (j + nums[i] <= T) {
                    cur[j + nums[i] + T] += dp[j+T]; // Take 2
                }
            }
            dp = cur;
        }
        return dp[target+T];
    }
}
```

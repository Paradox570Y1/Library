[312. Burst Balloons](https://leetcode.com/problems/burst-balloons/)

You are given `n` balloons, indexed from `0` to `n - 1`. Each balloon is painted with a number on it represented by an array `nums`. You are asked to burst all the balloons.

If you burst the `ith` balloon, you will get `nums[i - 1] * nums[i] * nums[i + 1]` coins. If `i - 1` or `i + 1` goes out of bounds of the array, then treat it as if there is a balloon with a `1` painted on it.

Return _the maximum coins you can collect by bursting the balloons wisely_.

**Example 1:**

**Input:** nums = [3,1,5,8]
**Output:** 167
**Explanation:**
nums = [3,1,5,8] --> [3,5,8] --> [3,8] --> [8] --> []
coins =  3*1*5    +   3*5*8   +  1*3*8  + 1*8*1 = 167

**Example 2:**

**Input:** nums = [1,5]
**Output:** 10

**Constraints:**

- `n == nums.length`
- `1 <= n <= 300`
- `0 <= nums[i] <= 100`


# Recursion

```java
class Solution {
    private int helper(int arr[], int i, int j) {
        if (i > j) return 0;
        int res = -1;
        for (int k = i; k <= j; k++) {
            int cost = arr[i-1] * arr[k] * arr[j+1];
            res = Math.max(res, cost + helper(arr, i, k-1) + helper(arr, k+1, j));
        }
        return res;
    }
    public int maxCoins(int[] nums) {
        int n = nums.length;
        int arr[] = new int[n+2];
        arr[0] = 1;
        arr[n+1] = 1;
        for (int i = 0; i < n; i++) {
            arr[i+1] = nums[i];
        }
        return helper(arr, 1, n);
    }
}
```

```java
class Solution {
    int memo[][];
    private int helper(int arr[], int i, int j) {
        if (i > j) return 0;
        if(memo[i][j] != -1) return memo[i][j];
        int res = -1;
        for (int k = i; k <= j; k++) {
            int cost = arr[i-1] * arr[k] * arr[j+1];
            res = Math.max(res, cost + helper(arr, i, k-1) + helper(arr, k+1, j));
        }
        return memo[i][j] = res;
    }
    public int maxCoins(int[] nums) {
        int n = nums.length;
        int arr[] = new int[n+2];
        arr[0] = 1;
        arr[n+1] = 1;
        for (int i = 0; i < n; i++) {
            arr[i+1] = nums[i];
        }

        for (int i = 0; i < n+2; i++){
            Arrays.fill(memo[i], -1);
        }
        return helper(arr, 1, n);
    }
}
```

```java
class Solution {

        if (memo[i][j] != -1) return memo[i][j];

        for (int k = i+1; k < j; k++) {
            int cal = arr[i] * arr[k] * arr[j];
            res = Math.max(res, cal + helper(arr, i, k) + helper(arr, k, j));
        }
        return memo[i][j] = res;
    }
    public int maxCoins(int[] nums) {
        int n = nums.length;
        memo = new int[n+2][n+2];
        int arr[] = new int[n+2];
        arr[0] = arr[n+1] = 1;
        for (int i = 1 ; i < n+1; i++) {
            arr[i] = nums[i-1];
        }
        for (int i = 0; i < n+2; i++) {
            Arrays.fill(memo[i], -1);
        }
        return helper(arr, 0, n+1);
    }
}
```

# DP

```java
class Solution {
    public int maxCoins(int[] nums) {
        int n = nums.length;
        int arr[] = new int[n+2];
        arr[0] = 1;
        arr[n+1] = 1;
        for (int i = 0; i < n; i++) {
            arr[i+1] = nums[i];
        }
        
        int dp[][] = new int[n+2][n+1];
        for (int i = n+1; i >= 1; i--) {
            for (int j = i; j <= n; j++) {
                int res = -1;
                for (int k = i; k <= j; k++) {
                    int cost = arr[i-1] * arr[k] * arr[j+1];
                    res = Math.max(res, cost + dp[i][k-1] + dp[k+1][j]);
                }
                dp[i][j] = res;
            }
        }
        return dp[1][n];
    }
}
```

```java
class Solution {
    public int maxCoins(int[] nums) {
        int n = nums.length;
        int[] arr = new int[n+2];
        arr[0] = arr[n+1] = 1;
        int dp[][] = new int[n+2][n+2];
        for (int i = 1; i < n+1; i++) {
            arr[i] = nums[i-1];
        }

        for (int j = 2; j < n+2; j++) {
            for (int i = j-2; i >= 0; i--) {
                for (int k = i+1; k < j; k++) {
                    int cal = arr[i] * arr[k] * arr[j];
                    dp[i+1][j-1] = Math.max(dp[i+1][j-1], dp[i+1][k-1] + dp[k+1][j-1] + cal);
                }
            }
        }
        return dp[1][n];
    }
}
```

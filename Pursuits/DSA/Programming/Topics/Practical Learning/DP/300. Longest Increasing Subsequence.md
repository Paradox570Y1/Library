[300. Longest Increasing Subsequence](https://leetcode.com/problems/longest-increasing-subsequence/)

Given an integer array `nums`, return _the length of the longest **strictly increasing**_ _**subsequence**_.

**Example 1:**

**Input:** nums = [10,9,2,5,3,7,101,18]
**Output:** 4
**Explanation:** The longest increasing subsequence is [2,3,7,101], therefore the length is 4.

**Example 2:**

**Input:** nums = [0,1,0,3,2,3]
**Output:** 4

**Example 3:**

**Input:** nums = [7,7,7,7,7,7,7]
**Output:** 1

**Constraints:**

- `1 <= nums.length <= 2500`
- `-104 <= nums[i] <= 104`


# Brute

**Using Recursion **

```java
class Solution {
    int res;
    private void helper(int[] nums,int i,List<Integer> li){
        if(i==nums.length){
            res = Math.max(li.size(),res);
            return;
        }
        helper(nums,i+1,li);
        if(li.size() == 0 || li.get(li.size()-1)<nums[i]){
            li.add(nums[i]);
            helper(nums,i+1,li);
            li.remove(li.size()-1);
        }
    }
    public int lengthOfLIS(int[] nums) {
        helper(nums,0,new ArrayList<>());
        return res;
    }
}
```

 **Using Bits to Generate Power Set**
 (will not work as length of array can be as long as 2500 which will generate 2^2500 power sets.)
```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int totalSubsequences = 1<<n;
        int maxLen=0;
        for(int mask=1;mask<totalSubsequences;mask++){
            int len=0;
            int prev=-2147483648;
            for(int i=0;i<n;i++){
                if((mask & 1<<i) !=0 && nums[i]>prev){
                    prev=nums[i];
                    len++;
                }
            }
            maxLen = Math.max(maxLen,len);
        }
        return maxLen;
    }
}
```

```java
class Solution {
    int memo[][];
    private int helper(int[] nums,int cur,int prev){
        if(cur >= nums.length)return 0;
        if(prev!=-1 && memo[cur][prev] != -1) return memo[cur][prev];
        int maxLen = helper(nums,cur+1,prev);
        if(prev==-1 || nums[cur]>nums[prev]){
            maxLen = Math.max(1 + helper(nums,cur+1,cur),maxLen);
        }
        if(prev!=-1)memo[cur][prev] = maxLen;
        return maxLen;
    }
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        memo = new int[n][n];
        for(int i=0;i<n;i++){
            Arrays.fill(memo[i],-1);
        }
        return helper(nums,0,-1);
    }
}
```

```java
class Solution {
    private int helper(int[] nums, int i, int j, int memo[][]) {
        if (i < 0) return 0;
        if (memo[i][j] != -1) return memo[i][j];
        int res = 0;
        if(j == nums.length || nums[i] < nums[j]) {
            int w1 = 1 + helper(nums, i-1, i, memo);
            int w2 = helper(nums, i-1, j, memo);
            res = Math.max(w1, w2);
        } else {
            res = helper(nums, i-1, j, memo);
        }
        return memo[i][j] = res;
    }
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int[][] memo = new int[n][n+1];
        for (int i = 0; i < n; i++) {
            Arrays.fill(memo[i], -1);
        }
        return helper(nums, n-1, n, memo);
    }
}
```

```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        int dp[][] = new int[n+1][n+1];
        for(int i=n-1;i>=0;i--){
            for(int j=i-1;j>=-1;j--){
                int len = dp[i+1][j+1];
                if(j==-1 || nums[i] > nums[j]){
                    len = Math.max(len,1 + dp[i+1][i+1]);
                }
	                dp[i][j+1] = len;
            }
        }
        return dp[0][0];
    }
}
```

# Generating back sequence

[Problem](https://www.geeksforgeeks.org/problems/printing-longest-increasing-subsequence/1) (similar problem)

```java
class Solution {
    public int[] lengthOfLIS(int[] nums) {
        int n = nums.length;
        int dp[] = new int[n];
        int hash[] = new int[n];
        int maxLen=0;
        int lastIdx = 0;
        for(int i=0;i<n;i++){
            dp[i] = 1;
        }
        for(int i=0;i<n;i++){
            hash[i] = i;
            for(int j=0;j<i;j++){
                if(nums[i]>nums[j] && dp[i]<1+dp[j]){
                    dp[i] = 1+dp[j];
                    hash[i] = j;
                }
            }
            if(maxLen < dp[i]){
                maxLen = dp[i];
                lastIdx = i;
            }
        }
        int arr[] = new int[maxLen];
        int idx = maxLen-1;
        arr[idx--] = nums[lastIdx];
        while(hash[lastIdx] != lastIdx){
            lastIdx = hash[lastIdx];
            arr[idx--] = nums[lastIdx];
        }
        return arr;
    }
}
```

# Optimal only to find length
TC-> O(n log n)
```java
class Solution {
    private int lower_bound(ArrayList<Integer> li,int ele){
        int i = 0;
        int j = li.size()-1;
        int res = -1;
        while(i<=j){
            int mid = i + (j-i)/2;
            int num = li.get(mid);
            if(num >= ele){
                res = mid;
                j=mid-1;
            }
            else i=mid+1;
        }
        return res;
    }
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        ArrayList<Integer> li = new ArrayList<>();
        for(int num:nums){
            int idx = lower_bound(li,num);
            if(idx==-1)li.add(num);
            else li.set(idx,num);
        }
        return li.size();
    }
}
```

# Converting LIS to LCS
- Take `TreeSet()` to get sorted unique values from the given array.
- Now take `LCS` of the generated set and the given array.
- You will get the length.
- I think it might be possible to generate the sequence back from it, Yet to test that.

Or without using `TreeSet()` as well
```java
class Solution {
    private int lcs(int[] arr1, int[] arr2) {
        int n = arr1.length;
        int[] dp = new int[n+1];
        for (int i = 1; i <= n; i++) {
            if (i > 1 && arr2[i-1] == arr2[i-2]) continue;
            int prev = 0;
            for (int j = 1; j <= n; j++) {
                int b = dp[j];
                if (arr2[i-1] == arr1[j-1]) {
                    dp[j] = prev + 1;
                } else {
                    dp[j] = Math.max(dp[j], dp[j-1]);
                }
                prev = b;
            }
        }
        return dp[n];
    }
    public int lengthOfLIS(int[] nums) {
        int[] sorted_nums = nums.clone();
        Arrays.sort(sorted_nums);
        return lcs(nums, sorted_nums);
    }
}
```


# Optimal

Using Binary Search
```java
class Solution {
    private int findPos(ArrayList<Integer> list, int target) {
        int high = list.size()-1;
        if (list.get(high) == target) return high;
        int low = 0;
        while (low <= high) {
            int mid = (low + high)/2;
            if (list.get(mid) == target) return mid;
            else if (list.get(mid) < target) {
                low = mid + 1;
            } else high = mid - 1;
        }
        return low;
    }
    public int lengthOfLIS(int[] nums) {
        int n = nums.length;
        ArrayList<Integer> list = new ArrayList<>();
        list.add(nums[0]);
        for (int i = 1; i < n; i++) {
            if (list.get(list.size()-1) < nums[i]) {
                list.add(nums[i]);
            } else {
                int pos = findPos(list, nums[i]);
                list.set(pos, nums[i]);
            }
        }
        return list.size();
    }
}
```

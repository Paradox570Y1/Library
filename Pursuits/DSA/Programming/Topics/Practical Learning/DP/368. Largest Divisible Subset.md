[368. Largest Divisible Subset](https://leetcode.com/problems/largest-divisible-subset/)


Given a set of **distinct** positive integers `nums`, return the largest subset `answer` such that every pair `(answer[i], answer[j])` of elements in this subset satisfies:

- `answer[i] % answer[j] == 0`, or
- `answer[j] % answer[i] == 0`

If there are multiple solutions, return any of them.

**Example 1:**

**Input:** nums = [1,2,3]
**Output:** [1,2]
**Explanation:** [1,3] is also accepted.

**Example 2:**

**Input:** nums = [1,2,4,8]
**Output:** [1,2,4,8]

**Constraints:**

- `1 <= nums.length <= 1000`
- `1 <= nums[i] <= 2 * 109`
- All the integers in `nums` are **unique**.


# Recursion

```java
class Solution {
    private List<Integer> res;
    private void helper(int[] nums,int i,int prev,List<Integer> li){
        if(i == nums.length){
            if(li.size() > res.size()){
                res = new ArrayList<>(li);
            }
            return;
        }
        helper(nums,i+1,prev,li);
        if(prev == -1 || (nums[i] % nums[prev] == 0)){
            li.add(nums[i]);
            helper(nums,i+1,i,li);
            li.remove(li.size()-1);
        }
    }
    public List<Integer> largestDivisibleSubset(int[] nums) {
        Arrays.sort(nums);
        res = new ArrayList<>();
        helper(nums,0,-1,new ArrayList<>());
        return res;
    }
}
```

```java
class Solution {
    private List<Integer> reverse(List<Integer> li){
        int i=0,j=li.size()-1;
        while(i<j){
            int temp = li.get(i);
            li.set(i,li.get(j));
            li.set(j,temp);
            i++;j--;
        }
        return li;
    }
    public List<Integer> largestDivisibleSubset(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int dp[] = new int[n];
        int hash[] = new int[n];
        for(int i=0;i<n;i++)dp[i]=1;
        int maxLen=1;
        int lastIdx=0;
        for(int i=0;i<n;i++){
            hash[i] = i;
            for(int j=0;j<i;j++){
                if(nums[i]%nums[j] == 0 && dp[i]<dp[j]+1){
                    dp[i] = dp[j]+1;
                    hash[i] = j;
                }
            }
            if(maxLen<dp[i]){
                maxLen = dp[i];
                lastIdx = i;
            }
        }
        List<Integer> res = new ArrayList<>(List.of(nums[lastIdx]));
        while(hash[lastIdx] != lastIdx){
            lastIdx = hash[lastIdx];
            res.add(nums[lastIdx]);
        }
        return reverse(res);
    }
}
```

```java
class Solution {
    public List<Integer> largestDivisibleSubset(int[] nums) {
        Arrays.sort(nums);
        int n = nums.length;
        int[] dp = new int[n];
        int[] parent = new int[n];
        Arrays.fill(parent, -1);
        int res = 0;
        int p = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[i] % nums[j] == 0 && dp[i] < dp[j]+1) {
                    dp[i] = dp[j]+1;
                    parent[i] = j;
                }
            }
            if (res < dp[i]) {
                res = dp[i];
                p = i;
            }
        }

        List<Integer> ans = new ArrayList<>();
        while (parent[p] != -1) {
            ans.add(nums[p]);
            p = parent[p];
        }
        ans.add(nums[p]);
        return ans;
    }
}
```

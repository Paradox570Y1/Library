[188. Best Time to Buy and Sell Stock IV](https://leetcode.com/problems/best-time-to-buy-and-sell-stock-iv/)

You are given an integer array `prices` where `prices[i]` is the price of a given stock on the `ith` day, and an integer `k`.

Find the maximum profit you can achieve. You may complete at most `k` transactions: i.e. you may buy at most `k` times and sell at most `k` times.

**Note:** You may not engage in multiple transactions simultaneously (i.e., you must sell the stock before you buy again).

**Example 1:**

**Input:** k = 2, prices = [2,4,1]
**Output:** 2
**Explanation:** Buy on day 1 (price = 2) and sell on day 2 (price = 4), profit = 4-2 = 2.

**Example 2:**

**Input:** k = 2, prices = [3,2,6,5,0,3]
**Output:** 7
**Explanation:** Buy on day 2 (price = 2) and sell on day 3 (price = 6), profit = 6-2 = 4. Then buy on day 5 (price = 0) and sell on day 6 (price = 3), profit = 3-0 = 3.

**Constraints:**

- `1 <= k <= 100`
- `1 <= prices.length <= 1000`
- `0 <= prices[i] <= 1000`

# Recursion

```java
class Solution {
    private int helper(int[] prices,int i,int buy,int cap){
        if(i == prices.length || cap == 0)return 0;
        int profit=0;
        if(buy == 1){
            int b = -prices[i] + helper(prices,i+1,0,cap);
            int nb = helper(prices,i+1,1,cap);
            profit = Math.max(b,nb);
        } else {
            int s = prices[i] + helper(prices,i+1,1,cap-1);
            int ns = helper(prices,i+1,0,cap);
            profit = Math.max(s,ns);
        }
        return profit;
    }
    public int maxProfit(int k, int[] prices) {
        return helper(prices,0,1,k);
    }
}
```

```java
class Solution {
    int memo[][][];
    private int helper(int[] prices,int i,int buy,int cap){
        if(i == prices.length || cap == 0)return 0;
        if(memo[i][buy][cap] != -1)return memo[i][buy][cap];
        int profit=0;
        if(buy == 1){
            int b = -prices[i] + helper(prices,i+1,0,cap);
            int nb = helper(prices,i+1,1,cap);
            profit = Math.max(b,nb);
        } else {
            int s = prices[i] + helper(prices,i+1,1,cap-1);
            int ns = helper(prices,i+1,0,cap);
            profit = Math.max(s,ns);
        }
        return memo[i][buy][cap] = profit;
    }
    public int maxProfit(int k, int[] prices) {
        int n = prices.length;
        memo = new int[n+1][2][k+1];
        for(int i=0;i<=n;i++){
            Arrays.fill(memo[i][0],-1);
            Arrays.fill(memo[i][1],-1);
        }
        return helper(prices,0,1,k);
    }
}
```

```java
class Solution {
    public int maxProfit(int k, int[] prices) {
        int n = prices.length;
        int[][][] dp = new int[n+1][2][k+1];
        int profit=0;
        for(int i=n-1;i>=0;i--){
            for(int buy=0;buy<=1;buy++){
                for(int cap=1;cap<=k;cap++){
                    if(buy==1){
                        int b = -prices[i] + dp[i+1][0][cap];
                        int nb = dp[i+1][1][cap];
                        profit = Math.max(b,nb);
                    }
                    else{
                        int s = prices[i] + dp[i+1][1][cap-1];
                        int ns = dp[i+1][0][cap];
                        profit = Math.max(s,ns);
                    }
                    dp[i][buy][cap] = profit;
                }
            }
        }
        return dp[0][1][k];
    }
}
```

```java
        for(int i=n-1;i>=0;i--){
```

```java
                for(int cap=1;cap<=k;cap++){

                        profit = Math.max(s,ns);
                    }
                    cur[buy][cap] = profit;
                }
            }
            prev = cur;
        }
        return prev[1][k];
    }
}
```

```java
class Solution {
    public int maxProfit(int k, int[] prices) {
        int[][] dp = new int[k+1][2];
        int n = prices.length;
        for (int i = 0; i <= k; i++) {
            dp[i][0] = 0;
            dp[i][1] = (int)-1e9;
        }

        for (int i = n-1; i >= 0; i--) {
            for (int t = 1; t <= k; t++) {
                dp[t][0] = Math.max(dp[t][0], dp[t][1] - prices[i]);
                dp[t][1] = Math.max(dp[t][1], dp[t-1][0] + prices[i]);
            }
        }
        
        return dp[k][0];
    }
}
```

```java
class Solution {
    public int maxProfit(int k, int[] prices) {
        int[][] dp = new int[k+1][2];
        int n = prices.length;
        for (int i = 0; i <= k; i++) {
            dp[i][0] = Integer.MAX_VALUE;
        }
        for (int price : prices) {
            for (int i = 1; i <= k ; i++) {
                dp[i][0] = Math.min(dp[i][0], price - dp[i-1][1]);
                dp[i][1] = Math.max(dp[i][1], price - dp[i][0]);
            }
        }
        return dp[k][1];
    }
}
```

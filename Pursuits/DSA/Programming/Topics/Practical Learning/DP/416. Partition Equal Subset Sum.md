[416. Partition Equal Subset Sum](https://leetcode.com/problems/partition-equal-subset-sum/)

Given an integer array `nums`, return `true` _if you can partition the array into two subsets such that the sum of the elements in both subsets is equal or_ `false` _otherwise_.

**Example 1:**

**Input:** nums = [1,5,11,5]
**Output:** true
**Explanation:** The array can be partitioned as [1, 5, 5] and [11].

**Example 2:**

**Input:** nums = [1,2,3,5]
**Output:** false
**Explanation:** The array cannot be partitioned into equal sum subsets.

**Constraints:**

- `1 <= nums.length <= 200`
- `1 <= nums[i] <= 100`


# Using recursion

```java
class Solution {
    private boolean helper(int[] nums,int i,int c1,int c2){
        if(i==0){
            if(c1+nums[i] == c2 || c2+nums[i] == c1)return true;
            return false;
        }
        boolean set1 = helper(nums,i-1,c1+nums[i],c2);
        boolean set2 = helper(nums,i-1,c1,c2+nums[i]);
        return (set1 || set2);
    }
    public boolean canPartition(int[] nums) {
        int n = nums.length;
        
        return helper(nums,n-1,0,0);
    }
}
```



# Using Memoization (with subset sum equal k method)

```java
class Solution {
    private boolean helper(int[] nums,int i,int target,Boolean memo[][]){
        if(target==0)return true;
        if(i==0)return (target==nums[i]);
        if(memo[i][target]!=null)return memo[i][target];
        boolean take = false;
        if(nums[i]<=target)take = helper(nums,i-1,target-nums[i],memo);
        boolean leaveIt = helper(nums,i-1,target,memo);
        return memo[i][target]  = (take | leaveIt);
    }
    public boolean canPartition(int[] nums) {
        int n = nums.length;
        int sum=0;
        for(int i=0;i<n;i++){
            sum+=nums[i];
        }
        if(sum%2 == 1)return false;
        Boolean memo[][] = new Boolean[n][sum/2 + 1];
        return helper(nums,n-1,sum/2,memo);
    }
}
```




# Using Tabulation

With space optimization
```java
class Solution {
    public boolean canPartition(int[] nums) {
        int n = nums.length;
        int sum=0;
        for(int i=0;i<n;i++){
            sum+=nums[i];
        }
        if(sum%2 == 1)return false;
        int target = sum/2;
        boolean prev[] = new boolean[target + 1];
        prev[0] = true;
        if(nums[0]<=target)prev[nums[0]] = true;
        for(int i=1;i<n;i++){
            boolean cur[] = new boolean[target+1];
            cur[0] = true;
            for(int j=1;j<=target;j++){
                boolean notPick = prev[j];
                boolean pick = false;
                if(nums[i]<=j)pick = prev[j-nums[i]];
                cur[j] = pick | notPick;
            }
            prev=cur;
        }
        return prev[target];
    }
}
```

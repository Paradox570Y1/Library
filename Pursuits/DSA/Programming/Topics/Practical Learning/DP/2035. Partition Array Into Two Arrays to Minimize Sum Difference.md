[2035. Partition Array Into Two Arrays to Minimize Sum Difference](https://leetcode.com/problems/partition-array-into-two-arrays-to-minimize-sum-difference/)


You are given an integer array `nums` of `2 * n` integers. You need to partition `nums` into **two** arrays of length `n` to **minimize the absolute difference** of the **sums** of the arrays. To partition `nums`, put each element of `nums` into **one** of the two arrays.

Return _the **minimum** possible absolute difference_.

**Example 1:**

![example-1](https://assets.leetcode.com/uploads/2021/10/02/ex1.png)

**Input:** nums = [3,9,7,3]
**Output:** 2
**Explanation:** One optimal partition is: [3,9] and [7,3].
The absolute difference between the sums of the arrays is abs((3 + 9) - (7 + 3)) = 2.

**Example 2:**

**Input:** nums = [-36,36]
**Output:** 72
**Explanation:** One optimal partition is: [-36] and [36].
The absolute difference between the sums of the arrays is abs((-36) - (36)) = 72.

**Example 3:**

![example-3](https://assets.leetcode.com/uploads/2021/10/02/ex3.png)

**Input:** nums = [2,-1,0,4,-2,-9]
**Output:** 0
**Explanation:** One optimal partition is: [2,4,-9] and [-1,0,-2].
The absolute difference between the sums of the arrays is abs((2 + 4 + -9) - (-1 + 0 + -2)) = 0.

**Constraints:**

- `1 <= n <= 15`
- `nums.length == 2 * n`
- `-107 <= nums[i] <= 107`

# Brute
MLE

```java
class Solution {
    public int minimumDifference(int[] nums) {
        int n = nums.length;
        int half = n/2;
        HashSet<Integer>[] dp = new HashSet[half+1];
        for(int i=0;i<=half;i++){
            dp[i] = new HashSet<>();
        }
        dp[0].add(0);
        for(int num:nums){
            for(int i=half;i>0;i--){
                for(int j=0;j<dp[i-1].size();j++){
                    dp[i].add(dp[i-1].get(j)+num);
                }
            }
        }
        int total = Arrays.stream(nums).sum();
        int size = dp[half].size();
        int ans=Integer.MAX_VALUE;
        for(int i=0;i<size;i++){
            int sum1 = dp[half].get(i);
            int sum2 = total-sum1;
            ans = Math.min(ans,Math.abs(sum1-sum2));
        }
        return ans;
    }
}
```

### **4. Comparison with Meet-in-the-Middle**

| Approach               | Time Complexity  | Space Complexity | Notes                          |
| ---------------------- | ---------------- | ---------------- | ------------------------------ |
| **Above DP (HashSet)** | `O(n² × 2^n)`    | `O(2^n)`         | Simple but slow for large `n`. |
| **Meet-in-the-Middle** | `O(n × 2^(n/2))` | `O(2^(n/2))`     | Faster for `n ≥ 20`.           |

# Using DP + Meet in the Middle

```java
class Solution {
    public int minimumDifference(int[] nums) {
        int n = nums.length;
        int half = n/2;
        List<Integer>[] left = new ArrayList[half+1];
        List<Integer>[] right = new ArrayList[half+1];
        for(int i=0;i<=half;i++){
            left[i] = new ArrayList<>();
            right[i] = new ArrayList<>();
        }
        for(int set=0;set<(1<<half);set++){
            int eleCnt = Integer.bitCount(set);
            int sum=0;
            for(int i=0;i<half;i++){
                if((set & (1<<i)) != 0){
                    sum += nums[i];
                }
            }
            left[eleCnt].add(sum);
        }
        for(int set=0;set<(1<<half);set++){
            int eleCnt = Integer.bitCount(set);
            int sum=0;
            for(int i=half;i<n;i++){
                if((set & (1<<(i-half))) != 0){
                    sum += nums[i];
                }
            }
            right[eleCnt].add(sum);
        }
        int ans = Integer.MAX_VALUE;
        int totalSum = Arrays.stream(nums).sum();
        for(List<Integer> sets:right){
            Collections.sort(sets);
        }
        for(int i=0;i<=half;i++){ //i means cnt of ele
            List<Integer> leftSum = left[i];
            List<Integer> rightSum = right[half-i];
            for(int sum1:leftSum){
                int idealSum2 = (totalSum/2 - sum1);
                int closestSumId = Collections.binarySearch(rightSum,idealSum2);
                if(closestSumId>=0){
                    int sumOfHalfEle = sum1+idealSum2;
                    int diff = Math.abs(totalSum- 2*sumOfHalfEle);
                    ans = Math.min(ans,diff);
                }
                else{
                    int closestSum1ID = -closestSumId -1;
                    int closestSum2ID = closestSum1ID-1;
                    if(closestSum1ID<rightSum.size()){
                        int sum2 = rightSum.get(closestSum1ID);
                        int sumOfHalfEle = sum1+sum2;
                        int diff = Math.abs(totalSum- 2*sumOfHalfEle);
                        ans = Math.min(ans,diff);
                    }
                    if(closestSum1ID>0){
                        int sum2 = rightSum.get(closestSum2ID);
                        int sumOfHalfEle = sum1+sum2;
                        int diff = Math.abs(totalSum- 2*sumOfHalfEle);
                        ans = Math.min(ans,diff);
                    }
                }
            }
        }
        return ans;
    }
}
```

```java
class Solution {
    private int binarySearch(List<Integer> list,int target){
        int left = 0,right = list.size()-1;
        int ans=list.size();
        while(left<=right){
            int mid = (left+right)>>1;
            if(list.get(mid)>=target){
                ans = mid;
                right = mid-1;
            }
            else left = mid+1;
        }
        return ans;
    }

                    int sumOfHalfEle = sum1+idealSum2;
                    int diff = Math.abs(totalSum- 2*sumOfHalfEle);
                    ans = Math.min(ans,diff);
                }
                else{
                    int closestSum2ID = closestSumId-1;
                    if(closestSumId<rightSum.size()){
                        int sum2 = rightSum.get(closestSumId);
                        int sumOfHalfEle = sum1+sum2;
                        int diff = Math.abs(totalSum- 2*sumOfHalfEle);

                        int diff = Math.abs(totalSum- 2*sumOfHalfEle);
                        ans = Math.min(ans,diff);
                    }
                }
            }
        }
        return ans;
    }
```

```java
class Solution {
    private void helper(int[] nums, int st, int end, List<Integer>[] list, int sum, int count) {
        if (st == end) {
            list[count].add(sum);
            return;
        }
        helper(nums, st+1, end, list, sum, count);
        helper(nums, st+1, end, list, sum + nums[st], count+1);
    }
    public int minimumDifference(int[] nums) {
        int n = nums.length;
        int T = 0;
        for (int num : nums) T += num;
        List<Integer>[] left = new ArrayList[n/2+1];
        List<Integer>[] right = new ArrayList[n/2+1];
        for (int i = 0; i <= n/2; i++) {
            left[i] = new ArrayList<>();
            right[i] = new ArrayList<>();
        }
        helper(nums, 0, n/2, left, 0, 0);
        helper(nums, n/2, n, right, 0, 0);

        for (int i = 0; i <= n/2; i++) {
            Collections.sort(right[i]);
        }
        int ans = Integer.MAX_VALUE;
        for (int i = 0; i <= n/2; i++) {
            List<Integer> leftArr = left[i];
            List<Integer> rightArr = right[n/2-i];
            for (int lsum : leftArr) {
                int ideal_rsum = T/2 - lsum;
                int idx = Collections.binarySearch(rightArr, ideal_rsum);
                if (idx < 0) idx = -(idx+1);
                if (idx < rightArr.size()) {
                    int sum = lsum + rightArr.get(idx);
                    ans = Math.min(ans, Math.abs(T-2*sum));
                }
                if (idx > 0) {
                    int sum = lsum + rightArr.get(idx-1);
                    ans = Math.min(ans, Math.abs(T-2*sum));
                }
            }
        }
        return ans;
    }
}
```

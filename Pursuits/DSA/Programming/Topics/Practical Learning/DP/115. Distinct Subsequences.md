[115. Distinct Subsequences](https://leetcode.com/problems/distinct-subsequences/)

Given two strings s and t, return _the number of distinct_ **_subsequences_** _of_ s _which equals_ t.

The test cases are generated so that the answer fits on a 32-bit signed integer.

**Example 1:**

**Input:** s = "rabbbit", t = "rabbit"
**Output:** 3
**Explanation:**
As shown below, there are 3 ways you can generate "rabbit" from s.
`**rabb**b**it**`
`**ra**b**bbit**`
`**rab**b**bit**`

**Example 2:**

**Input:** s = "babgbag", t = "bag"
**Output:** 5
**Explanation:**
As shown below, there are 5 ways you can generate "bag" from s.
`**ba**b**g**bag`
`**ba**bgba**g**`
`**b**abgb**ag**`
`ba**b**gb**ag**`
`babg**bag**`

**Constraints:**

- `1 <= s.length, t.length <= 1000`
- `s` and `t` consist of English letters.


# Using Recursion

```java
class Solution {
    private int helper(String s,String t,int i,int j){
        if(j==0)return 1;
        if(i==0)return 0;
        int cnt=0;
        if(s.charAt(i-1) == t.charAt(j-1)){
            cnt = helper(s,t,i-1,j-1)+helper(s,t,i-1,j);
        }
        else{
            cnt = helper(s,t,i-1,j);
        }
        return cnt;
    }
    public int numDistinct(String s, String t) {
        int n = s.length();
        int m = t.length();
        return helper(s,t,n,m);
    }
}
```


# Using Memoization

```java
class Solution {
    int memo[][];
    private int helper(String s,String t,int i,int j){
        if(j==0)return 1;
        if(i==0)return 0;
        if(memo[i][j]!=-1)return memo[i][j];
        int cnt=0;
        if(s.charAt(i-1) == t.charAt(j-1)){
            cnt = helper(s,t,i-1,j-1)+helper(s,t,i-1,j);
        }
        else{
            cnt = helper(s,t,i-1,j);
        }
        return memo[i][j] = cnt;
    }
    public int numDistinct(String s, String t) {
        int n = s.length();
        int m = t.length();
        memo = new int[n+1][m+1];
        for(int i=0;i<=n;i++){
            Arrays.fill(memo[i],-1);
        }
        return helper(s,t,n,m);
    }
}
```

# Using DP

```java
class Solution {
    public int numDistinct(String s, String t) {
        int n = s.length();
        int m = t.length();
        int dp[] = new int[m+1];
        dp[0] = 1;
        for(int i=1;i<=n;i++){
            int cur[] = new int[m+1];
            cur[0] = 1;
            for(int j=1;j<=m;j++){
                if(s.charAt(i-1) == t.charAt(j-1)){
                    cur[j] = dp[j-1]+dp[j];
                }
                else{
                    cur[j] = dp[j];
                }
            }
            dp = cur;
        }
        return dp[m];
    }
}
```

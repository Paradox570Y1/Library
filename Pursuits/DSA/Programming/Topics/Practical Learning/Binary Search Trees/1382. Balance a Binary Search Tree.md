[1382. Balance a Binary Search Tree](https://leetcode.com/problems/balance-a-binary-search-tree/)


Given the `root` of a binary search tree, return _a **balanced** binary search tree with the same node values_. If there is more than one answer, return **any of them**.

A binary search tree is **balanced** if the depth of the two subtrees of every node never differs by more than `1`.

**Example 1:**

![](https://assets.leetcode.com/uploads/2021/08/10/balance1-tree.jpg)

**Input:** root = [1,null,2,null,3,null,4,null,null]
**Output:** [2,1,3,null,null,null,4]
**Explanation:** This is not the only correct answer, [3,1,4,null,2] is also correct.

**Example 2:**

![](https://assets.leetcode.com/uploads/2021/08/10/balanced2-tree.jpg)

**Input:** root = [2,1,3]
**Output:** [2,1,3]

**Constraints:**

- The number of nodes in the tree is in the range `[1, 104]`.
- `1 <= Node.val <= 105`


# Brute
```java
class Solution {
    private TreeNode balancedTree;
    private TreeNode insert(TreeNode root, int val) {
        if (root == null) return new TreeNode(val);
        if (val > root.val) root.right = insert(root.right, val);
        else if (val < root.val) root.left = insert(root.left, val);
        return root;
    }
    private void buildTree(List<Integer> list , int low, int high) {
        if (low > high) return;
        int mid = low + (high-low)/2;
        insert(balancedTree, list.get(mid));
        buildTree(list, low, mid-1);
        buildTree(list, mid+1, high);
    }
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        traverseTree(root, list);
        int mid = list.size()/2;
        balancedTree = new TreeNode(list.get(mid));
        buildTree(list, 0, mid);
        buildTree(list, mid+1, list.size()-1);
        return balancedTree;
    }
    private void traverseTree(TreeNode root, List<Integer> list) {
        if (root == null) return;
        traverseTree(root.left, list);
        list.add(root.val);
        traverseTree(root.right, list);
    }
}
```

```java
class Solution {
    private TreeNode buildTree(List<Integer> list , int low, int high) {
        if (low > high) return null;
        int mid = low + (high-low)/2;
        TreeNode root = new TreeNode(list.get(mid));
        root.left = buildTree(list, low, mid-1);
        root.right = buildTree(list, mid+1, high);
        return root;
    }
    public TreeNode balanceBST(TreeNode root) {
        List<Integer> list = new ArrayList<>();
        traverseTree(root, list);
        return buildTree(list, 0, list.size()-1);
    }
    private void traverseTree(TreeNode root, List<Integer> list) {
        if (root == null) return;
        traverseTree(root.left, list);
        list.add(root.val);
        traverseTree(root.right, list);
    }
}
```

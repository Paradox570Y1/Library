[653. Two Sum IV - Input is a BST](https://leetcode.com/problems/two-sum-iv-input-is-a-bst/)

Given the `root` of a binary search tree and an integer `k`, return `true` _if there exist two elements in the BST such that their sum is equal to_ `k`, _or_ `false` _otherwise_.

**Example 1:**

![](https://assets.leetcode.com/uploads/2020/09/21/sum_tree_1.jpg)

**Input:** root = [5,3,6,2,4,null,7], k = 9
**Output:** true

**Example 2:**

![](https://assets.leetcode.com/uploads/2020/09/21/sum_tree_2.jpg)

**Input:** root = [5,3,6,2,4,null,7], k = 28
**Output:** false

**Constraints:**

- The number of nodes in the tree is in the range `[1, 104]`.
- `-104 <= Node.val <= 104`
- `root` is guaranteed to be a **valid** binary search tree.
- `-105 <= k <= 105`


# Brute
TC-> O(N)
SC-> O(N)
```java
class Solution {
    HashSet<Integer> hs = new HashSet<>();
    private boolean dfs(TreeNode node,int k){
        if(node==null)return false;
        if(hs.contains(k-node.val))return true;
        hs.add(node.val);
        if(dfs(node.left,k) || dfs(node.right,k))return true;
        return false;
    }
    public boolean findTarget(TreeNode root, int k) {
        return dfs(root,k);
    }
}
```

```java
class Solution {
    private void next(Stack<TreeNode> asc){
        TreeNode trav = asc.pop().right;
        while(trav!=null){
            asc.push(trav);
            trav = trav.left;
        }
    }
    private void prev(Stack<TreeNode> desc){
        TreeNode trav = desc.pop().left;
        while(trav!=null){
            desc.push(trav);
            trav = trav.right;
        }
    }
    private void basicSetup(Stack<TreeNode> asc,Stack<TreeNode> desc,TreeNode root){
        TreeNode trav=root;
        while(trav!=null){
            asc.push(trav);
            trav=trav.left;
        }
        trav=root;
        while(trav!=null){
            desc.push(trav);
            trav=trav.right;
        }
    }
    public boolean findTarget(TreeNode root, int k) {
        Stack<TreeNode> asc = new Stack<>();
        Stack<TreeNode> desc = new Stack<>();
        basicSetup(asc,desc,root);
        while((!asc.isEmpty() || !desc.isEmpty()) && asc.peek() != desc.peek()){
            TreeNode a = asc.peek();
            TreeNode b = desc.peek();
            int cur = a.val+b.val;
            if(cur==k)return true;
            else if(cur<k)next(asc);
            else prev(desc);
        }
        return false;
    }
}
```

# Using Morris in-order traversal (Optimal with space)

```java
class Solution {
    public boolean findTarget(TreeNode root, int k) {
        List<Integer> inorder = new ArrayList<>();
        TreeNode cur = root;
        while(cur!=null){
            if(cur.left!=null){
                TreeNode node = cur.left;
                while(node.right!=null && node.right!=cur){
                    node=node.right;
                }
                if(node.right==null){
                    node.right=cur;
                    cur=cur.left;
                }
                else{
                    node.right=null;
                    inorder.add(cur.val);
                    cur=cur.right;
                }
            }
            else{
                inorder.add(cur.val);
                cur=cur.right;
            }
        }
        int i=0,j=inorder.size()-1;
        while(i<j){
            int a = inorder.get(i);
            int b = inorder.get(j);
            if(a+b==k)return true;
            else if(a+b<k)i++;
            else j--;
        }
        return false;
    }
}
```


# Using BST-Iterator method (striver's way)

```java
class BSTIterator{
    Stack<TreeNode> st = new Stack<>();
    boolean rev=true; //asc-> true
    BSTIterator(TreeNode root,boolean rev){
        this.rev=rev;
        pushAll(root);
    }
    public int next(){
        TreeNode node = st.pop();
        TreeNode trav = node;
        if(rev)pushAll(trav.right);
        else pushAll(trav.left);
        return node.val;
    }
    private void pushAll(TreeNode node){
        while(node!=null){
            st.push(node);
            if(rev){
                node=node.left;
            }
            else {
                node=node.right;
            }
        }
    }
}
class Solution {
    public boolean findTarget(TreeNode root, int k) {
        BSTIterator asc = new BSTIterator(root,true);
        BSTIterator desc = new BSTIterator(root,false);
        int i=asc.next();
        int j=desc.next();
        while(i<j){
            if(i+j==k)return true;
            else if(i+j<k)i=asc.next();
            else j=desc.next();
        }
        return false;
    }
}
```

```java
class Solution {
    public boolean findTarget(TreeNode root, int k) {
        LeftIterator itl = new LeftIterator(root);
        RightIterator itr = new RightIterator(root);
        while(itl.node != itr.node) {
            int val = itl.node.val + itr.node.val;
            if(val == k) return true;
            if(val > k) {
                if(itr.hasNext()) itr.next();
                else return false;
            }
            else {
                if(itl.hasNext()) itl.next();
                else return false;
            }
        }
        return false;
    }
    class LeftIterator {
        Stack<TreeNode> st;
        TreeNode node;
        private void leftPush(TreeNode node) {
            while (node != null) {
                st.push(node);
                node = node.left;
            }
        }
        LeftIterator(TreeNode root) {
            st = new Stack<>();
            leftPush(root);
            node = st.peek();
        }
        public TreeNode next() {
            TreeNode cur = st.pop();
            leftPush(cur.right);
            if(this.hasNext()) node = st.peek();
            return cur;
        }
        public boolean hasNext() {
            return !st.isEmpty();
        }
    }
    class RightIterator {
        Stack<TreeNode> st;
        TreeNode node;
        private void rightPush(TreeNode node) {
            while (node != null) {
                st.push(node);
                node = node.right;
            }
        }
        RightIterator(TreeNode root) {
            st = new Stack<>();
            rightPush(root);
            node = st.peek();
        }
        public TreeNode next() {
            TreeNode cur = st.pop();
            rightPush(cur.left);
            if(this.hasNext()) node = st.peek();
            return cur;
        }
```

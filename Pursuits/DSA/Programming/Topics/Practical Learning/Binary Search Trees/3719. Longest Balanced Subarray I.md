[3719. Longest Balanced Subarray I](https://leetcode.com/problems/longest-balanced-subarray-i/)

You are given an integer array `nums`.

A **subarray** is called **balanced** if the number of **distinct even** numbers in the subarray is equal to the number of **distinct odd** numbers.

Return the length of the **longest** balanced subarray.

**Example 1:**

**Input:** nums = [2,5,4,3]

**Output:** 4

**Explanation:**

- The longest balanced subarray is `[2, 5, 4, 3]`.
- It has 2 distinct even numbers `[2, 4]` and 2 distinct odd numbers `[5, 3]`. Thus, the answer is 4.

**Example 2:**

**Input:** nums = [3,2,2,5,4]

**Output:** 5

**Explanation:**

- The longest balanced subarray is `[3, 2, 2, 5, 4]`.
- It has 2 distinct even numbers `[2, 4]` and 2 distinct odd numbers `[3, 5]`. Thus, the answer is 5.

**Example 3:**

**Input:** nums = [1,2,3,2]

**Output:** 3

**Explanation:**

- The longest balanced subarray is `[2, 3, 2]`.
- It has 1 distinct even number `[2]` and 1 distinct odd number `[3]`. Thus, the answer is 3.

**Constraints:**

- `1 <= nums.length <= 1500`
- `1 <= nums[i] <= 105`



# Brute

```java
class Solution {
    public int longestBalanced(int[] nums) {
        int n = nums.length;
        int len = 0;
        for (int i = 0; i < n; i++) {
            Set<Integer> odd = new HashSet<>();
            Set<Integer> even = new HashSet<>();
            for (int j = i; j < n; j++) {
                if ((nums[j] % 2) == 0) even.add(nums[j]);
                else odd.add(nums[j]);
                if (odd.size() == even.size()) len = Math.max(len, j-i+1);
            }
        }
        return len;
    }
}
```



# Using prefix sum + map

```java
class Solution {
    public int longestBalanced(int[] nums) {
        int n = nums.length;
        int res = 0;
        Map<Integer, Integer> map = new HashMap<>();
        int[] prefix = new int[n];
        for (int i = 0; i < n; i++) {
            int len = map.getOrDefault(nums[i], -1) + 1;
            int op = (nums[i] % 2 == 0)?1:-1;
            for (int j = i; j >= 0; j--) {
                if (j >= len) prefix[j] += op;
                if (prefix[j] == 0) res = Math.max(res, i-j+1);
            }
            map.put(nums[i], i);
        }
        return res;
    }
}
```

# Using Segment Trees


```java
class Solution {
    private int[] min, max, lazy;
    private int n;
    private void updateRange(int l, int r, int val) {
        updateTree(1, 0, n-1, l, r, val);
    }
    
    private void propagate(int node, int L, int R) {
        if (lazy[node] != 0) {
            min[node] += lazy[node];
            max[node] += lazy[node];
            if (L != R) {
                lazy[2*node] += lazy[node];
                lazy[2*node+1] += lazy[node];
            }
            lazy[node] = 0;
        }
    }

    private void updateTree(int node, int L, int R, int l, int r, int val) {
        propagate(node, L, R);
        if (r < L || l > R) return;
        if (l <= L && R <= r) {
            lazy[node] += val;
            propagate(node, L, R);
            return;
        }

        int mid = L + (R-L)/2;
        updateTree(2*node, L, mid, l, r, val);
        updateTree(2*node+1, mid+1, R, l, r, val);
        min[node] = Math.min(min[2*node], min[2*node+1]);
        max[node] = Math.max(max[2*node], max[2*node+1]);
    }

    private int leftMostZero(int l, int r) {
        return findZero(1, 0, n-1, l, r);
    }

    private int findZero(int node, int L, int R, int l, int r) {
        propagate(node, L, R);
        if (min[node] > 0 || max[node] < 0 || r < L || R < l) return -1;
        if (L == R) return L;
        int mid = L + (R-L)/2;
        int left = findZero(2*node, L, mid, l, r);
        if (left != -1) return left;
        return findZero(2*node+1, mid+1, R, l, r);
    }

    public int longestBalanced(int[] nums) {
        this.n = nums.length;
        min = new int[4*n];
        max = new int[4*n];
        lazy = new int[4*n];
        Map<Integer, Integer> map = new HashMap<>();
        int res = 0;
        for (int r = 0; r < n; r++) {
            int val = ((nums[r] % 2) == 0)? 1: -1;
            int l = map.getOrDefault(nums[r], -1) + 1;
            updateRange(l, r, val);
            l = leftMostZero(0, n-1);
            if (l != -1) res = Math.max(res, r-l+1);
            map.put(nums[r], r);
        }
        return res;
    }
}
```

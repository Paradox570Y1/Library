[684. Redundant Connection](https://leetcode.com/problems/redundant-connection/)

In this problem, a tree is an **undirected graph** that is connected and has no cycles.

You are given a graph that started as a tree with `n` nodes labeled from `1` to `n`, with one additional edge added. The added edge has two **different** vertices chosen from `1` to `n`, and was not an edge that already existed. The graph is represented as an array `edges` of length `n` where `edges[i] = [ai, bi]` indicates that there is an edge between nodes `ai` and `bi` in the graph.

Return _an edge that can be removed so that the resulting graph is a tree of_ `n` _nodes_. If there are multiple answers, return the answer that occurs last in the input.

**Example 1:**

![](https://assets.leetcode.com/uploads/2021/05/02/reduntant1-1-graph.jpg)

**Input:** edges = [[1,2],[1,3],[2,3]]
**Output:** [2,3]

**Example 2:**

![](https://assets.leetcode.com/uploads/2021/05/02/reduntant1-2-graph.jpg)

**Input:** edges = [[1,2],[2,3],[3,4],[1,4],[1,5]]
**Output:** [1,4]

**Constraints:**

- `n == edges.length`
- `3 <= n <= 1000`
- `edges[i].length == 2`
- `1 <= ai < bi <= edges.length`
- `ai != bi`
- There are no repeated edges.
- The given graph is connected.

# Using Disjoint Set

```java
class Solution {
    class DSU {
        int parent[];
        int size[];
        int N;

        DSU(int N) {
            this.N = N;
            parent = new int[N];
            size = new int[N];
            for (int i = 0; i < N; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int findParent(int node) {
            if(parent[node] == node) return node;
            int masterParent = findParent(parent[node]);
            parent[node] = masterParent; // compression
            return masterParent;
        }

        public boolean union(int node1, int node2) {
            int masterParent1 = findParent(node1);
            int masterParent2 = findParent(node2);

            if(masterParent1 == masterParent2) return true;
            if (size[masterParent1] >= size[masterParent2]) {
                size[masterParent1] += size[masterParent2];
                parent[masterParent2] = masterParent1;
            } else {
                size[masterParent2] += size[masterParent1];
                parent[masterParent1] = masterParent2;
            }
            return false;
        }
    }
    public int[] findRedundantConnection(int[][] edges) {
        DSU dsu = new DSU(edges.length);
        for (int[] edge : edges) {
            int node1 = edge[0]-1;
            int node2 = edge[1]-1;

            if (dsu.union(node1, node2)) {
                return edge;
            }
        }
        return new int[2];
    }
}
```

```java
class Solution {
    int[] parent, rank;
    private int find(int node) {
        if (parent[node] == node) return node;
        return parent[node] = find(parent[node]);
    }
    private void union(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (rank[pu] > rank[pv]) {
            parent[pv] = pu;
        } else if (rank[pu] < rank[pv]) {
            parent[pu] = pv;
        } else {
            parent[pu] = pv;
            rank[pv]++;
        }
    }
    public int[] findRedundantConnection(int[][] edges) {
        int V = edges.length+1;
        parent = new int[V];
        for (int i = 0; i < V; i++) parent[i] = i;
        rank = new int[V];
        int[] res = new int[]{-1, -1};
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1];
            if (find(u) != find(v)) union(u, v);
            else res = edge;
        }
        return res;
    }
}
```

```java
class Solution {
    private boolean flag = false;
    public int[] findRedundantConnection(int[][] edges) {
        List<List<int[]>> adjList = new ArrayList<>();
        int n = edges.length;
        for (int i = 0; i <= n; i++) {
            adjList.add(new ArrayList<>());
        }
        for (int i = 0; i < n; i++) {
            int edge[] = edges[i];
            int u = edge[0], v = edge[1];
            adjList.get(u).add(new int[]{v, i});
            adjList.get(v).add(new int[]{u, i});
        }
        boolean vis[] = new boolean[n+1];
        int idx = dfs(1, adjList, vis, -1);
        if (idx == -1) return new int[]{-1, -1};
        return edges[idx];
    }
    private int dfs(int u, List<List<int[]>> adjList, boolean[] vis, int parent) {
        vis[u] = true;
        for (int[] next : adjList.get(u)) {
            int v = next[0];
            int idx = next[1];
            if (!vis[v]) {
                int res = dfs(v, adjList, vis, u);
                if (res != -1 && flag) {
                    if (!vis[u])  flag = false;
                    return Math.max(res, idx);
                } else if(res != -1) {
                    vis[u] = false;
                    return res;
                }
            } else if (v != parent) {
                vis[v] = false;
                flag = true;
                return idx;
            }
        }
        return -1;
    }
}
```

```java
class Solution {
    private boolean flag = false;
    public int[] findRedundantConnection(int[][] edges) {
        int n = edges.length;
        int adjMat[][] = new int[n+1][n+1];
        for (int i = 0; i <= n; i++) {
            Arrays.fill(adjMat[i], -1);
        }
        for (int i = 0; i < n; i++) {
            int edge[] = edges[i];
            int u = edge[0], v = edge[1];
            adjMat[u][v]=i;
            adjMat[v][u]=i;
        }
        boolean vis[] = new boolean[n+1];
        int idx = dfs(1, adjMat, vis, -1);
        if (idx == -1) return new int[]{-1, -1};
        return edges[idx];
    }
    private int dfs(int u, int[][] adjMat, boolean[] vis, int parent) {
        vis[u] = true;
        for (int v = 0; v < adjMat.length; v++) {
            if (adjMat[u][v] == -1) continue;
            if (!vis[v]) {
                int res = dfs(v, adjMat, vis, u);
                if (res != -1 && flag) {
                    if (!vis[u])  flag = false;
                    return Math.max(res, adjMat[u][v]);
                } else if(res != -1) {
                    vis[u] = false;
                    return res;
                }
            } else if (v != parent) {
                vis[v] = false;
                flag = true;
                return adjMat[u][v];
            }
        }
        return -1;
    }
}
```

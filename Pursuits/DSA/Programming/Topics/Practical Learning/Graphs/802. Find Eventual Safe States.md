[802. Find Eventual Safe States](https://leetcode.com/problems/find-eventual-safe-states/)

There is a directed graph of `n` nodes with each node labeled from `0` to `n - 1`. The graph is represented by a **0-indexed** 2D integer array `graph` where `graph[i]` is an integer array of nodes adjacent to node `i`, meaning there is an edge from node `i` to each node in `graph[i]`.

A node is a **terminal node** if there are no outgoing edges. A node is a **safe node** if every possible path starting from that node leads to a **terminal node** (or another safe node).

Return _an array containing all the **safe nodes** of the graph_. The answer should be sorted in **ascending** order.

**Example 1:**

![Illustration of graph](https://s3-lc-upload.s3.amazonaws.com/uploads/2018/03/17/picture1.png)

**Input:** graph = `[[1,2],[2,3],[5],[0],[5],[],[]]`
**Output:** [2,4,5,6]
**Explanation:** The given graph is shown above.
Nodes 5 and 6 are terminal nodes as there are no outgoing edges from either of them.
Every path starting at nodes 2, 4, 5, and 6 all lead to either node 5 or 6.

**Example 2:**

**Input:** graph = `[[1,2,3,4],[1,2],[3,4],[0,4],[]]`
**Output:** [4]
**Explanation:**
Only node 4 is a terminal node, and every path starting at node 4 leads to node 4.

**Constraints:**

- `n == graph.length`
- `1 <= n <= 104`
- `0 <= graph[i].length <= n`
- `0 <= graph[i][j] <= n - 1`
- `graph[i]` is sorted in a strictly increasing order.
- The graph may contain self-loops.
- The number of edges in the graph will be in the range `[1, 4 * 104]`.



# My approaches
#### BFS

```java
class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        List<List<Integer>> adjList = new ArrayList<>();
        int[] outdegree = new int[graph.length]; // outdegree for given graph
        
        for (int i = 0; i < graph.length; i++) {
            adjList.add(new ArrayList<>());
        }
        for (int i = 0; i < graph.length; i++) {
            for (int j = 0; j < graph[i].length; j++) {
                adjList.get(graph[i][j]).add(i);
                outdegree[i]++; //its indegree for reversed graph hence i
            }
        }

        Queue<Integer> q = new LinkedList<>();

        for (int i = 0; i < outdegree.length; i++) {
            if (outdegree[i] == 0) {
                q.offer(i);
            }
        }
        List<Integer> res = new ArrayList<>();
        while (!q.isEmpty()) {
            int node = q.poll();
            res.add(node);
            for (int next:adjList.get(node)) {
                outdegree[next]--;
                if(outdegree[next] == 0) q.offer(next);
            }
        }
        Collections.sort(res);
        return res;
    }
}
```

```java
class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int V = graph.length;
        LinkedList<Integer>[] revGraph = new LinkedList[V];
        for (int v = 0; v < V; v++) {
            revGraph[v] = new LinkedList<>();
        }
        int inDegree[]  = new int[V];
        for (int v = 0; v < V; v++) {
            for (int u : graph[v]) {
                revGraph[u].add(v);
                inDegree[v]++;
            }
        }

        Queue<Integer> queue = new LinkedList<>();
        for (int v = 0; v < V; v++) {
            if (inDegree[v] == 0) queue.offer(v);
        }

        List<Integer> res = new ArrayList<>();
        while (!queue.isEmpty()) {
            int u = queue.poll();
            res.add(u);
            for (int v : revGraph[u]) {
                inDegree[v]--;
                if (inDegree[v] == 0) queue.offer(v);
            }
        }
        Collections.sort(res);
        return res;
    }
}
```

### Unconventional mixture of  DFS and BFS 😅

```java
class Solution {
    List<Integer> res;
    int[] outdegree;
    boolean[] vis;
    public List<Integer> eventualSafeNodes(int[][] graph) {
        List<List<Integer>> adjList = new ArrayList<>();
        this.outdegree = new int[graph.length];
        this.vis = new boolean[graph.length];
        for (int i = 0; i < graph.length; i++) {
            adjList.add(new ArrayList<>());
        }
        for (int i = 0; i < graph.length; i++) {
            for (int j = 0; j < graph[i].length; j++) {
                adjList.get(graph[i][j]).add(i);
                outdegree[i]++;
            }
        }
        this.res = new ArrayList<>();
        for (int i = 0; i < graph.length; i++) {
            if (outdegree[i] == 0 && !vis[i]) {
                dfs(adjList, i);
            }
        }
        Collections.sort(this.res);
        return this.res;
    }
    private void dfs(List<List<Integer>> adjList, int node) {
        res.add(node);
        vis[node] = true;
        for (int next : adjList.get(node)) {
            outdegree[next]--;
            if (outdegree[next] == 0) {
                if(!vis[next]) dfs(adjList, next);
            }
        }
    }
}
```

# Observation to solve it Using DFS

- Every node which leads to cycle by connecting to itself or to any node in visited path cannot be safe node.
- So if any node is a part of cycle or it leads to cycle , it cannot be a safe node.
- This observation can only be used in DFS

### Using DFS

```java
class Solution {
    List<Integer> res;
    boolean[] vis;
    boolean[] visPath;
    public List<Integer> eventualSafeNodes(int[][] graph) {
        this.vis = new boolean[graph.length];
        this.visPath = new boolean[graph.length];
        this.res = new ArrayList<>();
        for (int i = 0; i < graph.length; i++) {
            if (!vis[i]) {
                isCyclic(graph, i);
            }
        }
        Collections.sort(res);
        return this.res;
    }
    private boolean isCyclic(int[][] adjMat, int node) {
        vis[node] = true;
        visPath[node] = true;
        for (int next : adjMat[node]) {
            if (!vis[next] && isCyclic(adjMat, next)) {
                return true;
            } 
            else if (visPath[next]) {
                return true;
            }
        }
        this.res.add(node);
        visPath[node] = false;
        return false;
    }
}
```

```java
class Solution {
    boolean[] safe;
    boolean[] vis;
    boolean[] visPath;
    public List<Integer> eventualSafeNodes(int[][] graph) {
        this.vis = new boolean[graph.length];
        this.visPath = new boolean[graph.length];
        this.safe = new boolean[graph.length];
        List<Integer> res = new ArrayList<>();
        for (int i = 0; i < graph.length; i++) {
            if (!vis[i]) {
                isCyclic(graph, i);
            }
        }
        
        for (int i = 0; i < graph.length; i++) {
            if (safe[i]) {
                res.add(i);
            }
        }
        return res;
    }
    private boolean isCyclic(int[][] adjMat, int node) {
        vis[node] = true;
        visPath[node] = true;
        for (int next : adjMat[node]) {
            if (!vis[next] && isCyclic(adjMat, next)) {
                return true;
            } 
            else if (visPath[next]) {
                return true;
            }
        }
        this.safe[node] = true;
        visPath[node] = false;
        return false;
    }
}
```

```java
class Solution {
    public List<Integer> eventualSafeNodes(int[][] graph) {
        int V = graph.length;
        boolean[] vis = new boolean[V];
        boolean[] unSafe = new boolean[V];
        boolean[] path = new boolean[V];
        for (int i = 0; i < V; i++) {
            if (!vis[i]) {
                if (helper(graph, vis, unSafe, path, i)) unSafe[i] = true;
            }
        }
        List<Integer> res = new ArrayList<>();
        for (int u = 0; u < V ; u++) {
            if (!unSafe[u]) {
                res.add(u);
            }
        }
        return res;
    }

    private boolean helper(int[][] graph, boolean[] vis, boolean[] unSafe, boolean[] path, int u) {
        vis[u] = path[u] = true;
        for (int v : graph[u]) {
            if (!vis[v]) {
                if(helper(graph, vis, unSafe, path, v)) {
                    unSafe[v] = true;
                    return true;
                }
            } else if(path[v]) {
                return true;
            }
        }
        path[u] = false;
        return false;
    }
}
```

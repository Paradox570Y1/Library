>Multi Source BFS Problem


[994. Rotting Oranges](https://leetcode.com/problems/rotting-oranges/)

You are given an `m x n` `grid` where each cell can have one of three values:

- `0` representing an empty cell,
- `1` representing a fresh orange, or
- `2` representing a rotten orange.

Every minute, any fresh orange that is **4-directionally adjacent** to a rotten orange becomes rotten.

Return _the minimum number of minutes that must elapse until no cell has a fresh orange_. If _this is impossible, return_ `-1`.

**Example 1:**

![](https://assets.leetcode.com/uploads/2019/02/16/oranges.png)

**Input:** grid = `[[2,1,1],[1,1,0],[0,1,1]]`
**Output:** 4

**Example 2:**

**Input:** grid = `[[2,1,1],[0,1,1],[1,0,1]]`
**Output:** -1
**Explanation:** The orange in the bottom left corner (row 2, column 0) is never rotten, because rotting only happens 4-directionally.

**Example 3:**

**Input:** grid = `[[0,2]]`
**Output:** 0
**Explanation:** Since there are already no fresh oranges at minute 0, the answer is just 0.

**Constraints:**

- `m == grid.length`
- `n == grid[i].length`
- `1 <= m, n <= 10`
- `grid[i][j]` is `0`, `1`, or `2`.


# Using BFS

```java
class Solution {
    class Orange{
        int i, j;
        Orange(int i, int j) {
            this.i = i;
            this.j = j;
        }
    }
    private int bfs(int grid[][], Queue<Orange> rottenOranges) {
        int time = 0;
        int n = grid.length;
        int m = grid[0].length;
        while(!rottenOranges.isEmpty()) {
            time++;
            int size = rottenOranges.size();
            for(int k = 0; k < size; k++) {
                Orange orange = rottenOranges.poll();
                if(orange.j > 0 && grid[orange.i][orange.j - 1] == 1) {
                    grid[orange.i][orange.j - 1] = 2;
                    rottenOranges.offer(new Orange(orange.i, orange.j - 1));
                }
                if (orange.i > 0 && grid[orange.i - 1][orange.j] == 1) {
                    grid[orange.i - 1][orange.j] = 2;
                    rottenOranges.offer(new Orange(orange.i - 1, orange.j));
                }
                if (orange.j + 1 < m && grid[orange.i][orange.j + 1] == 1) {
                    grid[orange.i][orange.j + 1] = 2;
                    rottenOranges.offer(new Orange(orange.i, orange.j + 1));
                }
                if (orange.i + 1 < n && grid[orange.i + 1][orange.j] == 1) {
                    grid[orange.i + 1][orange.j] = 2;
                    rottenOranges.offer(new Orange(orange.i + 1, orange.j));
                }
            }
        }
        return time - 1;
    }
    public int orangesRotting(int[][] grid) {
        Queue<Orange> queue = new LinkedList<>();
        int fresh = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    fresh++;
                }
                else if (grid[i][j] == 2) {
                    queue.offer(new Orange(i, j));
                }
            }
        }
        if(fresh == 0) return 0;
        int res = bfs(grid, queue);

        for(int i = 0; i < grid.length; i++) {
            for(int j = 0; j < grid[0].length; j++) {
                if(grid[i][j] == 1) return -1;
            }
        }
        return res;
    }
}
```

```java
class Solution {
    class Orange{
        int i, j;
        Orange(int i, int j) {
            this.i = i;
            this.j = j;
        }
    }
    private int bfs(int grid[][], Queue<Orange> rottenOranges, int[] fresh) {
        int time = 0;
        int n = grid.length;
        int m = grid[0].length;
        int op[][] = {{0,1}, {1,0}, {-1,0}, {0,-1}};
        while(!rottenOranges.isEmpty()) {
            int size = rottenOranges.size();
            for(int k = 0; k < size; k++) {
                Orange orange = rottenOranges.poll();
                for (int t = 0; t < 4; t++) {
                    int p = orange.i + op[t][0];
                    int q = orange.j + op[t][1];
                    if(p >= 0 && p < n && q >= 0 && q < m && grid[p][q] == 1) {
                        fresh[0]--;
                        grid[p][q] = 2;
                        rottenOranges.offer(new Orange(p, q));
                    }
                }
            }
            if(!rottenOranges.isEmpty())time++;
        }
        return time;
    }
    public int orangesRotting(int[][] grid) {
        Queue<Orange> queue = new LinkedList<>();
        int fresh[] = new int[1];
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == 1) {
                    fresh[0]++;
                }
                else if (grid[i][j] == 2) {
                    queue.offer(new Orange(i, j));
                }
            }
        }
        if(fresh[0] == 0) return 0;
        if(queue.isEmpty())return -1;
        int res = bfs(grid, queue, fresh);
        return fresh[0] == 0 ? res: -1;
    }
}
```

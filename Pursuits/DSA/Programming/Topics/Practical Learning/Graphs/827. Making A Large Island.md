[827. Making A Large Island](https://leetcode.com/problems/making-a-large-island/)

You are given an `n x n` binary matrix `grid`. You are allowed to change **at most one** `0` to be `1`.

Return _the size of the largest **island** in_ `grid` _after applying this operation_.

An **island** is a 4-directionally connected group of `1`s.

**Example 1:**

**Input:** grid = `[[1,0],[0,1]]`
**Output:** 3
**Explanation:** Change one 0 to 1 and connect two 1s, then we get an island with area = 3.

**Example 2:**

**Input:** grid = `[[1,1],[1,0]]`
**Output:** 4
**Explanation:** Change the 0 to 1 and make the island bigger, only one island with area = 4.

**Example 3:**

**Input:** grid = `[[1,1],[1,1]]`
**Output:** 4
**Explanation:** Can't change any 0 to 1, only one island with area = 4.

**Constraints:**

- `n == grid.length`
- `n == grid[i].length`
- `1 <= n <= 500`
- `grid[i][j]` is either `0` or `1`.



# Using DisjointSet

```java
class Solution {
    class DisjointSet{
        int[] parent, size;
        DisjointSet(int n) {
            parent = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int findUParent(int node) {
            if (node == parent[node]) return node;
            return parent[node] = findUParent(parent[node]);
        }

        public void union(int u, int v) {
            int up = findUParent(u);
            int vp = findUParent(v);
            if (size[up] < size[vp]) {
                parent[up] = vp;
                size[vp] += size[up];
            } else {
                parent[vp] = up;
                size[up] += size[vp];
            }
        }
    }

    public int largestIsland(int[][] grid) {
        int n = grid.length;
        DisjointSet ds = new DisjointSet(n * n);
        boolean[][] vis = new boolean[n][n];
        int maxi = 1;
        for (int row = 0; row < n; row++) {
            for (int col = 0; col < n; col++) {
                if(grid[row][col] == 0) continue;
                int child = row * n + col;
                int prev = -1;
                if (col-1 >= 0 && grid[row][col-1] == 1) {
                    int parent = row * n + col-1;
                    ds.union(parent, child);
                    maxi = Math.max(maxi, ds.size[ds.findUParent(parent)]);
                }
                if (row-1 >= 0 && grid[row-1][col] == 1) {
                    int parent = (row-1) * n + col;
                    if (ds.findUParent(parent) != ds.findUParent(child)){
                        ds.union(parent, child);
                        maxi = Math.max(maxi, ds.size[ds.findUParent(parent)]);
                    }
                }
            }
        }

        int dirs[][] = {{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
        for (int row = 0; row < n; row++) {
            for (int col = 0; col < n; col++) { 
                if(grid[row][col] == 1) continue;
                int count = 1;
                List<Integer> visP = new ArrayList<>();
                for (int[] dir : dirs) {
                    int x = dir[0] + row;
                    int y = dir[1] + col;
                    if (isValid(x, y, n) && grid[x][y] == 1) {
                        int node = x * n + y;
                        int parent = ds.findUParent(node);
                        if(!visP.contains(parent)) {
                            count += ds.size[parent];
                            visP.add(parent);
                        }
                    }
                }
                maxi = Math.max(maxi, count);
            }
        }
        return maxi;
    }

    private boolean isValid(int x, int y, int n) {
        return x >= 0 && y >= 0 && x < n && y < n;
    }
}
```

```java
class Solution {
    private int[] parent, size;
    private int[][] dirs = {{-1, 0}, {0, -1}};
    private int n;
    private int node(int i, int j) {
        return i * n + j;
    }
    private int find(int node) {
        if(node == parent[node]) return node;
        return parent[node] = find(parent[node]);
    }
    private void union(int u , int v) {
        int pu = find(u);
        int pv = find(v);
        if (size[pu] < size[pv]) {
            parent[pu] = pv;
            size[pv] += size[pu];
        } else {
            parent[pv] = pu;
            size[pu] += size[pv];
        }
    }
    public int largestIsland(int[][] grid) {
        n = grid.length;
        int V = n * n;
        parent = new int[V];
        size = new int[V];
        for (int i = 0; i < V; i++) {
            parent[i] = i;
            size[i] = 0;
        }
        int res = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    int u = node(i, j);
                    size[u] = 1;
                    for (int[] dir : dirs) {
                        int x = i + dir[0], y = j + dir[1];
                        if (x >= 0 && y >= 0 && x < n && y < n && grid[x][y] == 1) {
                            int v = node(x,y);
                            if(find(u) != find(v)) {
                                union(u, v);
                            }
                        }
                    }
                    res = Math.max(res, size[find(u)]);
                }
            }
        }
        dirs = new int[][]{{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 0) {
                    int area = 1;
                    Set<Integer> st = new HashSet<>();
                    for (int[] dir : dirs) {
                        int x = dir[0] + i, y = dir[1] + j;
                        if (x >= 0 && x < n && y >= 0 && y < n && grid[x][y] == 1) {
                            int pv = find(node(x, y));
                            if(!st.contains(pv)) {
                                area += size[pv];
                                st.add(pv);
                            }
                        }
                    }
                    res = Math.max(res, area);
                }
            }
        }
        return res;
    }
}
```

```java
class Solution {
    class DisjointSet{
        int[] parent, size;
        DisjointSet(int n) {
            parent = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int findUParent(int node) {
            if (node == parent[node]) return node;
            return parent[node] = findUParent(parent[node]);
        }

        public void union(int u, int v) {
            int up = findUParent(u);
            int vp = findUParent(v);
            if(up == vp)return; // to prevent from adding same component again
            if (size[up] < size[vp]) {
                parent[up] = vp;
                size[vp] += size[up];
            } else {
                parent[vp] = up;
                size[up] += size[vp];
            }
        }
    }

    public int largestIsland(int[][] grid) {
        int n = grid.length;
        DisjointSet ds = new DisjointSet(n * n);
        int dirs[][] = {{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
        for (int row = 0; row < n; row++) {
            for (int col = 0; col < n; col++) {
                if(grid[row][col] == 0) continue;
                int node = row * n + col;
                for (int[] dir : dirs) {
                    int x = dir[0] + row;
                    int y = dir[1] + col;
                    if (isValid(x, y, n) && grid[x][y] == 1) {
                        int adjNode = x * n + y;
                        ds.union(node, adjNode);
                    }
                }
            }
        }
        int maxi = 0;
        
        for (int row = 0; row < n; row++) {
            for (int col = 0; col < n; col++) { 
                if(grid[row][col] == 1) continue;
                Set<Integer> visP = new HashSet<>();
                for (int[] dir : dirs) {
                    int x = dir[0] + row;
                    int y = dir[1] + col;
                    if (isValid(x, y, n) && grid[x][y] == 1) {
                        int node = x * n + y;
                        visP.add(ds.findUParent(node));
                    }
                }
                int size = 0;
                for (int comp : visP) {
                    size += ds.size[comp];
                }
                maxi = Math.max(maxi, size + 1);
            }
        }
        for (int cell = 0; cell < n*n; cell++) {
            maxi = Math.max(maxi, ds.size[cell]);
        }
        return maxi;
    }

    private boolean isValid(int x, int y, int n) {
        return x >= 0 && y >= 0 && x < n && y < n;
    }
}
```

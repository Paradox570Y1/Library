[130. Surrounded Regions](https://leetcode.com/problems/surrounded-regions/)

You are given an `m x n` matrix `board` containing **letters** `'X'` and `'O'`, **capture regions** that are **surrounded**:

- **Connect**: A cell is connected to adjacent cells horizontally or vertically.
- **Region**: To form a region **connect every** `'O'` cell.
- **Surround**: The region is surrounded with `'X'` cells if you can **connect the region** with `'X'` cells and none of the region cells are on the edge of the `board`.

To capture a **surrounded region**, replace all `'O'`s with `'X'`s **in-place** within the original board. You do not need to return anything.

**Example 1:**

**Input:** board = `[["X","X","X","X"],["X","O","O","X"],["X","X","O","X"],["X","O","X","X"]]`

**Output:** `[["X","X","X","X"],["X","X","X","X"],["X","X","X","X"],["X","O","X","X"]]`

**Explanation:**

![300](https://assets.leetcode.com/uploads/2021/02/19/xogrid.jpg)

In the above diagram, the bottom region is not captured because it is on the edge of the board and cannot be surrounded.

**Example 2:**

**Input:** board = `[["X"]]`

**Output:** `[["X"]]`

**Constraints:**

- `m == board.length`
- `n == board[i].length`
- `1 <= m, n <= 200`
- `board[i][j]` is `'X'` or `'O'`.


# Using BFS

```java
class Solution {
    int rows, cols;
    int ops[][];
    class Region {
        int x;
        int y;
        Region(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
    public void solve(char[][] board) {
        this.rows = board.length;
        this.cols = board[0].length;
        ops = new int[][]{{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
        Queue<Region> queue = new LinkedList<>();
        for (int i = 0; i < rows; i++) {
            if (board[i][0] == 'O') {
                board[i][0] = '#';
                queue.offer(new Region(i, 0));
            }
            if (board[i][cols-1] == 'O') {
                board[i][cols-1] = '#';
                queue.offer(new Region(i, cols-1));
            }
        }

        for (int j = 1; j < cols-1; j++) {
            if (board[0][j] == 'O') {
                board[0][j] = '#';
                queue.offer(new Region(0, j));
            }
            if (board[rows-1][j] == 'O') {
                board[rows-1][j] = '#';
                queue.offer(new Region(rows-1, j));
            }
        }
        bfs(board, queue);
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (board[i][j] == 'O') {
                    board[i][j] = 'X';
                }
                else if (board[i][j] == '#') {
                    board[i][j] = 'O';
                }
            }
        }
    }
    private void bfs(char[][] board, Queue<Region> queue) {
        while(!queue.isEmpty()) {
            Region cur = queue.poll();
            for (int[] op:ops) {
                int i = op[0] + cur.x;
                int j = op[1] + cur.y;
                if (i >= 0 && j >= 0 && i < rows && j < cols && board[i][j] == 'O') {
                    board[i][j] = '#';
                    queue.offer(new Region(i, j));
                }
            }
        }
    }
}
```

# Using DFS + Queue

```java
class Solution {
    int rows, cols;
    int op[][];
    Boolean vis[][];
    Queue<int[]> queue;
    public void solve(char[][] board) {
        this.rows = board.length;
        this.cols = board[0].length;

        vis = new Boolean[rows][cols];
        op = new int[][]{{0, -1}, {-1, 0}, {0, 1}, {1, 0}};
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (board[i][j] == 'O' && vis[i][j] == null) {
                    this.queue = new LinkedList<>();
                    if (helper(board, i, j)) {
                        captureRegion(board, queue);
                    }
                }
            }
        }
    }
    private boolean helper(char[][] board, int x, int y) {
        if (x < 0 || y < 0 || x == rows || y == cols) {
            return false;
        }
        if(board[x][y] == 'X') return true;
        if (vis[x][y] != null) {
            return vis[x][y];
        }
        boolean isSurrounded = true;
        vis[x][y] = true;
        for (int i = 0; i < 4 && isSurrounded; i++) {
            int p = x + op[i][0];
            int q = y + op[i][1];
            if(!helper(board, p, q)) {
                isSurrounded = false;
            }
        }
        if(isSurrounded) {
            queue.offer(new int[]{x, y});
        } else {
            vis[x][y] = false;
        }
        return isSurrounded;
    }
    private void captureRegion(char[][] board, Queue<int[]> queue) {
        while(!queue.isEmpty()) {
            int[] ele = queue.poll();
            board[ele[0]][ele[1]]  = 'X';
        }
    }
}
```

# Using DFS with reverse Strategy
- Instead of find what spots we can capture find what we can not capture.

✅ **Summary**:

- Your solution is **DFS + visited marking**, **not memoization**.
- It’s correct, simple, and runs in `O(m×n)` time.
- Downsides: recursion depth limit, extra boolean matrix, and minor inefficiency in initializing `canCapture`.
- **Cons:** Risk of stack overflow for large grids.
```java
class Solution {
    int rows, cols;
    int op[][];
    public void solve(char[][] board) {
        this.rows = board.length;
        this.cols = board[0].length;
        boolean[][] canCapture = new boolean[rows][cols];
        op = new int[][]{{0, -1}, {-1, 0}, {0, 1}, {1, 0}};

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                canCapture[i][j] = true;
            }
        }

        for (int i = 0; i < rows; i++) {
            if (board[i][0] == 'O') {
                dfs(board, canCapture, i, 0);
            }
            if (board[i][cols-1] == 'O') {
                dfs(board, canCapture, i, cols-1);
            }
        }

        for (int j = 1; j < cols-1; j++) {
            if (board[0][j] == 'O') {
                dfs(board, canCapture, 0, j);
            }
            if (board[rows-1][j] == 'O') {
                dfs(board, canCapture, rows-1, j);
            }
        }

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if ((board[i][j] == 'O') && canCapture[i][j]) {
                    board[i][j] = 'X';
                }
            }
        }
    }
    private void dfs(char[][] board, boolean[][] canCapture, int x, int y) {
        if (x < 0 || y < 0 || x == rows || y == cols || board[x][y] == 'X' || !canCapture[x][y]) {
            return;
        }
        canCapture[x][y] = false;
        for (int i = 0; i < 4; i++) {
            dfs(board, canCapture, x+op[i][0], y + op[i][1]);
        }
    }
}
```

#### 1. Scope of Variables

The variables `rows`, `cols`, and `ops` are declared as class members. However, their values are set within the `solve` method and are not used by any other method in a way that requires them to be instance variables.

- **Suggestion💡**: Make them local to the `solve` method or pass them as parameters to the `bfs` helper. This improves encapsulation and makes the `Solution` class stateless, which is generally better design.

#### 2. The `Region` Inner Class

Your `Region` class works perfectly, but it adds a bit of overhead. For a simple coordinate pair, you can avoid creating custom objects in a few ways:

- **Suggestion 1 (Simple)**: Use a `Queue<int[]>` and store coordinates as `new int[]{row, col}`. This is a very common pattern in competitive programming that avoids the boilerplate of a custom class.
    
- **Suggestion 2 (Modern Java)**: If you're using Java 16+, a `record` is a more concise way to define a simple data carrier: `private record Region(int x, int y) {}`.
    
- **Suggestion 3 (High Performance)**: For maximum performance and zero object creation in the queue, you can encode the coordinates into a single integer: `int id = row * cols + col;`. Then, you can decode it back inside the loop: `int r = id / cols; int c = id % cols;`. This is faster but less readable.
    

The `int[]` approach offers the best balance of performance and readability.

#### 3. Redundant Border Checks

The loops for checking the borders can be slightly condensed to avoid re-checking the corners, although this is a very minor point and your current code is perfectly fine.


```java
class Solution {
    // The direction array can be a static final constant.
    private static final int[][] DIRECTIONS = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};

    public void solve(char[][] board) {
        if (board == null || board.length == 0) {
            return;
        }

        int rows = board.length;
        int cols = board[0].length;
        
        // Using Queue<int[]> is more direct than a custom Region class.
        Queue<int[]> queue = new LinkedList<>();

        // 1. Identify all 'O's on the border and add to the queue.
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // Check if the cell is on the border and is 'O'
                if ((i == 0 || i == rows - 1 || j == 0 || j == cols - 1) && board[i][j] == 'O') {
                    board[i][j] = '#'; // Mark as safe
                    queue.offer(new int[]{i, j});
                }
            }
        }
        
        // 2. Perform BFS to mark all connected 'O's as safe.
        bfs(board, queue, rows, cols);

        // 3. Flip remaining 'O's to 'X' and revert '#' back to 'O'.
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (board[i][j] == 'O') {
                    board[i][j] = 'X'; // Surrounded
                } else if (board[i][j] == '#') {
                    board[i][j] = 'O'; // Safe
                }
            }
        }
    }
    
    // Helper method now takes board dimensions as parameters.
    private void bfs(char[][] board, Queue<int[]> queue, int rows, int cols) {
        while (!queue.isEmpty()) {
            int[] current = queue.poll();
            int r = current[0];
            int c = current[1];

            for (int[] dir : DIRECTIONS) {
                int newRow = r + dir[0];
                int newCol = c + dir[1];

                // Check boundaries and if the new cell is an unflippable 'O'
                if (newRow >= 0 && newRow < rows && newCol >= 0 && newCol < cols && board[newRow][newCol] == 'O') {
                    board[newRow][newCol] = '#';
                    queue.offer(new int[]{newRow, newCol});
                }
            }
        }
    }
}
```


# Most Optimal

- Mark whole grid capture and then start DFS from outer edges to mark not capturable region.
```java
class Solution {
    public void solve(char[][] board) {
        if(board==null || board.length==1 || board[0].length==1) return;
        int m=board.length;
        int n=board[0].length;
        char out[][]= new char[m][n];
        for(char row[] : out){
            Arrays.fill(row,'X');
        }
        // left
        for(int j=0;j<n;j++){
            if(board[0][j]=='O') dfs(board,0,j,out);
        }
        // right
        for(int j=0;j<n;j++){
            if(board[m-1][j]=='O') dfs(board,m-1,j,out);
        }
        // top
        for(int j=0;j<m;j++){
            if(board[j][0]=='O') dfs(board,j,0,out);
        }
        //bottom
        for(int j=0;j<m;j++){
            if(board[j][n-1]=='O') dfs(board,j,n-1,out);
        }
        for (int i=0;i<m;i++){
            board[i]=out[i];
        }
    }
    void dfs(char b[][], int i, int j, char out[][]){
        if(i<0 || j<0 || i>=b.length || j>=b[0].length || b[i][j]!='O') return;
        out[i][j]='O';
        b[i][j]='1';
        dfs(b,i+1,j,out);
        dfs(b,i-1,j,out);
        dfs(b,i,j+1,out);
        dfs(b,i,j-1,out);
    }
}
```

[787. Cheapest Flights Within K Stops](https://leetcode.com/problems/cheapest-flights-within-k-stops/)

There are `n` cities connected by some number of flights. You are given an array `flights` where `flights[i] = [fromi, toi, pricei]` indicates that there is a flight from city `fromi` to city `toi` with cost `pricei`.

You are also given three integers `src`, `dst`, and `k`, return _**the cheapest price** from_ `src` _to_ `dst` _with at most_ `k` _stops._ If there is no such route, return `-1`.

```java
class Solution {
    class Flight {
        int id, price, stops;
        Flight(int id, int price, int stops) {
            this.id = id;
            this.price = price;
            this.stops = stops;
        }
    }

    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        List<List<int[]>> adjList = new ArrayList<>();
        for (int i = 0; i < n; i++) adjList.add(new ArrayList<>());
        for (int[] f : flights) adjList.get(f[0]).add(new int[]{f[1], f[2]});

        int[] prices = new int[n];
        Arrays.fill(prices, (int)1e9);

        PriorityQueue<Flight> pq = new PriorityQueue<>((a, b) -> a.price - b.price);
        pq.offer(new Flight(src, 0, 0));
        prices[src] = 0;

        while (!pq.isEmpty()) {
            Flight cur = pq.poll();
            if (cur.id == dst) return cur.price;
            if (cur.stops > k) continue;

            for (int[] next : adjList.get(cur.id)) {
                int newPrice = cur.price + next[1];
                if (newPrice < prices[next[0]] || cur.stops < k) {
                    pq.offer(new Flight(next[0], newPrice, cur.stops + 1));
                    prices[next[0]] = newPrice;
                }
            }
        }
        return -1;
    }
}
```

```java
class Solution {
    class Pair implements Comparable<Pair>{
        int city;
        int cost;
        int stops;
        Pair(int city, int cost, int stops) {
            this.city = city;
            this.cost = cost;
            this.stops = stops;
        }

        @Override
        public int compareTo(Pair other) {
            return Integer.compare(this.cost, other.cost);
        }
    }
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        k=k+1;
        List<List<int[]>> graph = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            graph.add(new ArrayList<>());
        }
        for (int[] flight : flights) {
            int from = flight[0];
            int to = flight[1];
            int price = flight[2];
            graph.get(from).add(new int[]{to, price});
        }

        PriorityQueue<Pair> pq = new PriorityQueue<>();
        pq.offer(new Pair(src, 0, 0));
        int[][] Cost = new int[n][k+1];
        for (int i = 0; i < n; i++) {
            Arrays.fill(Cost[i], (int) 1e9);
        }
        Cost[src][0] = 0;
        while (!pq.isEmpty()) {
            Pair cur = pq.poll();
            int from = cur.city;
            int cost = cur.cost;
            if (from == dst) return cost;
            if (cur.stops == k) continue;
            for (int[] next : graph.get(cur.city)) {
                int to = next[0];
                int price = next[1];
                int stops = cur.stops + 1;
                if (cost + price < Cost[to][stops]) {
                    Cost[to][stops] = cost + price;
                    pq.offer(new Pair(to, Cost[to][stops], cur.stops + 1));
                }
            }
        }
        return -1;
    }
}
```

```java
class Solution {
    class Flight{
        int id;
        int price;
        int stops;
        Flight(int id, int price, int stops) {
            this.id = id;
            this.price = price;
            this.stops = stops;
        }
    }
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        List<List<int[]>> adjList = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
        }

        for (int[] flight : flights) {
            int from = flight[0];
            int to = flight[1];
            int price = flight[2];
            adjList.get(from).add(new int[]{to, price});
        }

        int prices[][] = new int[n][k+1];
        for (int i = 0; i < n; i++) {
            Arrays.fill(prices[i], (int)1e9);
        }

        PriorityQueue<Flight> pq = new PriorityQueue<>((a, b) -> a.price - b.price);
        pq.offer(new Flight(src, 0, k + 1));
        prices[src][k] = 0;

        while (!pq.isEmpty()) {
            Flight cur = pq.poll();
            if (cur.stops == 0) continue;
            for (int[] next : adjList.get(cur.id)) {
                int price = cur.price + next[1];
                if (price < prices[next[0]][cur.stops - 1]) {
                    prices[next[0]][cur.stops - 1] = price;
                    pq.offer(new Flight(next[0], price, cur.stops - 1));
                }
            }
        }
        int cheapestFlight = prices[dst][0];
        for (int i = 1; i <= k; i++) {
            cheapestFlight = Math.min(cheapestFlight, prices[dst][i]);
        }
        if (cheapestFlight != (int) 1e9) return cheapestFlight;
        return -1;
    }
}
```

# Using Dijkstra Algorithm on Stops

- In this problem there could be cases where we reach at particular stop in cheaper price early but that was the last city we can visit in that route and we stop in that route. But now if i arrive at the same stop again through another route  with higher price then it will not let me pass through it but in this new route i have taken less stops and hence i can move next to the stop where i just arrived which could be the destination. This is where our normal logic fails.
- The issue comes when a **more expensive path with fewer stops** can later lead to a cheaper destination. (in a nut shell)
- So take stops as the criteria to create priority queue.
- In this approach we are mimicking level order traversal using dijkstra which allows it till k stops and minimize prices for all cities during the process.
- But putting cap at the level is only what we need so as long as we are able to do that  it doesn't matter if we go in level order or not.
- Since the stops are increasing at constant rate of 1, so they are basically unit weight and general BFS will suffice

```java
class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        int[] cost = new int[n];
        Arrays.fill(cost, (int) 1e9);
        List<List<int[]>> adjList = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
        }

        for (int[] flight : flights) {
            int from = flight[0];
            int to = flight[1];
            int price = flight[2];
            adjList.get(from).add(new int[]{to, price});
        }

        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{src, 0});
        int stops = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            if (stops == k+1) break;
            while (size-->0) {
                int[] cur = queue.poll();
                int u = cur[0];
                int pu = cur[1];
                for (int[] next : adjList.get(u)) {
                    int v = next[0];
                    int w = next[1];
                    if (cost[v] > pu + w) {
                        cost[v] = pu + w;
                        queue.offer(new int[]{v, cost[v]});
                    }
                }
            }
            stops++;
        }
        if (cost[dst] == (int)1e9) return -1;
        return cost[dst];
    }
}
```

```java
class Solution {
    class Flight{
        int id;
        int price;
        int stops;
        Flight(int id, int price, int stops) {
            this.id = id;
            this.price = price;
            this.stops = stops;
        }
    }
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        int[][] adjMatrix = new int[n][n];

        for (int[] flight : flights) {
            int from = flight[0];
            int to = flight[1];
            int price = flight[2];
            adjMatrix[from][to] = price;
        }

        int prices[] = new int[n];
        Arrays.fill(prices, (int)1e9);

        Queue<Flight> q = new LinkedList<>();
        q.offer(new Flight(src, 0, 0));
        prices[src] = 0;

        while (!q.isEmpty()) {
            Flight cur = q.poll();
            if (cur.stops == k+1) {
                continue;
            }
            for (int v = 0; v < n; v++) {
                int w = adjMatrix[cur.id][v];
                if (w != 0) {
                    int price = cur.price + w;
                    if (price < prices[v]) {
                        prices[v] = price;
                        q.offer(new Flight(v, price, cur.stops + 1));
                    }
                }
            }
        }
        if(prices[dst] != (int)1e9) return prices[dst];
        return -1;
    }
}
```

# Directly using institution of Bellman ford algorithm

```java
class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        Integer[] prices = new Integer[n];
        prices[src] = 0;
        for (int i = 0; i < k+1; i++) {
            Integer[] temp = prices.clone();
            for (int[] flight : flights) {
                int source = flight[0];
                int dest = flight[1];
                int price = flight[2];
                if (prices[source] != null && (temp[dest] == null || prices[source] + price < temp[dest])) {
                    temp[dest] = price + prices[source];
                }
            }
            prices = temp;
        }
        if(prices[dst] == null) return -1;
        return prices[dst];
    }
}
```

```java
class Solution {
    public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
        int[] dist = new int[n];
        Arrays.fill(dist, (int) 1e8);
        dist[src] = 0;
        for (int i = 0; i <= k; i++) {
            int[] buffer = dist.clone();
            for (int[] flight : flights) {
                int from = flight[0], to = flight[1], price = flight[2];
                buffer[to] = Math.min(buffer[to], dist[from] + price);
            }
            dist = buffer;
        }
        if (dist[dst] == (int) 1e8) return -1;
        return dist[dst];
    }
}
```

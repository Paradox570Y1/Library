[542. 01 Matrix](https://leetcode.com/problems/01-matrix/)

Given an `m x n` binary matrix `mat`, return _the distance of the nearest_ `0` _for each cell_.

The distance between two cells sharing a common edge is `1`.

**Example 1:**

![](https://assets.leetcode.com/uploads/2021/04/24/01-1-grid.jpg)

**Input:** mat = `[[0,0,0],[0,1,0],[0,0,0]]`
**Output:** `[[0,0,0],[0,1,0],[0,0,0]]`

**Example 2:**

![](https://assets.leetcode.com/uploads/2021/04/24/01-2-grid.jpg)

**Input:** mat = `[[0,0,0],[0,1,0],[1,1,1]]`
**Output:** `[[0,0,0],[0,1,0],[1,2,1]]`

**Constraints:**

- `m == mat.length`
- `n == mat[i].length`
- `1 <= m, n <= 104`
- `1 <= m * n <= 104`
- `mat[i][j]` is either `0` or `1`.
- There is at least one `0` in `mat`.


>NOTE: The "distance to nearest zero" is a **shortest path problem on an unweighted grid**.
# Using BFS
Gives TLE
```java
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int n = mat.length;
        int m = mat[0].length;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if(mat[i][j] == 1) {
                    mat[i][j] = nearestZero(mat, i, j, n, m);
                }
            }
        }
        return mat;
    }
    private int nearestZero(int[][] mat, int i, int j, int n, int m) {
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{i, j});
        boolean vis[][] = new boolean[n][m];
        int op[][] = {{0,1},{1,0},{-1,0},{0,-1}};
        int dist = 0;
        while(!queue.isEmpty()) {
            dist++;
            int size = queue.size();
            for (int a = 0; a < size; a++) {
                int[] cur = queue.poll();
                for (int b = 0; b < 4; b++) {
                    int x = cur[0] + op[b][0];
                    int y = cur[1] + op[b][1];
                    if (x>=0 && y>=0 && x<n && y<m && !vis[x][y]){
                        if(mat[x][y] == 0) {
                            return dist;
                        } else {
                            vis[x][y] = true;
                            queue.offer(new int[]{x, y});
                        }
                    }
                }
            }
        }
        return dist;
    }
}
```

```java
class Solution {
    class Cell{
        int x;
        int y;
        Cell(int x, int y) {
            this.x = x;
            this.y = y;
        }
    }
    public int[][] updateMatrix(int[][] mat) {
        int rows = mat.length;
        int cols = mat[0].length;
        boolean[][] vis = new boolean[rows][cols];
        Queue<Cell> queue = new LinkedList<>();

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (mat[i][j] == 0) {
                    vis[i][j] = true;
                    queue.offer(new Cell(i, j));
                }
            }
        }
        int ops[][] = {{0, 1}, {1, 0}, {-1, 0}, {0, -1}};
        while(!queue.isEmpty()) {
            Cell cur = queue.poll();
            for (int[] op:ops){
                int p = op[0] + cur.x;
                int q = op[1] + cur.y;
                if (p >= 0 && q >= 0 && p < rows && q < cols && !vis[p][q] && mat[p][q] == 1) {
                    vis[p][q] = true;
                    mat[p][q] = mat[cur.x][cur.y] + 1;
                    queue.offer(new Cell(p, q));
                }
            }
        }
        return mat;
    }
}
```

# Using DP

```java
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int rows = mat.length;
        int cols = mat[0].length;
        int dp[][] = new int[rows][cols];
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (mat[i][j] == 1) {
                    dp[i][j] = 10001;
                } 
            }
        }

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                if (mat[i][j] == 1) {
                    if (i > 0) {
                        dp[i][j] = Math.min(dp[i][j], dp[i-1][j] + 1);
                    }
                    if (j > 0) {
                        dp[i][j] = Math.min(dp[i][j], dp[i][j-1] + 1);
                    }
                }
            }
        }

        for (int i = rows-1; i >= 0; i--) {
            for (int j = cols-1; j >= 0; j--) {
                if (mat[i][j] == 1) {
                    if (i < rows-1) {
                        dp[i][j] = Math.min(dp[i][j], dp[i+1][j] + 1);
                    }
                    if (j < cols-1) {
                        dp[i][j] = Math.min(dp[i][j], dp[i][j+1] + 1);
                    }
                }
            }
        }
        return dp;
    }
}
```

```java
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int n = mat.length;
        int m = mat[0].length;
        int dp[][] = new int[n][m];
        int INF = n+m;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                dp[i][j] = (mat[i][j] == 1)? INF:0;
            }
        }

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 1) {
                    if (i-1 >= 0) dp[i][j] = Math.min(dp[i][j], dp[i-1][j] + 1);
                    if (j-1 >= 0) dp[i][j] = Math.min(dp[i][j], dp[i][j-1] + 1);
                }
            }
        }
        for (int i = n-1; i >= 0; i--) {
            for (int j = m-1; j >= 0; j--) {
                if (mat[i][j] == 1) {
                    if (i+1 < n) dp[i][j] = Math.min(dp[i][j], dp[i+1][j] + 1);
                    if (j+1 < m) dp[i][j] = Math.min(dp[i][j], dp[i][j+1] + 1);
                }
            }
        }
        return dp;
    }
}
```

```java
class Solution {
    public int[][] updateMatrix(int[][] mat) {
        int n = mat.length, m = mat[0].length;
        int[][] res = new int[n][m];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (mat[i][j] == 0) continue;
                int val = 10001;
                if (i-1 >= 0) val = Math.min(val, res[i-1][j]+1);
                if (j-1 >= 0) val = Math.min(val, res[i][j-1]+1);
                res[i][j] = val;
            }
        }
        for (int i = n-1; i >= 0; i--) {
            for (int j = m-1; j >= 0; j--) {
                if (mat[i][j] == 0) continue;
                int val = res[i][j];
                if (i+1 < n) val = Math.min(val, res[i+1][j]+1);
                if (j+1 < m) val = Math.min(val, res[i][j+1]+1);
                res[i][j] = val;
            }
        }

        return res;
    }
}
```

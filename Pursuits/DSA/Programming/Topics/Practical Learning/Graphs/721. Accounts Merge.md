[721. Accounts Merge](https://leetcode.com/problems/accounts-merge/)

Given a list of `accounts` where each element `accounts[i]` is a list of strings, where the first element `accounts[i][0]` is a name, and the rest of the elements are **emails** representing emails of the account.

Now, we would like to merge these accounts. Two accounts definitely belong to the same person if there is some common email to both accounts. Note that even if two accounts have the same name, they may belong to different people as people could have the same name. A person can have any number of accounts initially, but all of their accounts definitely have the same name.

After merging the accounts, return the accounts in the following format: the first element of each account is the name, and the rest of the elements are emails **in sorted order**. The accounts themselves can be returned in **any order**.

**Example 1:**

**Input:** accounts = [["John","johnsmith@mail.com","john_newyork@mail.com"],["John","johnsmith@mail.com","john00@mail.com"],["Mary","mary@mail.com"],["John","johnnybravo@mail.com"]]
**Output:** [["John","john00@mail.com","john_newyork@mail.com","johnsmith@mail.com"],["Mary","mary@mail.com"],["John","johnnybravo@mail.com"]]
**Explanation:**
The first and second John's are the same person as they have the common email "johnsmith@mail.com".
The third John and Mary are different people as none of their email addresses are used by other accounts.
We could return these lists in any order, for example the answer [['Mary', 'mary@mail.com'], ['John', 'johnnybravo@mail.com'], 
['John', 'john00@mail.com', 'john_newyork@mail.com', 'johnsmith@mail.com']] would still be accepted.

**Example 2:**

**Input:** accounts = [["Gabe","Gabe0@m.co","Gabe3@m.co","Gabe1@m.co"],["Kevin","Kevin3@m.co","Kevin5@m.co","Kevin0@m.co"],["Ethan","Ethan5@m.co","Ethan4@m.co","Ethan0@m.co"],["Hanzo","Hanzo3@m.co","Hanzo1@m.co","Hanzo0@m.co"],["Fern","Fern5@m.co","Fern1@m.co","Fern0@m.co"]]
**Output:** [["Ethan","Ethan0@m.co","Ethan4@m.co","Ethan5@m.co"],["Gabe","Gabe0@m.co","Gabe1@m.co","Gabe3@m.co"],["Hanzo","Hanzo0@m.co","Hanzo1@m.co","Hanzo3@m.co"],["Kevin","Kevin0@m.co","Kevin3@m.co","Kevin5@m.co"],["Fern","Fern0@m.co","Fern1@m.co","Fern5@m.co"]]

**Constraints:**

- `1 <= accounts.length <= 1000`
- `2 <= accounts[i].length <= 10`
- `1 <= accounts[i][j].length <= 30`
- `accounts[i][0]` consists of English letters.
- `accounts[i][j] (for j > 0)` is a valid email.

class Solution {
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        HashMap<Integer, TreeSet<String>> store = new HashMap<>();
        HashMap<String, List<Integer>> sharedName = new HashMap<>();
        int parent[] = new int[accounts.size()];
        for (int i = 0; i < accounts.size(); i++) {
            parent[i] = i;
        }
        for (int i = 0; i < accounts.size(); i++) {
            List<String> account = accounts.get(i);
            String name = account.get(0);
            if (sharedName.containsKey(name)) {
                List<Integer> conflictedAccounts = sharedName.get(name);
                boolean alreadyAdded = false;
                for (int t = 0; t < conflictedAccounts.size(); t++) {
                    int acc_id = conflictedAccounts.get(t);
                    TreeSet<String> mailSet = store.get(acc_id);
                    boolean common = false;
                    for (int j = 1; j < account.size(); j++) {
                        if(mailSet.contains(account.get(j))) {
                            if(!alreadyAdded) {
                                mailSet.addAll(account.subList(1, account.size()));
                                parent[i] = acc_id;
                                alreadyAdded = true;
                            } else {
                                store.get(parent[i]).addAll(mailSet);
                                store.remove(acc_id);
                                conflictedAccounts.remove(t);
                                t--;
                                parent[acc_id] = parent[i];
                            }
                            break;
                        }
                    }
                }
                if (!alreadyAdded) {
                    TreeSet<String> st = new TreeSet<>();
                    st.addAll(account.subList(1, account.size()));
                    sharedName.get(name).add(i);
                    store.put(parent[i], st);
                }
            } else {
                sharedName.put(name, new ArrayList<>());
                sharedName.get(name).add(i);
                TreeSet<String> st = new TreeSet<>();
                st.addAll(account.subList(1, account.size()));
                store.put(i, st);
            }
        }
        List<List<String>> res = new ArrayList<>();
        for (int i = 0; i < accounts.size(); i++) {
            if (parent[i] == i) {
                List<String> list = new ArrayList<>();
                list.add(accounts.get(i).get(0));
                TreeSet<String> st = store.get(i);
                if(st != null) list.addAll(st);
                res.add(list);
            }
        }
        return res;
    }
}

class Solution {
    int[] parent, size;
    private int find(int node) {
        if (parent[node] == node) return node;
        return parent[node] = find(parent[node]);
    }
    private void union(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (size[pu] < size[pv]) {
            parent[pu] = pv;
            size[pv]++;
        } else {
            parent[pv] = pu;
            size[pu]++;
        }
    }
    private boolean isRelated(List<String> acc1, List<String> acc2) {
        Set<String> acc = new HashSet<>();
        for (int i = 1; i < acc1.size(); i++) {
            acc.add(acc1.get(i));
        }
        for (int i = 1; i < acc2.size(); i++) {
            if (acc.contains(acc2.get(i))) return true;
        }
        return false;
    }
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        int n = accounts.size();
        parent = new int[n];
        size = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            size[i] = 1;
        }

        for (int i = 1;  i < n; i++) {
            int v = i;
            for (int j = 0; j < i; j++) {
                int u = j;
                if (isRelated(accounts.get(j), accounts.get(i))) {
                    union(u, v);
                }
            }
        }
        List<Set<String>> temp = new ArrayList<>();
        for (int i = 0; i < n; i++) temp.add(new HashSet<>());
        for (int i = 0; i < n; i++) {
            Set<String> cur = temp.get(find(i));
            List<String> account = accounts.get(i);
            for (int j = 1; j < account.size(); j++) {
                cur.add(account.get(j));
            }
        }
        List<List<String>> res = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            if (temp.get(i).size() == 0) continue;
            res.add(new ArrayList<>());
            res.get(res.size()-1).add(accounts.get(find(i)).get(0));
            List<String> buffer = new ArrayList<>(temp.get(i));
            Collections.sort(buffer);
            res.get(res.size()-1).addAll(buffer);
        }
        return res;
    }
}

class Solution {
    private int[] parent;
    private boolean canMerge(TreeSet<String> oldAcc, List<String> newAcc) {
        for (int i = 1; i < newAcc.size(); i++) {
            String mail = newAcc.get(i);
            if (oldAcc.contains(mail)) {
                return true;
            }
        }
        return false;
    }
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        int n = accounts.size();
        parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i;
        Map<Integer, TreeSet<String>> mergedAccounts = new HashMap<>();
        Map<String, List<Integer>> nameConflict = new HashMap<>();
        
        for (int i = 0; i < n; i++) {
            List<String> account = accounts.get(i);
            String name = account.get(0);
            if (nameConflict.containsKey(name)) {
                for (int conflictAccNo : nameConflict.get(name)) {
                    if (parent[conflictAccNo] != conflictAccNo) continue;
                    TreeSet<String> conflictAcc = mergedAccounts.get(conflictAccNo);
                    if (canMerge(conflictAcc, account)) {
                        if (parent[i] == i) {
                            conflictAcc.addAll(account.subList(1, account.size()));
                            parent[i] = conflictAccNo;
                        }
                        else {
                            mergedAccounts.get(parent[i]).addAll(conflictAcc);
                            parent[conflictAccNo] = parent[i];
                        }
                    }
                }
                if (parent[i] == i) {
                    nameConflict.get(name).add(i);
                    mergedAccounts.put(i, new TreeSet<>(account.subList(1, account.size())));
                }
            } else {
                mergedAccounts.put(i, new TreeSet<>(account.subList(1, account.size())));
                nameConflict.computeIfAbsent(name, k -> new ArrayList<>()).add(i);
            }
        }
        List<List<String>> res = new ArrayList<>();
        for (int accountNo : mergedAccounts.keySet()) {
            if (parent[accountNo] != accountNo) continue;
            res.add(new ArrayList<>());
            res.get(res.size()-1).add(accounts.get(accountNo).get(0));
            res.get(res.size()-1).addAll(mergedAccounts.get(accountNo));
        }
        return res;
    }
}

class Solution {
    class DisjointSet{
        int[] parent, size;
        DisjointSet(int n) {
            parent = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int findUParent(int node) {
            if (node == parent[node]) return node;
            return parent[node] = findUParent(parent[node]);
        }
        public void union(int u, int v) {
            int up = findUParent(u);
            int vp = findUParent(v);
            if (size[up] < size[vp]) {
                parent[up] = vp;
                size[vp] += size[up];
            } else {
                parent[vp] = up;
                size[up] += size[vp];
            }
        }
    }
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        int V = accounts.size();
        DisjointSet ds = new DisjointSet(V);

        for (int i = 0; i < V; i++) {
            List<String> account = accounts.get(i);
            Set<String> u = new HashSet<>(account.subList(1, account.size()));
            for (int j = i+1; j < V; j++) {
                List<String> v = accounts.get(j).subList(1, accounts.get(j).size());
                for (String s : v) {
                    if (u.contains(s)) {
                        ds.union(i, j);
                        break;
                    }
                }
            }
        }
        HashMap<Integer, TreeSet<String>> list = new HashMap<>();
        for (int i = 0; i < V; i++) {
            List<String> account = accounts.get(i);
            int parent = ds.findUParent(i);
            list.putIfAbsent(parent, new TreeSet<>());
            list.get(parent).addAll(account.subList(1,account.size()));
        }
        List<List<String>> ans = new ArrayList<>();
        for (Map.Entry<Integer, TreeSet<String>> e : list.entrySet()) {
            List<String> res = new ArrayList<>();
            res.add(accounts.get(e.getKey()).get(0));
            res.addAll(e.getValue());
            ans.add(res);
        }
        return ans;
    }
}

            res.add(new ArrayList<>());

            String mail = e.getKey();
            int id = e.getValue();
            res.get(ds.findUParent(id)).add(mail);
        }
        List<List<String>> ans = new ArrayList<>();
        for (int i = 0; i < V; i++) {
            if(res.get(i).size() == 0) continue;
            String name = accounts.get(i).get(0);
            ans.add(new ArrayList<>());
            ans.get(ans.size()-1).add(name);
            Collections.sort(res.get(i));
            ans.get(ans.size()-1).addAll(res.get(i));
        }
        return ans;
    }

class Solution {
    private int[] parent, size;
    private int findP(int node) {
        if (node == parent[node]) return node;
        return parent[node] = findP(parent[node]);
    }
    private void union(int u, int v) {
        int pu = findP(u);
        int pv = findP(v);
        if (size[pu] < size[pv]) {
            parent[pu] = pv;
            size[pv] += size[pu];
        } else {
            parent[pv] = pu;
            size[pu] += size[pv];
        }
    }
    public List<List<String>> accountsMerge(List<List<String>> accounts) {
        int n = accounts.size();
        parent = new int[n];
        size = new int[n];
        for (int i = 0; i < n; i++) {
            parent[i] = i;
            size[i] = i;
        }

        Map<String, Integer> mails = new HashMap<>();
        for (int i = 0; i < n; i++) {
            List<String> account = accounts.get(i);
            for (int j = 1; j < account.size(); j++) {
                String mail = account.get(j);
                if (mails.containsKey(mail)) union(mails.get(mail), i);
                else mails.put(mail, i);
            }
        }

        List<List<String>> buffer = new ArrayList<>();

        for (int i = 0; i < n; i++) buffer.add(new ArrayList<>());
        for (String mail : mails.keySet()) {
            int accNo = mails.get(mail);
            int parentAccNo = findP(accNo);
            buffer.get(parentAccNo).add(mail);
        }
        List<List<String>> res = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            List<String> account = buffer.get(i);
            if (account.size() == 0) continue;
            String name = accounts.get(i).get(0);
            res.add(new ArrayList<>());
            res.get(res.size()-1).add(name);
            Collections.sort(account);
            res.get(res.size()-1).addAll(account);
        }
        return res;
    }

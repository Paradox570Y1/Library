[1334. Find the City With the Smallest Number of Neighbors at a Threshold Distance](https://leetcode.com/problems/find-the-city-with-the-smallest-number-of-neighbors-at-a-threshold-distance/)

There are `n` cities numbered from `0` to `n-1`. Given the array `edges` where `edges[i] = [fromi, toi, weighti]` represents a bidirectional and weighted edge between cities `fromi` and `toi`, and given the integer `distanceThreshold`.

Return the city with the smallest number of cities that are reachable through some path and whose distance is **at most** `distanceThreshold`, If there are multiple such cities, return the city with the greatest number.

Notice that the distance of a path connecting cities _**i**_ and _**j**_ is equal to the sum of the edges' weights along that path.

**Example 1:**

![](https://assets.leetcode.com/uploads/2024/08/23/problem1334example1.png)

**Input:** n = 4, edges = [[0,1,3],[1,2,1],[1,3,4],[2,3,1]], distanceThreshold = 4
**Output:** 3
**Explanation:** The figure above describes the graph. 
The neighboring cities at a distanceThreshold = 4 for each city are:
City 0 -> [City 1, City 2] 
City 1 -> [City 0, City 2, City 3] 
City 2 -> [City 0, City 1, City 3] 
City 3 -> [City 1, City 2] 
Cities 0 and 3 have 2 neighboring cities at a distanceThreshold = 4, but we have to return city 3 since it has the greatest number.

**Example 2:**

![](https://assets.leetcode.com/uploads/2024/08/23/problem1334example0.png)

**Input:** n = 5, edges = [[0,1,2],[0,4,8],[1,2,3],[1,4,2],[2,3,1],[3,4,1]], distanceThreshold = 2
**Output:** 0
**Explanation:** The figure above describes the graph. 
The neighboring cities at a distanceThreshold = 2 for each city are:
City 0 -> [City 1] 
City 1 -> [City 0, City 4] 
City 2 -> [City 3, City 4] 
City 3 -> [City 2, City 4]
City 4 -> [City 1, City 2, City 3] 
The city 0 has 1 neighboring city at a distanceThreshold = 2.

**Constraints:**

- `2 <= n <= 100`
- `1 <= edges.length <= n * (n - 1) / 2`
- `edges[i].length == 3`
- `0 <= fromi < toi < n`
- `1 <= weighti, distanceThreshold <= 10^4`
- All pairs `(fromi, toi)` are distinct.

# Using Floyd Warshal Algo

```java
class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int[][] adjMat = new int[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (i != j) {
                    adjMat[i][j] = (int) 1e8;
                }
            }
        }
        for (int edge[] : edges) {
            int from = edge[0];
            int to = edge[1];
            int weight = edge[2];
            adjMat[from][to] = weight;
            adjMat[to][from] = weight;
        }
        
        for (int k = 0; k < n; k++) {
            for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                    adjMat[i][j] = Math.min(adjMat[i][j], adjMat[i][k] + adjMat[k][j]);
                }
            }
        }
        int minCount = 101;
        int res = 0;
        int count = 0;
        for (int i = 0; i < n; i++) {
            count = 0;
            for (int j = 0; j < n; j++) {
                if (adjMat[i][j] <= distanceThreshold) {
                    count++;
                }
            }
            if (minCount >= count) {
                res = i;
                minCount = count;
            }
        }

        return res;
    }
}
```

```java
class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        Integer[][] graph = new Integer[n][n];
        for (int i = 0; i < n; i++) {
            graph[i][i] = 0;
        }
        for (int[] edge : edges) {
            int u = edge[0], v = edge[1], w = edge[2];
            graph[u][v] = w;
            graph[v][u] = w;
        }

        for (int via = 0; via < n; via++) {
            for (int u = 0; u < n; u++) {
                for (int v = 0; v < n; v++) {
                    if (u == v) continue;
                    if (graph[u][via] != null && graph[via][v] != null) {
                        if (graph[u][v] == null || graph[u][via] + graph[via][v] < graph[u][v]) {
                            graph[u][v] = graph[u][via] + graph[via][v];
                        }
                    }
                }
            }
        }

        int minNeighbours = n-1;
        int city = -1;
        for (int u = 0; u < n; u++) {
            int count = 0;
            for (Integer dist : graph[u]) {
                if (dist == null || dist == 0) continue;
                if (dist <= distanceThreshold) count++;
            }
            if (count <= minNeighbours) {
                minNeighbours = count;
                city = u;
            }
        }
        return city;
    }
}
```

# Using Dijkstra

TC - > V x (E  x  V log n)

```java
class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        @SuppressWarnings("checked")
        LinkedList<int[]>[] adj = new LinkedList[n];
        for (int V = 0; V < n; V++) {
            adj[V] = new LinkedList<>();
        }
        for (int[] edge : edges) {
            int u = edge[0], v= edge[1], w = edge[2];
            adj[u].add(new int[]{v, w});
            adj[v].add(new int[]{u, w});
        }
        int res = 0;
        int min = (int) 1e9;
        for (int V = 0; V < n; V++) {
            int neighbourCount = bfs(V, adj, distanceThreshold, n);
            if (min >= neighbourCount) {
                res = V;
                min = neighbourCount;
            }
        }
        return res;
    }
    private int bfs(int src, LinkedList<int[]>[] adj, int threshold, int n) {
        PriorityQueue<int[]> q = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        q.offer(new int[] {src, 0});
        int[] dist = new int[n];
        Arrays.fill(dist, (int) 1e9);
        dist[src] = 0;
        int count = 0;
        while (!q.isEmpty()) {
            int[] cur = q.poll();
            int u = cur[0], d = cur[1];
            if (d > dist[u]) continue;
            for (int[] next : adj[u]) {
                int v = next[0], w = next[1];
                if (d + w < dist[v]) {
                    dist[v] = d + w;
                    q.offer(new int[] {v, dist[v]});
                }
            }
        }
        for (int V = 0; V < n; V++) {
            if (dist[V] <= threshold) count++;
        }
        return count;
    }
}
```

# Using bellmanFord

```java
class Solution {
    public int findTheCity(int n, int[][] edges, int distanceThreshold) {
        int res = 0;
        int min = (int) 1e9;
        for (int V = 0; V < n; V++) {
            int neighbours = bellmanFord(V, edges, distanceThreshold, n);
            if (neighbours <= min) {
                min = neighbours;
                res = V;
            }
        }
        return res;
    }
    
    private int bellmanFord(int src, int[][] edges, int k, int n) {
        int[] dist = new int[n];
        Arrays.fill(dist, (int) 1e9);
        dist[src] = 0;

        for (int i = 0; i < n-1; i++) {
            for (int[] edge: edges) {
                int u = edge[0], v = edge[1], w = edge[2];
                if (dist[u] + w < dist[v]) {
                    dist[v] = dist[u] + w;
                }
                if (dist[v] + w < dist[u]) {
                    dist[u] = dist[v] + w;
                }
            }
        }
        int count = -1;
        for (int i = 0; i < n; i++) {
            if (dist[i] <= k) count++;
        }
        return count;
    }
}
```

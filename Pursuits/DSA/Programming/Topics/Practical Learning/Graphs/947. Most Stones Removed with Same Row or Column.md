[947. Most Stones Removed with Same Row or Column](https://leetcode.com/problems/most-stones-removed-with-same-row-or-column/)

On a 2D plane, we place `n` stones at some integer coordinate points. Each coordinate point may have at most one stone.

A stone can be removed if it shares either **the same row or the same column** as another stone that has not been removed.

Given an array `stones` of length `n` where `stones[i] = [xi, yi]` represents the location of the `ith` stone, return _the largest possible number of stones that can be removed_.

**Example 1:**

**Input:** stones = `[[0,0],[0,1],[1,0],[1,2],[2,1],[2,2]]`
**Output:** 5
**Explanation:** One way to remove 5 stones is as follows:
1. Remove stone [2,2] because it shares the same row as [2,1].
2. Remove stone [2,1] because it shares the same column as [0,1].
3. Remove stone [1,2] because it shares the same row as [1,0].
4. Remove stone [1,0] because it shares the same column as [0,0].
5. Remove stone [0,1] because it shares the same row as [0,0].
Stone [0,0] cannot be removed since it does not share a row/column with another stone still on the plane.

**Example 2:**

**Input:** stones = [[0,0],[0,2],[1,1],[2,0],[2,2]]
**Output:** 3
**Explanation:** One way to make 3 moves is as follows:
1. Remove stone [2,2] because it shares the same row as [2,0].
2. Remove stone [2,0] because it shares the same column as [0,0].
3. Remove stone [0,2] because it shares the same row as [0,0].
Stones [0,0] and [1,1] cannot be removed since they do not share a row/column with another stone still on the plane.

**Example 3:**

**Input:** stones = [[0,0]]
**Output:** 0
**Explanation:** [0,0] is the only stone on the plane, so you cannot remove it.

**Constraints:**

- `1 <= stones.length <= 1000`
- `0 <= xi, yi <= 104`
- No two stones are at the same coordinate point.

# Using DFS
Although DFS works barely but it's not the right algorithm for such question where we need to connect two nodes.
```java
class Solution {
    public int removeStones(int[][] stones) {
        int V = stones.length;
        Map<Integer, List<Integer>> xCord = new HashMap<>();
        Map<Integer, List<Integer>> yCord = new HashMap<>();

        for (int node = 0; node < V; node++) {
            int x = stones[node][0];
            int y = stones[node][1];
            if(xCord.containsKey(x)) xCord.get(x).add(node);
            else {
                xCord.put(x, new ArrayList<>());
                xCord.get(x).add(node);
            }
            if(yCord.containsKey(y)) yCord.get(y).add(node);
            else {
                yCord.put(y, new ArrayList<>());
                yCord.get(y).add(node);
            }
        }
        int[][] adjMat = new int[V][V];
        for (int key : xCord.keySet()){
            List<Integer> cur = xCord.get(key);
            if (cur.size() > 1) {
                int u = cur.get(0);
                for (int v = 1; v < cur.size(); v++) {
                    adjMat[u][cur.get(v)] = 1;
                    adjMat[cur.get(v)][u] = 1;
                }
            }
        }
        
        for (int key : yCord.keySet()){
            List<Integer> cur = yCord.get(key);
            if (cur.size() > 1) {
                int u = cur.get(0);
                for (int v = 1; v < cur.size(); v++) {
                    adjMat[u][cur.get(v)] = 1;
                    adjMat[cur.get(v)][u] = 1;
                }
            }
        }
        boolean[] vis = new boolean[V];
        int components = 0;
        for (int i = 0; i < V; i++) {
            if(!vis[i]) {
                dfs(adjMat, i, vis);
                components++;
            }
        }
        return V - components;
    }
    private void dfs(int[][] adjMat, int node, boolean[] vis) {
        vis[node] = true;
        for (int i = 0; i < adjMat.length; i++) {
            if (adjMat[node][i] == 1 && !vis[i]) dfs(adjMat, i, vis);
        }
    }
}
```

# Using Kruskal Algorithm

Best Algorithm to connect to nodes
```java
class Solution {
    class DisjointSet {
        int[] parent, size;
        DisjointSet(int n) {
            parent = new int[n];
            size = new int[n];
            for (int i = 0; i < n; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }

        public int findUParent(int node) {
            if (node == parent[node]) return node;
            return parent[node] = findUParent(parent[node]);
        }

        public void union(int u, int v) {
            int up = findUParent(u);
            int vp = findUParent(v);

            if (size[up] < size[vp]) {
                parent[up] = vp;
                size[vp] += size[up];
            } else {
                parent[vp] =  up;
                size[up] += size[vp];
            }
        }
    }
    public int removeStones(int[][] stones) {
        int V = stones.length;
        DisjointSet ds = new DisjointSet(V);
        Map<Integer, List<Integer>> xCord = new HashMap<>();
        Map<Integer, List<Integer>> yCord = new HashMap<>();

        for (int node = 0; node < V; node++) {
            int x = stones[node][0];
            int y = stones[node][1];
            if(xCord.containsKey(x)) xCord.get(x).add(node);
            else {
                xCord.put(x, new ArrayList<>());
                xCord.get(x).add(node);
            }
            if(yCord.containsKey(y)) yCord.get(y).add(node);
            else {
                yCord.put(y, new ArrayList<>());
                yCord.get(y).add(node);
            }
        }

        for (int key : xCord.keySet()){
            List<Integer> cur = xCord.get(key);
            if (cur.size() > 1) {
                int u = cur.get(0);
                for (int v = 1; v < cur.size(); v++) {
                    if(ds.findUParent(u) != ds.findUParent(cur.get(v))) {
                        ds.union(u, cur.get(v));
                    }
                }
            }
        }
        
        for (int key : yCord.keySet()){
            List<Integer> cur = yCord.get(key);
            if (cur.size() > 1) {
                int u = cur.get(0);
                for (int v = 1; v < cur.size(); v++) {
                    if(ds.findUParent(u) != ds.findUParent(cur.get(v))) {
                        ds.union(u, cur.get(v));
                    }
                }
            }
        }

        int components = 0;
        for (int i = 0; i < V; i++) {
            if(ds.findUParent(i) == i) {
                components++;
            }
        }
        return V - components;
    }
}
```

# Different way to Implement Kruskal Algo 

- We can consider the matrix rows and columns to be nodes in themself.
- These nodes are connected by the rock.
- If there are bunch of rows and cols nodes connected then except one all rocks in that can be lifted.
- So components here doesn't represent how many rocks it can lift but the opposite in one component , one rock cannot be lifted.
- So if number of components greater than size 1 are equal to un-lifted rocks.
- Now subtract un-lifted rocks from total number of rocks to get the maximum lifted rocks.
- This also ensure if there is only one rock forming a component of size 2 then that same rock will not get lifted because 1 component = 1 un-lifted rock, unless size is more than 2 it will not increase the lifted rocks.


Fastest way

```java
            }
        }
    }
    public int removeStones(int[][] stones) {
        int totalStones = stones.length;
        DisjointSet ds = new DisjointSet(20002);
        int maxi = 0;
        int mini = 20001;
        for (int[] stone : stones) {
            int u = stone[0];
            int v = stone[1] + 10001;
            if (ds.findUParent(u) != ds.findUParent(v)){
                ds.union(u, v);
                maxi = Math.max(maxi, v);
                mini = Math.min(mini, u);
            }
        }
        int components = 0;
        for (int i = mini; i <= maxi; i++) {
            if (ds.parent[i] == i && ds.size[i] > 1) {
                components++;
            }
        }
        return totalStones - components;
    }
}
```

```java
            if (ds.findUParent(node) == node) components++;
        }
```

```java
class Solution {
    private int getNode(int[] stone) {
        return stone[0] * 10000 + stone[1]+1;
    }
    Map<Integer, Integer> parent;
    Map<Integer, Integer> size;
    private int find(int node) {
        if (parent.get(node) == -1) return node;
        int p = find(parent.get(node));
        parent.put(node, p);
        return p;
    }
    private void union(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (pu == pv) return;
        if (size.get(pu) > size.get(pv)) {
            parent.put(pv, pu);
            size.merge(pu, 1, Integer::sum);
        } else {
            parent.put(pu, pv);
            size.merge(pv, 1, Integer::sum);
        }
    }
    public int removeStones(int[][] stones) {
        parent = new HashMap<>();
        size = new HashMap<>();
        for (int[] stone : stones) {
            parent.put(getNode(stone), -1);
            size.put(getNode(stone), 1);
        }
        for (int i = 1; i < stones.length; i++) {
            int[] stone1 = stones[i];
            int u = getNode(stone1);
            for(int j = 0; j < i; j++) {
                int[] stone2 = stones[j];
                int v = getNode(stone2);
                if ((stone1[0] == stone2[0] || stone1[1] == stone2[1])) {
                    union(u, v);
                }
            }
        }
        int count = 0;
        for (int p : parent.values()) {
            if (p != -1) count++;
        }
        return count;
    }
}
```

```java
class Solution {
    int[] parent, rank;
    private int find(int node) {
        if (parent[node] == node) return node;
        return parent[node] = find(parent[node]);
    }
    private void union(int u, int v) {
        int pu = find(u);
        int pv = find(v);
        if (pu == pv) return;
        if (rank[pu] > rank[pv]) {
            parent[pv] = pu;
        } else if(rank[pv] > rank[pu]) {
            parent[pu] = pv;
        } else {
            parent[pu] = pv;
            rank[pv]++;
        }
    }
    public int removeStones(int[][] stones) {
        int mx = 0;
        int my = 0;
        for (int[] stone : stones) {
            mx = Math.max(mx, stone[0]);
            my = Math.max(my, stone[1]);
        }
        int V = mx + my + 2;
        parent = new int[V];
        rank = new int[V];
        for (int i = 0; i < V; i++) parent[i] = i;
        Set<Integer> nodes = new HashSet<>();
        for (int[] stone : stones) {
            int u = stone[0], v = stone[1] + mx + 1;
            union(u, v);
            nodes.add(u);
            nodes.add(v);
        }
        int comp = 0;
        for (int node : nodes) {
            if (parent[node] == node) comp++;
        }
        return stones.length - comp;
    }
}
```

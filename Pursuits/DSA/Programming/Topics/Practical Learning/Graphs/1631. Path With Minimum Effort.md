[1631. Path With Minimum Effort](https://leetcode.com/problems/path-with-minimum-effort/)

You are a hiker preparing for an upcoming hike. You are given `heights`, a 2D array of size `rows x columns`, where `heights[row][col]` represents the height of cell `(row, col)`. You are situated in the top-left cell, `(0, 0)`, and you hope to travel to the bottom-right cell, `(rows-1, columns-1)` (i.e., **0-indexed**). You can move **up**, **down**, **left**, or **right**, and you wish to find a route that requires the minimum **effort**.

A route's **effort** is the **maximum absolute difference** in heights between two consecutive cells of the route.

Return _the minimum **effort** required to travel from the top-left cell to the bottom-right cell._

**Example 1:**

![|250](https://assets.leetcode.com/uploads/2020/10/04/ex1.png)

**Input:** heights = `[[1,2,2],[3,8,2],[5,3,5]]`
**Output:** 2
**Explanation:** The route of [1,3,5,3,5] has a maximum absolute difference of 2 in consecutive cells.
This is better than the route of [1,2,2,2,5], where the maximum absolute difference is 3.

**Example 2:**

![|250](https://assets.leetcode.com/uploads/2020/10/04/ex2.png)

**Input:** heights = `[[1,2,3],[3,8,4],[5,3,5]]`
**Output:** 1
**Explanation:** The route of [1,2,3,4,5] has a maximum absolute difference of 1 in consecutive cells, which is better than route [1,3,5,3,5].

**Example 3:**

![|350](https://assets.leetcode.com/uploads/2020/10/04/ex3.png)

**Input:** heights = `[[1,2,1,1,1],[1,2,1,2,1],[1,2,1,2,1],[1,2,1,2,1],[1,1,1,2,1]]`
**Output:** 0
**Explanation:** This route does not require any effort.

**Constraints:**

- `rows == heights.length`
- `columns == heights[i].length`
- `1 <= rows, columns <= 100`
- `1 <= heights[i][j] <= 106`


# Intuition
- It is going to use Dijkstra algo based on effort so i have to add it as a parameter in object itself.
- But we have to maximize the effort so we need extra space to maximize it, parent will store the maximum effort faced till parent and compare it with what child has faced and finally at child's position store the maximized effort if it is less than earlier maximized effort at that position and that's the reason behind pre configuring the array with max integer.
# Using Dijkstra Algo
TC -> O(E log V) or here O ((n x m x 4) log (n x m)) bcz there are n x m vertices and edges in 4 direction

```java
class Solution {
    class Hike implements Comparable<Hike>{
        int x;
        int y;
        int effort;
        Hike(int effort, int x, int y) {
            this.effort = effort;
            this.x = x;
            this.y = y;
        }
        @Override
        public int compareTo(Hike ob) {
            return Integer.compare(this.effort, ob.effort);
        }
    }
    public int minimumEffortPath(int[][] heights) {
        int n = heights.length;
        int m = heights[0].length;
        int minEffort[][] = new int[n][m];
        for (int i = 0; i < n; i++) {
            Arrays.fill(minEffort[i], (int)1e9);
        }
        PriorityQueue<Hike> minHeap = new PriorityQueue<>();

        minHeap.offer(new Hike(0, 0, 0));
        minEffort[0][0] = 0;
        int dirs[][] = {{-1, 0}, {0, -1}, {0, 1}, {1, 0}};
        while (!minHeap.isEmpty()) {
            Hike cur = minHeap.poll();
            for (int dir[] : dirs) {
                int x = dir[0] + cur.x;
                int y = dir[1] + cur.y;
                if (x >= 0 && y >= 0 && x < n && y < m) {
                    int effort = Math.abs(heights[x][y] - heights[cur.x][cur.y]);
                    effort = Math.max(effort, minEffort[cur.x][cur.y]);
                    if (effort < minEffort[x][y]) {
                        minEffort[x][y] = effort;
                        minHeap.offer(new Hike(effort, x, y));
                    }
                }
            }
        }
        return minEffort[n-1][m-1];
    }
}
```

# Using Prims algorithm
```java
class Solution {
    class Pair implements Comparable<Pair>{
        int x, y;
        int w;
        Pair(int x, int y, int w) {
            this.x = x;
            this.y = y;
            this.w = w;
        }

        @Override
        public int compareTo(Pair other) {
            return Integer.compare(this.w, other.w);
        }
    }
    private static int[][] dirs = {{-1, 0}, {0, -1}, {1, 0}, {0, 1}};
    public int minimumEffortPath(int[][] heights) {
        PriorityQueue<Pair> pq = new PriorityQueue<>();
        
        int n = heights.length, m = heights[0].length;
        
        boolean[] vis = new boolean[(n-1) * m + m];
        
        pq.offer(new Pair(0, 0, 0));
        while (!pq.isEmpty()) {
            Pair cur = pq.poll();
            int i = cur.x;
            int j = cur.y;
            int u = i*m + j;
            if (u == vis.length-1) return cur.w;
            if (vis[u]) continue;
            vis[u] = true;
            
            for (int[] dir : dirs) {
                int x = i + dir[0];
                int y = j + dir[1];
                if (x >= 0 && y >= 0 && x < n && y < m) {
                    int v = x*m + y;
                    if (!vis[v]) {
                        int w = Math.abs(heights[i][j] - heights[x][y]);
                        pq.offer(new Pair(x, y, Math.max(w, cur.w)));
                    }
                }
            }
        }
        return -1;
    }
}
```

# Using Kruskal Algorithm

```java
class Solution {
    class Edge implements Comparable<Edge> {
        int u;
        int v;
        int w;
        Edge(int u, int v, int w) {
            this.u = u;
            this.v = v;
            this.w = w;
        }

        @Override
        public int compareTo(Edge other) {
            return Integer.compare(this.w, other.w);
        }
    }
    class DSU {
        int[] parent, size;

        DSU(int V) {
            parent = new int[V];
            size = new int[V];
            for (int i = 0; i < V; i++) {
                parent[i] = i;
                size[i] = 1;
            }
        }
        private int find(int node) {
            if (parent[node] == node) return node;
            return parent[node] = find(parent[node]);
        }
        public void union(int u, int v) {
            int pu = find(u);
            int pv = find(v);
            if (size[pu] > size[pv]) {
                parent[pv] = pu;
                size[pu] += size[pv];
            } else {
                parent[pu] = pv;
                size[pv] += size[pu];
            }
        }
    }
    public int minimumEffortPath(int[][] heights) {
        int n = heights.length, m = heights[0].length;
        int V = (n-1) * m + m;
        DSU dsu = new DSU(V);

        List<Edge> edges = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int u = i * m + j;
                if (i+1 < n) {
                    int v = u + m;
                    int w = Math.abs(heights[i][j] - heights[i+1][j]);
                    edges.add(new Edge(u, v, w));
                }

                if (j+1 < m) {
                    int v = u + 1;
                    int w = Math.abs(heights[i][j] - heights[i][j+1]);
                    edges.add(new Edge(u, v, w));
                }
            }
        }

        Collections.sort(edges);
        int minEffort = 0;
        for (Edge e : edges) {
            if (dsu.find(e.u) != dsu.find(e.v)) {
                dsu.union(e.u, e.v);
                minEffort = Math.max(minEffort, e.w);
                if (dsu.find(0) == dsu.find(V-1)) return minEffort;
            }
        }
        return 0;
    }
}
```

[1976. Number of Ways to Arrive at Destination](https://leetcode.com/problems/number-of-ways-to-arrive-at-destination/)

You are in a city that consists of `n` intersections numbered from `0` to `n - 1` with **bi-directional** roads between some intersections. The inputs are generated such that you can reach any intersection from any other intersection and that there is at most one road between any two intersections.

You are given an integer `n` and a 2D integer array `roads` where `roads[i] = [ui, vi, timei]` means that there is a road between intersections `ui` and `vi` that takes `timei` minutes to travel. You want to know in how many ways you can travel from intersection `0` to intersection `n - 1` in the **shortest amount of time**.

Return _the **number of ways** you can arrive at your destination in the **shortest amount of time**_. Since the answer may be large, return it **modulo** `109 + 7`.

**Example 1:**

![|200](https://assets.leetcode.com/uploads/2025/02/14/1976_corrected.png)

**Input:** n = 7, roads = [[0,6,7],[0,1,2],[1,2,3],[1,3,3],[6,3,3],[3,5,1],[6,5,1],[2,5,1],[0,4,5],[4,6,2]]
**Output:** 4
**Explanation:** The shortest amount of time it takes to go from intersection 0 to intersection 6 is 7 minutes.
The four ways to get there in 7 minutes are:
- 0 ➝ 6
- 0 ➝ 4 ➝ 6
- 0 ➝ 1 ➝ 2 ➝ 5 ➝ 6
- 0 ➝ 1 ➝ 3 ➝ 5 ➝ 6

**Example 2:**

**Input:** n = 2, roads = [[1,0,10]]
**Output:** 1
**Explanation:** There is only one way to go from intersection 0 to intersection 1, and it takes 10 minutes.

**Constraints:**

- `1 <= n <= 200`
- `n - 1 <= roads.length <= n * (n - 1) / 2`
- `roads[i].length == 3`
- `0 <= ui, vi <= n - 1`
- `1 <= timei <= 109`
- `ui != vi`
- There is at most one road connecting any two intersections.
- You can reach any intersection from any other intersection.


# Using Dijkstra + DFS with Pruning

TLE
```java
class Solution {
    long count;
    int dst;
    int minTime;
    public int countPaths(int n, int[][] roads) {
        this.count =  0;
        List<List<int[]>> adjList = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
        }

        for (int road[] : roads) {
            int u = road[0];
            int v = road[1];
            int time = road[2];
            adjList.get(u).add(new int[]{v, time});
            adjList.get(v).add(new int[]{u, time});
        }

        int dist[] = new int[n];
        Arrays.fill(dist, (int)1e9);
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[1] - b[1]);
        pq.offer(new int[]{0, 0});
        dist[0] = 0;

        while (!pq.isEmpty()) {
            int[] node = pq.poll();
            int u = node[0];
            int du = node[1];
            if (u == n-1) {
                dist[u] = du;
                break;
            }
            if (du > dist[u]) continue;

            for (int[] next : adjList.get(u)) {
                int v = next[0];
                int dv = next[1];
                if (du + dv < dist[v]) {
                    dist[v] = du + dv;
                    pq.offer(new int[]{v, du + dv});
                }
            }
        }
        
        this.minTime = dist[n-1];
        this.dst = n-1;
        boolean vis[] = new boolean[n];
        dfs(0, 0, vis, adjList);
        return (int)(count % ((int)1e9 + 7));
    }
    private void dfs(int src, int time, boolean[] vis, List<List<int[]>> adjList) {
        if (src == dst) {
            if(time == minTime) count++;
            return;
        }
        if(time > minTime) return;
        vis[src] = true;
        for (int[] next: adjList.get(src)) {
            int v = next[0];
            int dv = next[1];
            dfs (v, time + dv, vis, adjList);
        }
    }
}
```

# Using Just Dijkstra

```java
class Solution {
    long count;
    int minTime;
    public int countPaths(int n, int[][] roads) {
        this.count =  0;
        List<List<int[]>> adjList = new ArrayList<>();

        for (int i = 0; i < n; i++) {
            adjList.add(new ArrayList<>());
        }

        for (int road[] : roads) {
            int u = road[0];
            int v = road[1];
            int time = road[2];
            adjList.get(u).add(new int[]{v, time});
            adjList.get(v).add(new int[]{u, time});
        }

        long dist[] = new long[n];
        long ways[] = new long[n];
        Arrays.fill(dist, Long.MAX_VALUE);
        PriorityQueue<long[]> pq = new PriorityQueue<>((a, b) -> Long.compare(a[1], b[1]));
        pq.offer(new long[]{0, 0});
        dist[0] = 0;
        ways[0] = 1;
        int MOD = ((int)1e9 + 7);
        while (!pq.isEmpty()) {
            long[] node = pq.poll();
            int u = (int)node[0];
            long du = node[1];
            if (du > dist[u]) continue;
            for (int[] next : adjList.get(u)) {
                int v = next[0];
                int dv = next[1];
                if (du + dv < dist[v]) {
                    dist[v] = du + dv;
                    ways[v] = ways[u];
                    pq.offer(new long[]{v, du + dv});
                }
                else if(du + dv == dist[v]) {
                    ways[v] = (ways[v] + ways[u]) % MOD;
                }
            }
        }
        return (int)(ways[n-1] % MOD);
    }
}
```

```java
class Solution {
    class Pair implements Comparable<Pair> {
        int city;
        long time;
        Pair(int city, long time) {
            this.city = city;
            this.time = time;
        }

        @Override
        public int compareTo(Pair other) {
            return Long.compare(this.time, other.time);
        }
    }
    public int countPaths(int n, int[][] roads) {
        int MOD = (int)1e9 + 7;
        @SuppressWarnings("unchecked")
        List<int[]>[] graph = new ArrayList[n];
        for (int v = 0; v < n; v++) {
            graph[v] = new ArrayList<>();
        }

        for (int road[] : roads) {
            int u = road[0], v = road[1], w = road[2];
            graph[u].add(new int[]{v, w});
            graph[v].add(new int[]{u, w});
        }

        PriorityQueue<Pair> pq = new PriorityQueue<>();
        pq.offer(new Pair(0, 0));
        long[] T = new long[n];
        Arrays.fill(T, Long.MAX_VALUE);
        T[0] = 0;
        long[] ways = new long[n];
        ways[0] = 1;
        while (!pq.isEmpty()) {
            Pair cur = pq.poll();
            int u = cur.city;
            long tu = cur.time;
            if (tu > T[u]) continue;
            for (int[] next : graph[u]) {
                int v = next[0], w = next[1];
                if (tu + w < T[v]) {
                    T[v] = tu + w;
                    ways[v] = ways[u];
                    pq.offer(new Pair(v, T[v]));
                } else if (tu + w == T[v]) {
                    ways[v] = (ways[v] + ways[u]) % MOD;
                }
            }
        }

        return (int)ways[n-1];
    }
}
```

```java
class Solution {
    class City implements Comparable<City> {
        int code;
        long dist;
        City(int code, long dist) {
            this.code = code;
            this.dist = dist;
        }

        @Override
        public int compareTo(City other) {
            return Long.compare(this.dist, other.dist);
        }
    }
    private void buildGraph(List<int[]>[] adj, int[][] roads, int n) {
        for (int i = 0; i < n; i++) {
            adj[i] = new ArrayList<>();
        }

        for (int[] road : roads) {
            int u = road[0], v = road[1], w = road[2];
            adj[u].add(new int[]{v, w});
            adj[v].add(new int[]{u, w});
        }
    }
    public int countPaths(int n, int[][] roads) {
        PriorityQueue<City> pq =new PriorityQueue<>();
        int MOD = (int) 1e9 + 7;
        @SuppressWarnings("unchecked")
        List<int[]>[] adj  = new ArrayList[n];
        buildGraph(adj, roads, n);
        
        pq.offer(new City(0, 0));
        long[] dist = new long[n];
        int[] freq = new int[n];
        Arrays.fill(dist, Long.MAX_VALUE);
        dist[0] = 0;
        freq[0] = 1;
        while (!pq.isEmpty()) {
            City cur = pq.poll();
            if (cur.dist > dist[cur.code]) continue;
            for (int[] next : adj[cur.code]) {
                int code = next[0], w = next[1];
                if (cur.dist + w < dist[code]) {
                    dist[code] = cur.dist + w;
                    freq[code] = freq[cur.code];
                    pq.offer(new City(code, dist[code]));
                } else if(cur.dist + w == dist[code]) {
                    freq[code] = (int)((long)freq[code] + freq[cur.code]) % MOD;
                }
            }
        }
        return freq[n-1];
    }
}
```

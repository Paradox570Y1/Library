[126. Word Ladder II](https://leetcode.com/problems/word-ladder-ii/)

A **transformation sequence** from word `beginWord` to word `endWord` using a dictionary `wordList` is a sequence of words `beginWord -> s1 -> s2 -> ... -> sk` such that:

- Every adjacent pair of words differs by a single letter.
- Every `si` for `1 <= i <= k` is in `wordList`. Note that `beginWord` does not need to be in `wordList`.
- `sk == endWord`

Given two words, `beginWord` and `endWord`, and a dictionary `wordList`, return _all the **shortest transformation sequences** from_ `beginWord` _to_ `endWord`_, or an empty list if no such sequence exists. Each sequence should be returned as a list of the words_ `[beginWord, s1, s2, ..., sk]`.

**Example 1:**

**Input:** beginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log","cog"]
**Output:** `[["hit","hot","dot","dog","cog"],["hit","hot","lot","log","cog"]]`
**Explanation:** There are 2 shortest transformation sequences:
"hit" -> "hot" -> "dot" -> "dog" -> "cog"
"hit" -> "hot" -> "lot" -> "log" -> "cog"

**Example 2:**

**Input:** beginWord = "hit", endWord = "cog", wordList = ["hot","dot","dog","lot","log"]
**Output:** []
**Explanation:** The endWord "cog" is not in wordList, therefore there is no valid transformation sequence.

**Constraints:**

- `1 <= beginWord.length <= 5`
- `endWord.length == beginWord.length`
- `1 <= wordList.length <= 500`
- `wordList[i].length == beginWord.length`
- `beginWord`, `endWord`, and `wordList[i]` consist of lowercase English letters.
- `beginWord != endWord`
- All the words in `wordList` are **unique**.
- The **sum** of all shortest transformation sequences does not exceed `105`.

```java
class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> wordSet = new HashSet<>(wordList);
        List<List<String>> res = new ArrayList<>();

        if (!wordSet.contains(endWord)) return res;

        Queue<List<String>> queue = new LinkedList<>();
        queue.offer(new ArrayList<>(List.of(beginWord)));
        wordSet.remove(beginWord);
        int level = 0;
        while(!queue.isEmpty()) {
            level++;
            int size = queue.size();
            Set<String> tempStorage = new HashSet<>();
            while (size-- > 0) {
                List<String> cur = queue.poll();
                if(cur.get(cur.size() - 1).equals(endWord)) {
                    res.add(new ArrayList<>(cur));
                }
                for (int pos = 0; pos < beginWord.length(); pos++) {
                    StringBuilder word = new StringBuilder(cur.get(cur.size() - 1));
                    for (int ch = 97; ch <= 122; ch++) {
                        word.setCharAt(pos, (char)ch);
                        if (wordSet.contains(word.toString())) {
                            List<String> newSeq = new ArrayList<>(cur);
                            tempStorage.add(word.toString());
                            newSeq.add(word.toString());
                            queue.offer(newSeq);
                        }
                    }
                }
            }
            wordSet.removeAll(tempStorage);
            if (res.size() > 0) return res;
        }
        return res;
    }
}
```

```java
class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> wordSet = new HashSet<>(wordList);
        List<List<String>> res = new ArrayList<>();
        if (!wordSet.contains(endWord)) return res;
        int wordLen = beginWord.length();
        Queue<List<String>> queue = new LinkedList<>();
        queue.offer(new ArrayList<>(List.of(beginWord)));
        wordSet.remove(beginWord);

        while (!queue.isEmpty()) {
            int size = queue.size();
            Set<String> deleteBuffer = new HashSet<>();
            while (size-- > 0) {
                List<String> list = queue.poll();
                String lastWord = list.get(list.size()-1);
                StringBuilder word = new StringBuilder(lastWord);
                for (int i = 0; i < wordLen; i++) {
                    char letter = word.charAt(i);
                    for (char ch = 'a'; ch <= 'z'; ch++) {
                        word.setCharAt(i, ch);
                        String transformedWord = word.toString();
                        List<String> transformedList = new ArrayList<>(list);
                        transformedList.add(transformedWord);
                        if (transformedWord.equals(endWord)) {
                            res.add(transformedList);
                        }
                        else if (wordSet.contains(transformedWord)) {
                            queue.offer(transformedList);
                            deleteBuffer.add(transformedWord);
                        }
                    }
                    word.setCharAt(i, letter);
                }
            }
            if (res.size() > 0) break;
            wordSet.removeAll(deleteBuffer);
        }
        return res;
    }
}
```

```java
class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> vis = new HashSet<>(wordList);
        Map<String, Integer> preComputedPath = new HashMap<>();
        
        Queue<String> queue = new LinkedList<>();
        queue.offer(beginWord);
        preComputedPath.put(beginWord, 0); 
        vis.remove(beginWord);
        int level = 0;
        while(!queue.isEmpty()) {
            level++;
            String cur = queue.peek();
            if(cur.equals(endWord)) break;
            int size = queue.size();
            while (size-- > 0) {
                cur = queue.poll();
                for (int i = 0; i < beginWord.length(); i++) {
                    StringBuilder word = new StringBuilder(cur);
                    for (char ch = 'a'; ch <= 'z'; ch++) {
                        if (cur.charAt(i) == ch) continue;
                        word.setCharAt(i, ch);
                        String str = word.toString();
                        if(vis.contains(str)) {
                            preComputedPath.put(str, level);
                            vis.remove(str);
                            queue.offer(str);
                        }
                    }
                }
            } 
        }
        List<List<String>> res = new ArrayList<>();
        if (preComputedPath.containsKey(endWord)) {
            dfs(beginWord, endWord, preComputedPath, new ArrayList<>(List.of(endWord)), res);
        }
        return res;
    }

    private void dfs(String beginWord, String cur, Map<String, Integer> preComputedPath, List<String> list, List<List<String>> res) {
        if (beginWord.equals(cur)) {
            List<String> ans = new ArrayList<>(list);
            reverse(ans);
            res.add(ans);
            return;
        }
        int level = preComputedPath.get(cur);
        for (int i = 0; i < beginWord.length(); i++) {
            StringBuilder word = new StringBuilder(cur);
            for (char ch = 'a'; ch <= 'z'; ch++) {
                word.setCharAt(i, ch);
                String str = word.toString();
                if (preComputedPath.containsKey(str) && preComputedPath.get(str) + 1 == level) {
                    list.add(str);
                    dfs(beginWord, str, preComputedPath, list, res);
                    list.remove(list.size()-1);
                }
            }
        }
    }
    private void reverse(List<String> list) {
        int i = 0, j = list.size()-1;
        while (i < j) {
            String temp = list.get(i);
            list.set(i, list.get(j));
            list.set(j, temp);
            i++;
            j--;
        }
    }
}
```

```java
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        Set<String> vis = new HashSet<>(wordList);
        Map<String, Integer> preComputedPath = new HashMap<>();
        
        Queue<String> queue = new LinkedList<>();
        queue.offer(beginWord);
        preComputedPath.put(beginWord, 1); 
        vis.remove(beginWord);
        
        while(!queue.isEmpty()) {
            String cur = queue.poll();
            if(cur.equals(endWord)) break;

            int steps = preComputedPath.get(cur);
            for (int i = 0; i < beginWord.length(); i++) {
                StringBuilder word = new StringBuilder(cur);
                for (char ch = 'a'; ch <= 'z'; ch++) {
                    if (cur.charAt(i) == ch) continue;

                    String str = word.toString();
                    if(vis.contains(str)) {
                        preComputedPath.put(str, steps+1);
                        vis.remove(str);
                        queue.offer(str);
                    }
                }
            }
        }
        List<List<String>> res = new ArrayList<>();
        if (preComputedPath.containsKey(endWord)) {
            dfs(beginWord, endWord, preComputedPath, new ArrayList<>(List.of(endWord)), res);
        }
        return res;
    }

            res.add(ans);
            return;

    }
```

```java
class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        List<List<String>> res = new ArrayList<>();
        Set<String> wordSet = new HashSet<>(wordList);
        if (!wordSet.contains(endWord)) return res;
        Map<String, List<String>> adjMap = new HashMap<>();

        Queue<String> queue = new LinkedList<>();
        queue.offer(beginWord);
        wordSet.remove(beginWord);
        boolean endFound = false;
        while (!queue.isEmpty() && !endFound) {
            int size = queue.size();
            Set<String> deleteBuffer = new HashSet<>();
            while (size-- > 0) {
                String curWord = queue.poll();
                StringBuilder word = new StringBuilder(curWord);
                for (int i = 0; i < word.length(); i++) {
                    char letter = word.charAt(i);
                    for (char ch = 'a'; ch <= 'z'; ch++) {
                        word.setCharAt(i, ch);
                        String transformedWord = word.toString();
                        if (wordSet.contains(transformedWord)) {
                            if (!adjMap.containsKey(transformedWord)) {
                                adjMap.put(transformedWord, new ArrayList<>(List.of(curWord)));
                                queue.offer(transformedWord);
                                deleteBuffer.add(transformedWord);
                            }
                            else adjMap.get(transformedWord).add(curWord);
                        }
                        if (transformedWord.equals(endWord)) endFound = true;
                    }
                    word.setCharAt(i, letter);
                }
            }
            wordSet.removeAll(deleteBuffer);
        }
        dfs(adjMap, beginWord, endWord, res, new ArrayList<>(List.of(endWord)));
        return res;
    }
    private void dfs(Map<String, List<String>> adjMap, String beginWord, String word, List<List<String>> res, List<String> list) {
        if (adjMap.get(word) == null) {
            if(word.equals(beginWord)) res.add(rev(list));
            return;
        }
        for (String s : adjMap.get(word)) {
            list.add(s);
            dfs(adjMap, beginWord, s, res, list);
            list.remove(list.size()-1);
        }
    }
    private List<String> rev(List<String> list) {
        List<String> res = new ArrayList<>();
        for (int i = list.size()-1; i>= 0; i--) {
            res.add(list.get(i));
        }
        return res;
    }
}
```

```java
class Solution {
    public List<List<String>> findLadders(String beginWord, String endWord, List<String> wordList) {
        List<List<String>> res = new ArrayList<>();
        Set<String> wordSet = new HashSet<>(wordList);
        if (!wordSet.contains(endWord)) return res;
        Map<String, List<String>> adjMap = new HashMap<>();

        Queue<String> queue = new LinkedList<>();
        queue.offer(beginWord);
        wordSet.remove(beginWord);
        boolean endFound = false;
        while (!queue.isEmpty() && !endFound) {
            int size = queue.size();
            Set<String> deleteBuffer = new HashSet<>();
            while (size-- > 0) {
                String curWord = queue.poll();
                char[] word = curWord.toCharArray();
                for (int i = 0; i < word.length; i++) {
                    char letter = word[i];
                    for (char ch = 'a'; ch <= 'z'; ch++) {
                        if (ch == letter) continue;
                        word[i] = ch;
                        String transformedWord = new String(word);
                        if (wordSet.contains(transformedWord)) {
                            if (!adjMap.containsKey(transformedWord)) {
                                adjMap.put(transformedWord, new ArrayList<>(List.of(curWord)));
                                queue.offer(transformedWord);
                                deleteBuffer.add(transformedWord);
                            }
                            else adjMap.get(transformedWord).add(curWord);
                        }
                        if (transformedWord.equals(endWord)) endFound = true;
                    }
                    word[i] = letter;
                }
            }
            wordSet.removeAll(deleteBuffer);
        }
        dfs(adjMap, beginWord, endWord, res, new ArrayList<>(List.of(endWord)));
        return res;
    }
    private void dfs(Map<String, List<String>> adjMap, String beginWord, String word, List<List<String>> res, List<String> list) {
        if (adjMap.get(word) == null) {
            if(word.equals(beginWord)) {
                List<String> cpy = new ArrayList<>(list);
                Collections.reverse(cpy);
                res.add(cpy);
            }
            return;
        }
        for (String s : adjMap.get(word)) {
            list.add(s);
            dfs(adjMap, beginWord, s, res, list);
            list.remove(list.size()-1);
        }
    }
}
```

[105. Construct Binary Tree from Preorder and Inorder Traversal](https://leetcode.com/problems/construct-binary-tree-from-preorder-and-inorder-traversal/)

Given two integer arrays `preorder` and `inorder` where `preorder` is the preorder traversal of a binary tree and `inorder` is the inorder traversal of the same tree, construct and return _the binary tree_.

**Example 1:**

![](https://assets.leetcode.com/uploads/2021/02/19/tree.jpg)

**Input:** preorder = [3,9,20,15,7], inorder = [9,3,15,20,7]
**Output:** [3,9,20,null,null,15,7]

**Example 2:**

**Input:** preorder = [-1], inorder = [-1]
**Output:** [-1]

**Constraints:**

- `1 <= preorder.length <= 3000`
- `inorder.length == preorder.length`
- `-3000 <= preorder[i], inorder[i] <= 3000`
- `preorder` and `inorder` consist of **unique** values.
- Each value of `inorder` also appears in `preorder`.
- `preorder` is **guaranteed** to be the preorder traversal of the tree.
- `inorder` is **guaranteed** to be the inorder traversal of the tree.


# Brute

```java
class Solution {
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        TreeNode root = new TreeNode(preorder[0]);
        TreeNode trav = root;
        int i,j,k;
        int n= inorder.length;
        for(k=0;k<n;k++){
            if(inorder[k]==preorder[0])break;
        }
        if(k>0)helper(preorder,inorder,0,k-1,1,trav,true);
        if(k<n-1)helper(preorder,inorder,k+1,n-1,1,trav,false);
        return root;
    }
    private void helper(int[] preorder,int[] inorder,int i,int j,int k,TreeNode node,boolean left){
        if(k>=preorder.length || i>j)return;
        int t=i-1;
        for(t=i;t<=j;t++){
            if(inorder[t]  == preorder[k])break;
        }
        if(t==i-1 || t==j+1)helper(preorder,inorder,i,j,k+1,node,left);
        else{
            if(left){
                node.left = new TreeNode(preorder[k]);
                helper(preorder,inorder,i,t-1,k+1,node.left,true);
                helper(preorder,inorder,t+1,j,k+1,node.left,false);
            }
            else{
                node.right = new TreeNode(preorder[k]);
                helper(preorder,inorder,i,t-1,k+1,node.right,true);
                helper(preorder,inorder,t+1,j,k+1,node.right,false);
            }
        }
    }
}
```

```java
class Solution {
    HashSet<Integer> hs =new HashSet<>();
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        TreeNode root = new TreeNode(-1);
        TreeNode trav = root;
        int n= inorder.length;
        helper(preorder,inorder,0,n-1,0,trav,true);
        return root.left;
    }
    private void helper(int[] preorder,int[] inorder,int i,int j,int k,TreeNode node,boolean left){
        if(k>=preorder.length || i>j)return;
        if(hs.contains(preorder[k])){
            helper(preorder,inorder,i,j,k+1,node,left);
            return;
        }
        int t=i-1;
        for(t=i;t<=j;t++){
            if(inorder[t]  == preorder[k])break;
        }
        if(t==i-1 || t==j+1)helper(preorder,inorder,i,j,k+1,node,left);
        else{
            hs.add(preorder[k]);
            if(left){
                node.left = new TreeNode(preorder[k]);
                helper(preorder,inorder,i,t-1,k+1,node.left,true);
                helper(preorder,inorder,t+1,j,k+1,node.left,false);
            }
            else{
                node.right = new TreeNode(preorder[k]);
                helper(preorder,inorder,i,t-1,k+1,node.right,true);
                helper(preorder,inorder,t+1,j,k+1,node.right,false);
            }
        }
    }
}
```

# Optimal

```java
class Solution {
    private HashMap<Integer,Integer> map = new HashMap<>();
    private TreeNode helper(int[] inorder,int inStart,int inEnd,int[] preorder,int preStart,int preEnd){
        if(inStart>inEnd || preStart>preEnd)return null;
        TreeNode root = new TreeNode(preorder[preStart]);
        int pivotIdx = map.get(root.val);
        int leftCnt = pivotIdx-inStart;

        root.left = helper(inorder,inStart,pivotIdx-1,preorder,preStart+1,preStart+leftCnt);
        root.right = helper(inorder,pivotIdx+1,inEnd,preorder,preStart+leftCnt+1,preEnd);
        return root;
    }
    public TreeNode buildTree(int[] preorder, int[] inorder) {
        int n = inorder.length;
        for(int i=0;i<n;i++){
            map.put(inorder[i],i);
        }
        TreeNode root = helper(inorder,0,n-1,preorder,0,n-1);
        return root;
    }
}
```

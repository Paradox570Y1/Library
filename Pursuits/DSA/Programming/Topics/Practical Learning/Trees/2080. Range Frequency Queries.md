[2080. Range Frequency Queries](https://leetcode.com/problems/range-frequency-queries/)

Design a data structure to find the **frequency** of a given value in a given subarray.

The **frequency** of a value in a subarray is the number of occurrences of that value in the subarray.

Implement the `RangeFreqQuery` class:

- `RangeFreqQuery(int[] arr)` Constructs an instance of the class with the given **0-indexed** integer array `arr`.
- `int query(int left, int right, int value)` Returns the **frequency** of `value` in the subarray `arr[left...right]`.

A **subarray** is a contiguous sequence of elements within an array. `arr[left...right]` denotes the subarray that contains the elements of `nums` between indices `left` and `right` (**inclusive**).

**Example 1:**

**Input**
["RangeFreqQuery", "query", "query"]
[[[12, 33, 4, 56, 22, 2, 34, 33, 22, 12, 34, 56]], [1, 2, 4], [0, 11, 33]]
**Output**
[null, 1, 2]

**Explanation**
RangeFreqQuery rangeFreqQuery = new RangeFreqQuery([12, 33, 4, 56, 22, 2, 34, 33, 22, 12, 34, 56]);
rangeFreqQuery.query(1, 2, 4); // return 1. The value 4 occurs 1 time in the subarray [33, 4]
rangeFreqQuery.query(0, 11, 33); // return 2. The value 33 occurs 2 times in the whole array.

**Constraints:**

- `1 <= arr.length <= 105`
- `1 <= arr[i], value <= 104`
- `0 <= left <= right < arr.length`
- At most `105` calls will be made to `query`



# Brute using prefix Map

Gives MLE although better time complexity
```java
class RangeFreqQuery {
    private Map<Integer, Integer>[] prefixMap;
    public RangeFreqQuery(int[] arr) {
        int n = arr.length;
        prefixMap = new HashMap[n];
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < n; i++) {
            map.merge(arr[i], 1, Integer::sum);
            prefixMap[i] = new HashMap<>(map);
        }
    }
    
    public int query(int left, int right, int value) {
        if (left == 0) return prefixMap[right].get(value);
        return prefixMap[right].get(value) - prefixMap[left-1].get(value);
    }
}
```

# Using Segment Trees

- To reduce the memory footprint use Segment Trees
```java
class RangeFreqQuery {
    HashMap<Integer, Integer>[] segTree;
    int n;
    private void cpyMap(HashMap<Integer, Integer> des, HashMap<Integer, Integer> src) {
        for (Map.Entry<Integer, Integer> e : src.entrySet()) {
            des.merge(e.getKey(), e.getValue(), Integer::sum);
        }
    }
    private void build(int[] arr, int low , int high, int idx) {
        if (low == high) {
            segTree[idx] = new HashMap<>();
            segTree[idx].put(arr[low], 1);
            return;
        }
        int mid = (low + high)/2;
        build(arr, low, mid, idx*2+1);
        build(arr, mid+1, high, idx*2+2);
        segTree[idx] = new HashMap<>();
        cpyMap(segTree[idx], segTree[idx*2 + 1]);
        cpyMap(segTree[idx], segTree[idx*2 + 2]);
    }
    public RangeFreqQuery(int[] arr) {
        n = arr.length;
        segTree = new HashMap[4*n];
        build(arr, 0, n-1, 0);
    }
    private int query(int left, int right, int value, int low, int high, int idx) {
        if (right < low || left > high) return 0;
        if (left <= low && high <= right) {
            if (segTree[idx].containsKey(value)) {
                return segTree[idx].get(value);
            }
            return 0;
        }
        int mid = (low + high)/2;
        int l = query(left, right, value, low, mid, idx*2+1);
        int r = query(left, right, value, mid+1, high, idx*2+2);
        return l+r;
    }
    public int query(int left, int right, int value) {
        return query(left, right, value, 0, n-1, 0);
    }
}
```

```java
class RangeFreqQuery {
    private Map<Integer, Integer>[] segTree;
    private int n;

    private void cpyMap(Map<Integer, Integer> des, Map<Integer, Integer> src) {
        for (int key : src.keySet()) {
            des.merge(key, src.get(key), Integer::sum);
        }
    }
    private void buildTree(int[] arr) {
        for (int i = 0; i < n; i++) {
            segTree[i+n] = new HashMap<>();
            segTree[i+n].put(arr[i], 1);
        }

        for (int i = n-1; i > 0; i--) {
            segTree[i] = new HashMap<>();
            cpyMap(segTree[i], segTree[i << 1]);
            cpyMap(segTree[i], segTree[i << 1 | 1]);
        }
    }

    public RangeFreqQuery(int[] arr) {
        this.n = arr.length;
        segTree = new HashMap[2*n];
        buildTree(arr);
    }
    
    public int query(int left, int right, int value) {
        int res = 0;
        int l = left + n, r = right + n + 1;
        for (; l < r; l >>= 1, r >>= 1) {
            if ((l & 1) > 0) {
                res += segTree[l++].getOrDefault(value, 0);
            }
            if ((r & 1) > 0) {
                res += segTree[--r].getOrDefault(value, 0);
            }
        }
        return res;
    }
}

 */
```

# Using Map + Binary Search

```java
class RangeFreqQuery {
    Map<Integer, List<Integer>> map;
    public RangeFreqQuery(int[] arr) {
        map = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            map.computeIfAbsent(arr[i], k -> new ArrayList<>()).add(i);
        }
    }
    
    public int query(int left, int right, int value) {
        if (!map.containsKey(value)) return 0;
        List<Integer> list = map.get(value);
        return lowerBound(list, right+1) - lowerBound(list, left);
    }

    private int lowerBound(List<Integer> list , int target) {
        int low = 0, high = list.size();
        while (low < high) {
            int mid = low + (high - low)/2;
            if (list.get(mid) < target) low = mid+1;
            else high = mid;
        }
        return low;
    }
}
```

```java
class RangeFreqQuery {
    Map<Integer, List<Integer>> map;
    public RangeFreqQuery(int[] arr) {
        map = new HashMap<>();
        for (int i = 0; i < arr.length; i++) {
            map.computeIfAbsent(arr[i], k -> new ArrayList<>()).add(i);
        }
    }
    
    public int query(int left, int right, int value) {
        if (!map.containsKey(value)) return 0;
        List<Integer> list = map.get(value);
        return lowerBound(list, right+1) - lowerBound(list, left);
    }

    private int lowerBound(List<Integer> list , int target) {
        int low = 0, high = list.size()-1;
        int ans = high+1;
        while (low <= high) {
            int mid = low + (high - low)/2;
            if (list.get(mid) < target) low = mid+1;
            else {
                ans = mid;
                high = mid-1;
            }
        }
        return ans;
    }
}
```

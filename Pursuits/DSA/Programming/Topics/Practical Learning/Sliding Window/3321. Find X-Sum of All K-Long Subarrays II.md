[3321. Find X-Sum of All K-Long Subarrays II](https://leetcode.com/problems/find-x-sum-of-all-k-long-subarrays-ii/)

You are given an array `nums` of `n` integers and two integers `k` and `x`.

The **x-sum** of an array is calculated by the following procedure:

- Count the occurrences of all elements in the array.
- Keep only the occurrences of the top `x` most frequent elements. If two elements have the same number of occurrences, the element with the **bigger** value is considered more frequent.
- Calculate the sum of the resulting array.

**Note** that if an array has less than `x` distinct elements, its **x-sum** is the sum of the array.

Return an integer array `answer` of length `n - k + 1` where `answer[i]` is the **x-sum** of the subarray `nums[i..i + k - 1]`.

**Example 1:**

**Input:** nums = [1,1,2,2,3,4,2,3], k = 6, x = 2

**Output:** [6,10,12]

**Explanation:**

- For subarray `[1, 1, 2, 2, 3, 4]`, only elements 1 and 2 will be kept in the resulting array. Hence, `answer[0] = 1 + 1 + 2 + 2`.
- For subarray `[1, 2, 2, 3, 4, 2]`, only elements 2 and 4 will be kept in the resulting array. Hence, `answer[1] = 2 + 2 + 2 + 4`. Note that 4 is kept in the array since it is bigger than 3 and 1 which occur the same number of times.
- For subarray `[2, 2, 3, 4, 2, 3]`, only elements 2 and 3 are kept in the resulting array. Hence, `answer[2] = 2 + 2 + 2 + 3 + 3`.

**Example 2:**

**Input:** nums = [3,8,7,8,7,5], k = 2, x = 2

**Output:** [11,15,15,15,12]

**Explanation:**

Since `k == x`, `answer[i]` is equal to the sum of the subarray `nums[i..i + k - 1]`.

**Constraints:**

- `nums.length == n`
- `1 <= n <= 105`
- `1 <= nums[i] <= 109`
- `1 <= x <= k <= nums.length`

# Using PriorityQueue (Brute)

```java
class Solution {
    public long[] findXSum(int[] nums, int k, int x) {
        PriorityQueue<long[]> maxHeap = new PriorityQueue<long[]>((a, b) -> {
            if (a[1] != b[1]) return Long.compare(b[1], a[1]);
            return Long.compare(b[0], a[0]);
        });
        Map<Integer, long[]> hm = new HashMap<>();
        
        long res[] = new long[nums.length - k + 1];
        int p = 0;
        int j = 0;
        for (int i = 0; i < nums.length; i++) {
            int num = nums[i];
            if (hm.containsKey(num)) {
                maxHeap.remove(hm.get(num));
                hm.get(num)[1]++;
                maxHeap.offer(hm.get(num));
            } else {
                long[] ele = {num, 1};
                hm.put(num, ele);
                maxHeap.offer(ele);
            }
            if (i >= k-1) {
                int y = x;
                Set<long[]> hs = new HashSet<>();
                while (!maxHeap.isEmpty() && y-- > 0) {
                    long[] ele = maxHeap.poll();
                    res[p] += ele[0] * ele[1];
                    hs.add(ele);
                }
                for (long[] ele : hs) {
                    maxHeap.offer(ele);
                }
                p++;
                maxHeap.remove(hm.get(nums[j]));
                hm.get(nums[j])[1]--;
                if (hm.get(nums[j])[1] == 0) {
                    hm.remove(nums[j]);
                } else {
                    maxHeap.offer(hm.get(nums[j]));
                }
                j++;
            }
        }
        return res;
    }
}
```

```java
class Solution {
```

```java
        int val;
        long freq;
        X(int val, long freq) {
            this.val = val;
            this.freq = freq;
        }
        public long score() {
            return val * freq;
        }
        
    }
    public void updateData(PriorityQueue<X> heap, Map<Integer, X> cache, X data, int updateBy) {
        heap.remove(data);
        data.freq += updateBy;
        if (data.freq == 0) {
            cache.remove(data.val);
        } else {
            heap.offer(data);
            cache.put(data.val, data);
        }
    }

                return Integer.compare(v2.val, v1.val);

        });

            public int compare(X v1, X v2) {

                return Integer.compare(v1.val, v2.val);
            }

        maxCache = new HashMap<>();

        int p = 0;
        xsum = 0;
        int start = 0;
        for (int i = 0; i < nums.length; i++) {
            int key = nums[i];
            if (minCache.containsKey(key) || maxCache.containsKey(key)) {
                X data;
                if(minCache.containsKey(key)) {
                    data = minCache.get(key);
                    updateData(minHeap, minCache, data, 1);
                    xsum += data.val;
                } else {
                    data = maxCache.get(key);
                    updateData(maxHeap, maxCache, data, 1);
                    distributeData();
                }
            } else {
                X data = new X(key, 1);
                if(minHeap.size() < x) {
                    minHeap.offer(data);
                    minCache.put(key, data);
                    xsum += data.val;
                } else {
                    maxHeap.offer(data);
                    maxCache.put(key, data);
                    distributeData();
                }
            }
            if(i >= k-1) {
                res[p++] = xsum;
                updateWindow(nums[start]);
                start++;
            }
        }
        return res;
    }
    public void distributeData() {
        X data1 = minHeap.peek();
        X data2 = maxHeap.peek();
        boolean needUpdate = false;
        if (data1.freq != data2.freq) needUpdate = (data2.freq > data1.freq);
        else needUpdate = (data2.val > data1.val);
        if (needUpdate) {
            minHeap.offer(maxHeap.poll());
            maxCache.remove(data2.val);
            minCache.put(data2.val, data2);
            
            maxHeap.offer(minHeap.poll());
            minCache.remove(data1.val);
            maxCache.put(data1.val, data1);
            xsum += data2.score() - data1.score();
        }
    }
    public void updateWindow(int key) {
        if(minCache.containsKey(key)) {
            X data = minCache.get(key);
            updateData(minHeap, minCache, data, -1);
            xsum -= key;
            if(!maxHeap.isEmpty()) distributeData();
        } else {
            X data = maxCache.get(key);
            updateData(maxHeap, maxCache, data, -1);
        }
    }
}
```

```java
class Solution {
    static class Pair implements Comparable<Pair> {
        int freq, val;
        Pair(int freq, int val) {
            this.freq = freq;
            this.val  = val;
        }
        public int compareTo(Pair o) {
            if (freq != o.freq) return Integer.compare(freq, o.freq);
            return Integer.compare(val, o.val);
        }
        @Override
        public boolean equals(Object o) {
            if (this == o) return true;
            if (!(o instanceof Pair)) return false;
            Pair p = (Pair)o;
            return freq == p.freq && val == p.val;
        }
        @Override
        public int hashCode() {
            return freq * 1000003 + val;
        }
    }

    TreeSet<Pair> s1, s2;
    HashMap<Integer, Integer> freq;
    long sum = 0;
    int x;

    void remove_from_set(int idx, int[] nums) {
        int f = freq.getOrDefault(nums[idx], 0);
        Pair p = new Pair(f, nums[idx]);
        long val = (long) f * nums[idx];
        if (s1.contains(p)) {
            sum -= val;
            s1.remove(p);
        }
        if (s2.contains(p)) {
            s2.remove(p);
        }
        if (s1.size() < x && s2.size() > 0) {
            Pair pp = s2.last();
            s1.add(pp);
            sum += (long) pp.freq * pp.val;
            s2.remove(pp);
        }
    }

    void add_to_set(int idx, int[] nums) {
        int f = freq.getOrDefault(nums[idx], 0);
        Pair p = new Pair(f, nums[idx]);
        long val = (long) f * nums[idx];
        s1.add(p);
        sum += val;
        if (s1.size() > x) {
            Pair pp = s1.first();
            s1.remove(pp);
            sum -= (long) pp.freq * pp.val;
            s2.add(pp);
        }
    }

    void add(int idx, int[] nums) {
        remove_from_set(idx, nums);
        freq.put(nums[idx], freq.getOrDefault(nums[idx], 0) + 1);
        add_to_set(idx, nums);
    }

    void remove(int idx, int[] nums) {
        remove_from_set(idx, nums);
        freq.put(nums[idx], freq.getOrDefault(nums[idx], 0) - 1);
        add_to_set(idx, nums);
    }

    public long[] findXSum(int[] nums, int k, int x) {
        int n = nums.length;

        s1 = new TreeSet<>();
        s2 = new TreeSet<>();
        freq = new HashMap<>();

        for (int i = 0; i < n; i++) {
            add(i, nums);
            if (i - k >= 0) remove(i - k, nums);
            if (i - k + 1 >= 0) ans.add(sum);
        }
        long[] result = new long[ans.size()];
        for (int i = 0; i < ans.size(); i++) result[i] = ans.get(i);
        return result;
    }
}
```

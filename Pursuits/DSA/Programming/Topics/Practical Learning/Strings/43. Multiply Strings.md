[43. Multiply Strings](https://leetcode.com/problems/multiply-strings/)

Given two non-negative integers `num1` and `num2` represented as strings, return the product of `num1` and `num2`, also represented as a string.

**Note:** You must not use any built-in BigInteger library or convert the inputs to integer directly.

**Example 1:**

**Input:** num1 = "2", num2 = "3"
**Output:** "6"

**Example 2:**

**Input:** num1 = "123", num2 = "456"
**Output:** "56088"

**Constraints:**

- `1 <= num1.length, num2.length <= 200`
- `num1` and `num2` consist of digits only.
- Both `num1` and `num2` do not contain any leading zero, except the number `0` itself.

class Solution {
    private int carry=0;
    private String mul(char a,char b){
        int c = a-'0';
        int d = b-'0';
        int cal = c*d + carry;
        int res = cal%10;
        carry = cal/10;
        return Integer.toString(res);
    }
    private String add(char a,char b){
        int c = a-'0';
        int d = b-'0';
        int res = c+d + carry;
        carry = res/10;
        res%=10;
        return Integer.toString(res);
    }
    private String addString(String a,String b){
        StringBuilder sb = new StringBuilder();
        int a1 = a.length(),b1 = b.length();
        int len = Math.min(a1,b1);
        int i=0;
        for(;i<len;i++){
            sb.append(add(a.charAt(i),b.charAt(i)));
        }
        if(a1>b1){
            while(i<a1){
                sb.append(add(a.charAt(i),'0'));
                i++;
            }
        }
        else if(b1>a1){
            while(i<b1){
                sb.append(add('0',b.charAt(i)));
                i++;
            }
        }
        if(carry>0)sb.append(carry);
        return sb.toString();
    }
    public String multiply(String num1, String num2) {
        if(num1.charAt(0) == '0' || num2.charAt(0) == '0')return "0";
        StringBuilder ans = new StringBuilder();
        StringBuilder sb = new StringBuilder();
        String inc = "";
        for(int i=num1.length()-1;i>=0;i--){
            char n1 = num1.charAt(i);
            sb.append(inc);
            for(int j=num2.length()-1;j>=0;j--){
                char n2 = num2.charAt(j);
                sb.append(mul(n1,n2));
            }
            if(carry>0)sb.append(Integer.toString(carry));
            carry=0;
            String buffer = addString(sb.toString(),ans.toString());
            ans.setLength(0);
            sb.setLength(0);
            carry=0;
            ans.append(buffer);
            inc+="0";
        }
        return ans.reverse().toString();
    }
}

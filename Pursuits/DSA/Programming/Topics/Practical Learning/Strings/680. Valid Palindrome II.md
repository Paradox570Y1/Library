[680. Valid Palindrome II](https://leetcode.com/problems/valid-palindrome-ii/)


Given a string `s`, return `true` _if the_ `s` _can be palindrome after deleting **at most one** character from it_.

**Example 1:**

**Input:** s = "aba"
**Output:** true

**Example 2:**

**Input:** s = "abca"
**Output:** true
**Explanation:** You could delete the character 'c'.

**Example 3:**

**Input:** s = "abc"
**Output:** false

**Constraints:**

- `1 <= s.length <= 105`
- `s` consists of lowercase English letters.


# Using recursion

```java
class Solution {
    private boolean checkPal(String s,int i,int j,boolean canSkip){
        if(i>=j)return true;
        if(s.charAt(i) != s.charAt(j)){
            if(!canSkip)return false;
            return checkPal(s,i+1,j,false) || checkPal(s,i,j-1,false);
        }
        return checkPal(s,i+1,j-1,canSkip);
    }
    public boolean validPalindrome(String s) {
        return checkPal(s,0,s.length()-1,true);
    }
}
```

# Using two pointers

```java
class Solution {
    private boolean checkPal(String s,int i,int j){
        while(i<j){
            if(s.charAt(i) != s.charAt(j))return false;
            i++;j--;
        }
        return true;
    }
    public boolean validPalindrome(String s) {
        int n = s.length();
        int i = 0;
        int j = n-1;
        int deletionLeft = 1;
        while(i<j){
            char left = s.charAt(i),right = s.charAt(j);
            if(left != right){
                if(deletionLeft==1){
                    char afterLeft = s.charAt(i+1);
                    char beforeRight = s.charAt(j-1);
                    if(afterLeft == right && beforeRight == left){
                        if(checkPal(s,i+1,j) || checkPal(s,i,j-1))return true;
                        else return false;
                    }
                    else if(afterLeft == right){
                        i++;
                        deletionLeft--;
                    }
                    else if(beforeRight == left){
                        j--;
                        deletionLeft--;
                    }
                    else return false;
                }
                else return false;
            }
            i++;j--;
        }
        return true;
    }
}
```

[966. Vowel Spellchecker](https://leetcode.com/problems/vowel-spellchecker/)

Given a `wordlist`, we want to implement a spellchecker that converts a query word into a correct word.

For a given `query` word, the spell checker handles two categories of spelling mistakes:

- Capitalization: If the query matches a word in the wordlist (**case-insensitive**), then the query word is returned with the same case as the case in the wordlist.
    - Example: `wordlist = ["yellow"]`, `query = "YellOw"`: `correct = "yellow"`
    - Example: `wordlist = ["Yellow"]`, `query = "yellow"`: `correct = "Yellow"`
    - Example: `wordlist = ["yellow"]`, `query = "yellow"`: `correct = "yellow"`
- Vowel Errors: If after replacing the vowels `('a', 'e', 'i', 'o', 'u')` of the query word with any vowel individually, it matches a word in the wordlist (**case-insensitive**), then the query word is returned with the same case as the match in the wordlist.
    - Example: `wordlist = ["YellOw"]`, `query = "yollow"`: `correct = "YellOw"`
    - Example: `wordlist = ["YellOw"]`, `query = "yeellow"`: `correct = ""` (no match)
    - Example: `wordlist = ["YellOw"]`, `query = "yllw"`: `correct = ""` (no match)

In addition, the spell checker operates under the following precedence rules:

- When the query exactly matches a word in the wordlist (**case-sensitive**), you should return the same word back.
- When the query matches a word up to capitlization, you should return the first such match in the wordlist.
- When the query matches a word up to vowel errors, you should return the first such match in the wordlist.
- If the query has no matches in the wordlist, you should return the empty string.

Given some `queries`, return a list of words `answer`, where `answer[i]` is the correct word for `query = queries[i]`.

**Example 1:**

**Input:** wordlist = ["KiTe","kite","hare","Hare"], queries = ["kite","Kite","KiTe","Hare","HARE","Hear","hear","keti","keet","keto"]
**Output:** ["kite","KiTe","KiTe","Hare","hare","","","KiTe","","KiTe"]

**Example 2:**

**Input:** wordlist = ["yellow"], queries = ["YellOw"]
**Output:** ["yellow"]

**Constraints:**

- `1 <= wordlist.length, queries.length <= 5000`
- `1 <= wordlist[i].length, queries[i].length <= 7`
- `wordlist[i]` and `queries[i]` consist only of only English letters.

```java
class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        HashSet<String> hs = new HashSet<>();
        for (String s : wordlist) {
            hs.add(s);
        }

        List<String> res = new ArrayList<>();
        for (String s : queries) {
            if (hs.contains(s)) {
                res.add(s);
            } else {
                String word = findCaseInsensitive(wordlist, s);
                if (word.length() == 0) {
                    word = findVowelInsensitive(wordlist, s);
                }
                res.add(word);
            }
        }
        String[] ans = new String[res.size()];
        for (int i = 0; i < res.size(); i++) {
            ans[i] = res.get(i);
        }
        return ans;
    }
    private String findCaseInsensitive(String[] wordlist, String s) {
        for (String word : wordlist) {
            if (compareStringCase(word, s)) return word;
        }
        return "";
    }
    private boolean compareStringCase(String s1, String s2) {
        if (s1.length() != s2.length()) return false;
        for (int i = 0; i < s1.length(); i++) {
            char a = s1.charAt(i);
            char b = s2.charAt(i);
            if (a < 97) a += 32;
            if (b < 97) b += 32;
            if (a != b) return false;
        }
        return true;
    }
    private String findVowelInsensitive(String[] wordlist, String s) {
        for (String word : wordlist) {
            if (compareStringVowel(word, s)) return word;
        }
        return "";
    }
    private boolean compareStringVowel(String s1, String s2) {
        if (s1.length() != s2.length()) return false;
        for (int i = 0; i < s1.length(); i++) {
            char a = s1.charAt(i);
            char b = s2.charAt(i);
            if (a < 97) a += 32;
            if (b < 97) b += 32;
            if(isVowel(a) && isVowel(b)) continue;
            if (a != b) return false;
        }
        return true;
    }
    private boolean isVowel(char c) {
        switch(c) {
            case 'a', 'e', 'i', 'o', 'u' :
                return true;
        }
        return false;
    }
}
```

```java
class Solution {
    public String[] spellchecker(String[] wordlist, String[] queries) {
        HashSet<String> exactWord = new HashSet<>();
        HashMap<String, Integer> lowerCase = new HashMap<>();
        HashMap<String, Integer> vowelInsensitive = new HashMap<>();
        for (int i = 0; i < wordlist.length; i++) {
            String s1 = wordlist[i];
            exactWord.add(s1);
            String s2 = s1.toLowerCase();
            if(!lowerCase.containsKey(s2)) lowerCase.put(s2, i);
            String s3 = replaceVowels(s2);
            if(!vowelInsensitive.containsKey(s3)) vowelInsensitive.put(s3, i);
        }

        List<String> res = new ArrayList<>();
        for (String s : queries) {
            if (exactWord.contains(s)) {
                res.add(s);
            }
            else if (lowerCase.containsKey(s = s.toLowerCase())) {
                res.add(wordlist[lowerCase.get(s)]);
            }
            else if (vowelInsensitive.containsKey(s = replaceVowels(s))) {
                res.add(wordlist[vowelInsensitive.get(s)]);
            } 
            else res.add("");
        }
        String[] ans = new String[res.size()];
        for (int i = 0; i < res.size(); i++) {
            ans[i] = res.get(i);
        }
        return ans;
    }

        }
```

```java
class Solution {
    private HashSet<String> exactWord;
    private HashMap<String, Integer> lowerCase;
    private HashMap<String, Integer> vowelInsensitive;
    public String[] spellchecker(String[] wordlist, String[] queries) {
        exactWord = new HashSet<>();
        lowerCase = new HashMap<>();
        vowelInsensitive = new HashMap<>();
        for (int i = 0; i < wordlist.length; i++) {
            String s1 = wordlist[i];
            exactWord.add(s1);
            String s2 = s1.toLowerCase();
            if(!lowerCase.containsKey(s2)) lowerCase.put(s2, i);
            String s3 = replaceVowels(s2);
            if(!vowelInsensitive.containsKey(s3)) vowelInsensitive.put(s3, i);
        }

        String[] ans = new String[queries.length];
        int i = 0;
        for (String s : queries) {
            ans[i++] = getResultantString(s, wordlist);
        }
        
        return ans;
    }
    private String replaceVowels(String s) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < s.length(); i++){
            char c = s.charAt(i);
            if(isVowel(c))sb.append('#');
            else sb.append(c);
        }
        return sb.toString();
    }
    private String getResultantString(String s, String[] wordlist) {
        if (exactWord.contains(s)) {
            return s;
        }
        else if (lowerCase.containsKey(s = s.toLowerCase())) {
            return wordlist[lowerCase.get(s)];
        }
        else if (vowelInsensitive.containsKey(s = replaceVowels(s))) {
            return wordlist[vowelInsensitive.get(s)];
        } 
        return "";
    }
    private boolean isVowel(char c) {
        switch(c) {
            case 'a', 'e', 'i', 'o', 'u' :
                return true;
        }
        return false;
    }
}
```

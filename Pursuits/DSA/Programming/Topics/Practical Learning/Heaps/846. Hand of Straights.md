[846. Hand of Straights](https://leetcode.com/problems/hand-of-straights/)
Alice has some number of cards and she wants to rearrange the cards into groups so that each group is of size `groupSize`, and consists of `groupSize` consecutive cards.

Given an integer array `hand` where `hand[i]` is the value written on the `ith` card and an integer `groupSize`, return `true` if she can rearrange the cards, or `false` otherwise.

**Example 1:**

**Input:** hand = [1,2,3,6,2,3,4,7,8], groupSize = 3
**Output:** true
**Explanation:** Alice's hand can be rearranged as [1,2,3],[2,3,4],[6,7,8]

**Example 2:**

**Input:** hand = [1,2,3,4,5], groupSize = 4
**Output:** false
**Explanation:** Alice's hand can not be rearranged into groups of 4.

**Constraints:**

- `1 <= hand.length <= 104`
- `0 <= hand[i] <= 109`
- `1 <= groupSize <= hand.length`

# Better
Using sorting , priority queue, stack 
```java
class Solution {
    class Card{
        int val,freq;
        Card(int val,int freq){
            this.val = val;
            this.freq = freq;
        }
    }
    public boolean isNStraightHand(int[] hand, int groupSize) {
        if(groupSize == 1)return true;
        int n = hand.length;
        if(n%groupSize != 0)return false;
        Arrays.sort(hand);
        PriorityQueue<Card> minHeap = new PriorityQueue<>((a,b)->a.val-b.val);
        for(int i=n-1;i>=0;i--){
            int cardVal = hand[i];
            if(!minHeap.isEmpty() && minHeap.peek().val == cardVal){
                Card root = minHeap.poll();
                root.freq++;
                minHeap.offer(root);
            }
            else minHeap.offer(new Card(cardVal,1));
        }
        Stack<Card> group = new Stack<>();
        while(!minHeap.isEmpty()){
            Card root = minHeap.poll();
            group.push(root);
            int consecutiveCnt=1;
            while(!minHeap.isEmpty() && minHeap.peek().val == group.peek().val+1){
                group.push(minHeap.poll());
                consecutiveCnt++;
                if(consecutiveCnt == groupSize)break;
            }
            if(groupSize != consecutiveCnt)return false;
            while(!group.isEmpty()){
                Card top = group.pop();
                top.freq--;
                if(top.freq>=1)minHeap.offer(top);
            }
        }
        return true;
    }
}
```

```java
class Solution {
    public boolean isNStraightHand(int[] hand, int groupSize) {
        int n = hand.length;
        if(n%groupSize!=0)return false;
        if(groupSize==1)return true;
        Arrays.sort(hand);
        PriorityQueue<Element> minHeap = new PriorityQueue<>((a,b)->a.ele-b.ele);
        for(int i=n-1;i>=0;i--){
            int ele = hand[i];
            if(!minHeap.isEmpty() && ele == minHeap.peek().ele){
                Element e =minHeap.poll();
                e.cnt++;
                minHeap.add(e);
            }
            else minHeap.add(new Element(ele,1));
        }
        int k=1;
        Element st[] = new Element[hand.length];
        int p=0;
        Element prev;
        while(!minHeap.isEmpty()){
            prev=minHeap.poll();
            st[p] = prev;
            k=1;
            p++;
            while(!minHeap.isEmpty() && minHeap.peek().ele == prev.ele+1){
                k++;
                st[p] = minHeap.poll();
                prev = st[p];
                p++;
                if(k==groupSize)break;
            }
            if(k!=groupSize)return false;
            while(p>0){
                p--;
                Element e = st[p];
                e.cnt--;
                if(e.cnt==0)continue;
                minHeap.add(e);
            }
        }
        return true;

    }
    class Element{
        int ele;
        int cnt;
        Element(int a,int b){
            ele = a;
            cnt = b;
        }
    }
}
```

```java
class Solution {
    public boolean isNStraightHand(int[] hand, int groupSize) {
        if(groupSize == 1)return true;
        int n = hand.length;
        if(groupSize>n)return false;
        HashMap<Integer,Integer> hm = new HashMap<>();

        for(int ele:hand){
            hm.put(ele,hm.getOrDefault(ele,0)+1);
        }
        for(int i=0;i<n;i++){
            int ele = hand[i];
            if(hm.containsKey(ele)){
                for(int j=0;j<groupSize;j++){
                    if(hm.containsKey(ele)){
                        int cnt = hm.get(ele);
                        if(cnt==1)hm.remove(ele);
                        else hm.put(ele,cnt-1);
                        ele++;
                    }
                    else return false;
                }
            }
        }
        return true;
    }
}
```

```java
class Solution {
    public boolean isNStraightHand(int[] hand, int groupSize) {
        if(groupSize == 1)return true;
        int n = hand.length;
        if(groupSize>n)return false;
        int cnt=0;
        Arrays.sort(hand);
        for(int i=0;i<n;i++){
            if(hand[i]==-1)continue;
            int nxtEle = hand[i] + 1;
            hand[i] = -1;
            cnt=1;
            int j=i+1;
            while(j<n && cnt<groupSize){
                if(hand[j]==nxtEle){
                    nxtEle++;
                    hand[j] = -1;
                    cnt++;
                }
                j++;
            }
            if(cnt!=groupSize)return false;
        }
        return true;
    }
}
```

[355. Design Twitter](https://leetcode.com/problems/design-twitter/)
Design a simplified version of Twitter where users can post tweets, follow/unfollow another user, and is able to see the `10` most recent tweets in the user's news feed.

Implement the `Twitter` class:

- `Twitter()` Initializes your twitter object.
- `void postTweet(int userId, int tweetId)` Composes a new tweet with ID `tweetId` by the user `userId`. Each call to this function will be made with a unique `tweetId`.
- `List<Integer> getNewsFeed(int userId)` Retrieves the `10` most recent tweet IDs in the user's news feed. Each item in the news feed must be posted by users who the user followed or by the user themself. Tweets must be **ordered from most recent to least recent**.
- `void follow(int followerId, int followeeId)` The user with ID `followerId` started following the user with ID `followeeId`.
- `void unfollow(int followerId, int followeeId)` The user with ID `followerId` started unfollowing the user with ID `followeeId`.

**Example 1:**

**Input**
["Twitter", "postTweet", "getNewsFeed", "follow", "postTweet", "getNewsFeed", "unfollow", "getNewsFeed"]
[[], [1, 5], [1], [1, 2], [2, 6], [1], [1, 2], [1]]
**Output**
[null, null, [5], null, null, [6, 5], null, [5]]

**Explanation**
Twitter twitter = new Twitter();
twitter.postTweet(1, 5); // User 1 posts a new tweet (id = 5).
twitter.getNewsFeed(1);  // User 1's news feed should return a list with 1 tweet id -> [5]. return [5]
twitter.follow(1, 2);    // User 1 follows user 2.
twitter.postTweet(2, 6); // User 2 posts a new tweet (id = 6).
twitter.getNewsFeed(1);  // User 1's news feed should return a list with 2 tweet ids -> [6, 5]. Tweet id 6 should precede tweet id 5 because it is posted after tweet id 5.
twitter.unfollow(1, 2);  // User 1 unfollows user 2.
twitter.getNewsFeed(1);  // User 1's news feed should return a list with 1 tweet id -> [5], since user 1 is no longer following user 2.

**Constraints:**

- `1 <= userId, followerId, followeeId <= 500`
- `0 <= tweetId <= 104`
- All the tweets have **unique** IDs.
- At most `3 * 104` calls will be made to `postTweet`, `getNewsFeed`, `follow`, and `unfollow`.
- A user cannot follow himself.

# Without Optimizations

```java
class Twitter {
    class Tweet implements Comparable<Tweet>{
        int tweetId;
        int timeCounter;
        Tweet(int tweetId,int timeCounter){
            this.tweetId = tweetId;
            this.timeCounter = timeCounter;
        }
        public int compareTo(Tweet that){
            return that.timeCounter - this.timeCounter;
        }
    }
    class User{
        int userId;
        List<Tweet> tweets;
        HashSet<Integer> followee;
        User(int userId){
            this.userId = userId;
            tweets = new LinkedList<>();
            followee = new HashSet<>();
        }
        public void postTweet(int tweetId,int timeCounter){
            Tweet newTweet = new Tweet(tweetId,timeCounter);
            tweets.addFirst(newTweet);
        }
        public void addFollowee(int followeeId){
            followee.add(followeeId);
        }
        public void removeFollowee(int followeeId){
            followee.remove(followeeId);
        }
    }
    HashMap<Integer,User> records;
    int timeCounter;
    public Twitter() {
        records = new HashMap<>();
        timeCounter=0;
    }
    
    public void postTweet(int userId, int tweetId) {
        timeCounter++;
        if(!records.containsKey(userId)){
            records.put(userId,new User(userId));
        }
        User curUser = records.get(userId);
        curUser.postTweet(tweetId,timeCounter);
    }
    
    public List<Integer> getNewsFeed(int userId) {
        List<Integer> recentTweets = new ArrayList<>();
        if(!records.containsKey(userId))return recentTweets;
        User user = records.get(userId);
        PriorityQueue<Tweet> maxHeap = new PriorityQueue<>();
        for(Tweet userTweet:user.tweets){
            maxHeap.offer(userTweet);
        }
        for(int userFolloweeId:user.followee){
            User userFollowee = records.get(userFolloweeId);
            for(Tweet followeeTweet:userFollowee.tweets){
                maxHeap.offer(followeeTweet);
            }
        }
        int cnt=1;
        while(!maxHeap.isEmpty() && cnt<=10){
            recentTweets.add(maxHeap.poll().tweetId);
            cnt++;
        }
        return recentTweets;
    }
    
    public void follow(int followerId, int followeeId) {
        if(!records.containsKey(followerId)){
            records.put(followerId,new User(followerId));
        }
        if(!records.containsKey(followeeId)){
            records.put(followeeId,new User(followeeId));
        }
        User follower = records.get(followerId);
        follower.addFollowee(followeeId);
    }
    
    public void unfollow(int followerId, int followeeId) {
        if(!records.containsKey(followerId)){
            records.put(followerId,new User(followerId));
        }
        if(!records.containsKey(followeeId)){
            records.put(followeeId,new User(followeeId));
        }
        User follower = records.get(followerId);
        follower.removeFollowee(followeeId);
    }
}
```

```java
class Twitter {
    class Tweet implements Comparable<Tweet>{
        int tweetId;
        int timeCounter;
        Tweet(int tweetId,int timeCounter){
            this.tweetId = tweetId;
            this.timeCounter = timeCounter;
        }
        public int compareTo(Tweet that){
            return that.timeCounter - this.timeCounter;
        }
    }
    class User{
        int userId;
        List<Tweet> tweets;
        HashSet<Integer> followee;
        User(int userId){
            this.userId = userId;
            tweets = new LinkedList<>();
            followee = new HashSet<>();
            followee.add(userId);
        }
        public void postTweet(int tweetId,int timeCounter){
            Tweet newTweet = new Tweet(tweetId,timeCounter);
            tweets.addFirst(newTweet);
        }
        public void addFollowee(int followeeId){
            followee.add(followeeId);
        }
        public void removeFollowee(int followeeId){
            followee.remove(followeeId);
        }
    }
    HashMap<Integer,User> records;
    int timeCounter;
    public Twitter() {
        records = new HashMap<>();
        timeCounter=0;
    }
    
    public void postTweet(int userId, int tweetId) {
        timeCounter++;
        if(!records.containsKey(userId)){
            records.put(userId,new User(userId));
        }
        User curUser = records.get(userId);
        curUser.postTweet(tweetId,timeCounter);
    }
    
    public List<Integer> getNewsFeed(int userId) {
        List<Integer> recentTweets = new ArrayList<>();
        if(!records.containsKey(userId))return recentTweets;
        User user = records.get(userId);
        PriorityQueue<Tweet> maxHeap = new PriorityQueue<>();
        for(int userFolloweeId:user.followee){
            User userFollowee = records.get(userFolloweeId);
            int cnt=1;
            for(Tweet followeeTweet:userFollowee.tweets){
                cnt++;
                maxHeap.offer(followeeTweet);
                if(cnt>10)break;
            }
        }
        int cnt=1;
        while(!maxHeap.isEmpty() && cnt<=10){
            recentTweets.add(maxHeap.poll().tweetId);
            cnt++;
        }
        return recentTweets;
    }
    
    public void follow(int followerId, int followeeId) {
        if(!records.containsKey(followerId)){
            records.put(followerId,new User(followerId));
        }
        if(!records.containsKey(followeeId)){
            records.put(followeeId,new User(followeeId));
        }
        User follower = records.get(followerId);
        follower.addFollowee(followeeId);
    }
    
    public void unfollow(int followerId, int followeeId) {
        if(!records.containsKey(followerId)){
            records.put(followerId,new User(followerId));
        }
        if(!records.containsKey(followeeId)){
            records.put(followeeId,new User(followeeId));
        }
        User follower = records.get(followerId);
        follower.removeFollowee(followeeId);
    }
}
```

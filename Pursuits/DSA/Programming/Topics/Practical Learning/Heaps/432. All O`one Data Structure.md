[432. All O`one Data Structure](https://leetcode.com/problems/all-oone-data-structure/)

Design a data structure to store the strings' count with the ability to return the strings with minimum and maximum counts.

Implement the `AllOne` class:

- `AllOne()` Initializes the object of the data structure.
- `inc(String key)` Increments the count of the string `key` by `1`. If `key` does not exist in the data structure, insert it with count `1`.
- `dec(String key)` Decrements the count of the string `key` by `1`. If the count of `key` is `0` after the decrement, remove it from the data structure. It is guaranteed that `key` exists in the data structure before the decrement.
- `getMaxKey()` Returns one of the keys with the maximal count. If no element exists, return an empty string `""`.
- `getMinKey()` Returns one of the keys with the minimum count. If no element exists, return an empty string `""`.

**Note** that each function must run in `O(1)` average time complexity.

**Example 1:**

**Input**
["AllOne", "inc", "inc", "getMaxKey", "getMinKey", "inc", "getMaxKey", "getMinKey"]
[[], ["hello"], ["hello"], [], [], ["leet"], [], []]
**Output**
[null, null, null, "hello", "hello", null, "hello", "leet"]

**Explanation**
AllOne allOne = new AllOne();
allOne.inc("hello");
allOne.inc("hello");
allOne.getMaxKey(); // return "hello"
allOne.getMinKey(); // return "hello"
allOne.inc("leet");
allOne.getMaxKey(); // return "hello"
allOne.getMinKey(); // return "leet"

**Constraints:**

- `1 <= key.length <= 10`
- `key` consists of lowercase English letters.
- It is guaranteed that for each call to `dec`, `key` is existing in the data structure.
- At most `5 * 104` calls will be made to `inc`, `dec`, `getMaxKey`, and `getMinKey`.


# Using HashMap

```java
class AllOne {
    HashMap<String,Integer> store;
    public AllOne() {
        store = new HashMap<>();
    }
    
    public void inc(String key) {
        int val = 1;
        if(store.containsKey(key)){
            val = store.get(key) + 1;
        }
        store.put(key, val);
    }
    
    public void dec(String key) {
        int val = store.get(key);
        val--;
        if(val == 0) {
            store.remove(key);
        } else {
            store.put(key, val);
        }
    }
    
    public String getMaxKey() {
        String res = "";
        int max = 0;
        for (Map.Entry<String,Integer> e:store.entrySet()) {
            int cnt = e.getValue();
            if(cnt > max) {
                max = cnt;
                res = e.getKey();
            }
        }
        return res;
    }
    
    public String getMinKey() {
        String res = "";
        int min = Integer.MAX_VALUE;
        for (Map.Entry<String,Integer> e:store.entrySet()) {
            int cnt = e.getValue();
            if(cnt < min) {
                min = cnt;
                res = e.getKey();
            }
        }
        return res;
    }
}
```

# Using PriorityQueue

```java
class AllOne {
    HashMap <String, Integer> store;
    PriorityQueue<String> minHeap;
    PriorityQueue<String> maxHeap;
    public AllOne() {
        store = new HashMap<>();
        minHeap = new PriorityQueue<>((a, b) -> store.get(a) - store.get(b));
        maxHeap = new PriorityQueue<>((a, b) -> store.get(b) - store.get(a));
    }
    
    public void inc(String key) {
        int cnt = 1;
        if (store.containsKey(key)) {
            cnt = store.get(key) + 1;
            minHeap.remove(key);
            maxHeap.remove(key);
        }
        store.put(key, cnt);
        minHeap.offer(key);
        maxHeap.offer(key);
    }
    
    public void dec(String key) {
        int cnt = store.get(key);
        cnt--;
        minHeap.remove(key);
        maxHeap.remove(key);
        if (cnt == 0) {
            store.remove(key);
            return;
        }
        store.put(key, cnt);
        minHeap.offer(key);
        maxHeap.offer(key);
    }
    
    public String getMaxKey() {
        if(maxHeap.isEmpty())return "";
        return maxHeap.peek();
    }
    
    public String getMinKey() {
        if(minHeap.isEmpty())return "";
        return minHeap.peek();
    }
}
```

```java
class AllOne {
    class Node {
        Node prev, next;
        int freq;
        HashSet<String> keys = new HashSet<>();
        Node(int freq) { this.freq = freq; }
    }

    Node head, tail;
    Map<String, Node> keyNode;

    public AllOne() {
        head = new Node(-1);
        tail = new Node(-1);
        head.next = tail;
        tail.prev = head;
        keyNode = new HashMap<>();
    }

    private Node insertAfter(Node prev, int freq) {
        Node cur = new Node(freq);
        cur.prev = prev;
        cur.next = prev.next;
        prev.next.prev = cur;
        prev.next = cur;
        return cur;
    }

    private void remove(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    public void inc(String key) {
        Node cur = keyNode.get(key);
        if (cur == null) {
            // new key, put in freq=1 bucket
            if (head.next.freq != 1) cur = insertAfter(head, 1);
            else cur = head.next;
            cur.keys.add(key);
            keyNode.put(key, cur);
        } else {
            // move key to next freq bucket
            int newFreq = cur.freq + 1;
            Node next = cur.next;
            if (next == tail || next.freq != newFreq) next = insertAfter(cur, newFreq);
            next.keys.add(key);
            keyNode.put(key, next);

            // cleanup old bucket
            cur.keys.remove(key);
            if (cur.keys.isEmpty()) remove(cur);
        }
    }

    public void dec(String key) {
        Node cur = keyNode.get(key);
        if (cur == null) return;

        if (cur.freq == 1) {
            // remove completely
            keyNode.remove(key);
            cur.keys.remove(key);
            if (cur.keys.isEmpty()) remove(cur);
        } else {
            // move key to prev freq bucket

            prev.keys.add(key);
            keyNode.put(key, prev);

            // cleanup old bucket
            cur.keys.remove(key);
            if (cur.keys.isEmpty()) remove(cur);
        }
    }
```
